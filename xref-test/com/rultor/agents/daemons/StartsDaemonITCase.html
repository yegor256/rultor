58  /**
59   * Integration test for ${@link StartsDaemon}.
60   *
61   * @author Yegor Bugayenko (yegor@teamed.io)
62   * @version $Id: 504b49b3e67abd0b281e3e16bcbb7c391f65a4af $
63   * @since 1.3.8
64   * @checkstyle MultipleStringLiteralsCheck (500 lines)
65   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
66   */
67  public final class StartsDaemonITCase {
68  
69      /**
70       * Temp directory.
71       * @checkstyle VisibilityModifierCheck (5 lines)
72       */
73      @Rule
74      public final transient TemporaryFolder temp = new TemporaryFolder();
75  
76      /**
77       * StartsDaemon can start a daemon.
78       * @throws Exception In case of error.
79       * @checkstyle ExecutableStatementCountCheck (50 lines)
80       */
81      @Test
82      public void startsDaemon() throws Exception {
83          Assume.assumeFalse(SystemUtils.IS_OS_WINDOWS);
84          final SSHD sshd = new SSHD(this.temp.newFolder());
85          final int port = sshd.port();
86          final Talk talk = new Talk.InFile();
87          final String executor;
88          if (SystemUtils.IS_OS_LINUX) {
89              executor = "md5sum";
90          } else {
91              executor = "md5";
92          }
93          talk.modify(
94              new Directives().xpath("/talk")
95                  .add("shell").attr("id", "abcdef")
96                  .add("host").set("localhost").up()
97                  .add("port").set(Integer.toString(port)).up()
98                  .add("login").set(sshd.login()).up()
99                  .add("key").set(sshd.key()).up().up()
100                 .add("daemon").attr("id", "fedcba")
101                 .add("title").set("some operation").up()
102                 .add("script").set(
103                     String.format("ls -al; %s file.bin; sleep 50000", executor)
104                 )
105         );
106         final Profile profile = Mockito.mock(Profile.class);
107         Mockito.doReturn(new XMLDocument("<p/>")).when(profile).read();
108         Mockito.doReturn(
109             new ArrayMap<String, InputStream>().with(
110                 "file.bin",
111                 new ByteArrayInputStream(
112                     // @checkstyle MagicNumber (1 line)
113                     new byte[] {0, 1, 7, 8, 9, 10, 13, 20}
114                 )
115             )
116         ).when(profile).assets();
117         final Agent agent = new StartsDaemon(profile);
118         agent.execute(talk);
119         MatcherAssert.assertThat(
120             talk.read(),
121             XhtmlMatchers.hasXPaths(
122                 "/talk/daemon[started and dir]",
123                 "/talk/daemon[ends-with(started,'Z')]"
124             )
125         );
126         final String dir = talk.read().xpath("/talk/daemon/dir/text()").get(0);
127         final ByteArrayOutputStream baos = new ByteArrayOutputStream();
128         TimeUnit.SECONDS.sleep(2L);
129         new Shell.Safe(new TalkShells(talk.read()).get()).exec(
130             String.format("cat %s/stdout", dir),
131             new NullInputStream(0L),
132             baos, baos
133         );
134         MatcherAssert.assertThat(
135             baos.toString(CharEncoding.UTF_8),
136             Matchers.allOf(
137                 Matchers.containsString("+ set -o pipefail"),
138                 Matchers.containsString("+ date"),
139                 Matchers.containsString("+ ls -al"),
140                 Matchers.containsString("182f61268e6e6e6cd1a547be31fd8583")
141             )
142         );
143         MatcherAssert.assertThat(
144             new File(dir, "status").exists(), Matchers.is(false)
145         );
146     }
