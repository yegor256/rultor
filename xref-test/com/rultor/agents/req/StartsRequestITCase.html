67      /**
68       * StartsRequest can start a request.
69       * @throws Exception In case of error.
70       */
71      @Test
72      public void startsDeployRequest() throws Exception {
73          final File repo = this.repo();
74          final Agent agent = new StartsRequest(
75              new Profile.Fixed(
76                  new XMLDocument(
77                      "<p><deploy><script>echo HEY</script></deploy></p>"
78                  )
79              )
80          );
81          final Talk talk = new Talk.InFile();
82          talk.modify(
83              new Directives().xpath("/talk")
84                  .add("request").attr("id", "abcd")
85                  .add("type").set("deploy").up()
86                  .add("args")
87                  .add("arg").attr("name", "head").set(repo.toString()).up()
88                  .add("arg").attr("name", "head_branch").set("master").up()
89          );
90          agent.execute(talk);
91          this.exec(talk);
92      }

151     /**
152      * Execute script from daemon.
153      * @param talk Talk to use
154      * @throws IOException If fails
155      */
156     private void exec(final Talk talk) throws IOException {
157         final String script = StringUtils.join(
158             "set -x\n",
159             "set -e\n",
160             "set -o pipefail\n",
161             "sudo=echo\n",
162             String.format("cd %s\n", this.temp.newFolder()),
163             talk.read().xpath("/talk/daemon/script/text()").get(0)
164         );
165         final Sshd sshd = new Sshd(this.temp.newFolder());
166         final int port = sshd.start();
167         new Shell.Empty(
168             new Shell.Safe(
169                 new SSH("localhost", port, sshd.login(), sshd.key())
170             )
171         ).exec(script);
172     }
173 
174     /**
175      * Create empty Git repo.
176      * @return Its location
177      * @throws IOException If fails
178      */
179     private File repo() throws IOException {
180         final File repo = this.temp.newFolder();
181         new VerboseProcess(
182             new ProcessBuilder().command(
183                 "/bin/bash",
184                 "-c",
185                 StringUtils.join(
186                     "set -x; set -e; set -o pipefail;",
187                     "git init .;",
188                     "git config user.email test@rultor.com;",
189                     "git config user.name test;",
190                     "echo 'hello, world!' > hello.txt;",
191                     "git add .;",
192                     "git commit -am 'first file';",
193                     "git checkout -b frk;",
194                     "echo 'good bye!' > hello.txt;",
195                     "git commit -am 'modified file';",
196                     "git checkout master;",
197                     "git config receive.denyCurrentBranch ignore"
198                 )
199             ).directory(repo)
200         ).stdout();
201         return repo;
202     }
