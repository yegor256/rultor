58      /**
59       * Mails can send a mail.
60       * @throws Exception In case of error.
61       */
62      @Test
63      @Ignore
64      public void sendsMail() throws Exception {
65          final Postman postman = Mockito.spy(Postman.CONSOLE);
66          final Agent agent = new Mails(
67              this.profile(),
68              postman
69          );
70          agent.execute(MailsTest.talk());
71          final ArgumentCaptor<Envelope> captor =
72              ArgumentCaptor.forClass(Envelope.class);
73          Mockito.verify(postman).send(captor.capture());
74          final Envelope envelope = captor.getValue();
75          MatcherAssert.assertThat(
76              envelope.unwrap().getContent().toString(),
77              Matchers.allOf(
78                  Matchers.containsString("See #456, release log:"),
79                  Matchers.containsString("Released by Rultor"),
80                  Matchers.containsString(Manifests.read("Rultor-Version")),
81                  Matchers.containsString(
82                      "see [build log](http://www.rultor.com/t/123-abcdef)"
83                  )
84              )
85          );
86          MatcherAssert.assertThat(
87              envelope.unwrap().getSubject(),
88              Matchers.equalTo("user/repo v2.0 released!")
89          );
90      }

104     /**
105      * Profile for test.
106      * @return Profile
107      */
108     private Profile.Fixed profile() {
109         return new Profile.Fixed(
110             new XMLDocument(
111                 StringUtils.join(
112                     "<p><entry key='release'>",
113                     "<entry key='email'>",
114                     "<item>test1@localhost</item>",
115                     "<item>test2@localhost</item>",
116                     "</entry>",
117                     "</entry></p>"
118                 )
119             )
120         );
121     }

123     /**
124      * Make a talk with this tag.
125      * @return Talk
126      * @throws java.io.IOException If fails
127      */
128     private static Talk talk() throws IOException {
129         final Talk talk = new Talk.InFile();
130         talk.modify(
131             new Directives().xpath("/talk")
132                 .attr("number", "123")
133                 .add("wire")
134                 .add("github-issue").set("456")
135                 .add("github-repo").set("user/repo")
136                 .up()
137                 .up()
138                 .add("request").attr("id", "abcdef")
139                 .add("type").set("release").up()
140                 .add("success").set("true").up()
141                 .add("args").add("arg").attr("name", "tag").set("v2.0")
142         );
143         return talk;
144     }
