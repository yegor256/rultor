30  package com.rultor.agents.github.qtn;
31  
32  import com.jcabi.github.Repo;
33  import com.jcabi.github.mock.MkGithub;
34  import org.hamcrest.MatcherAssert;
35  import org.hamcrest.Matchers;
36  import org.junit.Test;
37  
38  /**
39   * Tests for ${@link ReleaseTag}.
40   * @author Armin Braun (me@obrown.io)
41   * @version $Id: 127288b8fe4839c4ce44785081e7d04e9efff947 $
42   * @since 1.62
43   */
44  public final class ReleaseTagTest {
45  
46      /**
47       * ReleaseTag can deny release for outdated, semantically correct versions.
48       * It does however allow any version number, if it contains anything
49       * other than digits and dots.
50       * @throws Exception In case of error
51       */
52      @Test
53      public void validatesReleaseVersion() throws Exception {
54          final Repo repo = new MkGithub().randomRepo();
55          repo.releases().create("1.74");
56          MatcherAssert.assertThat(
57              new ReleaseTag(repo, "1.87.15").allowed(),
58              Matchers.is(true)
59          );
60          MatcherAssert.assertThat(
61              new ReleaseTag(repo, "1.5-bar").allowed(),
62              Matchers.is(true)
63          );
64          MatcherAssert.assertThat(
65              new ReleaseTag(repo, "1.9-beta").allowed(),
66              Matchers.is(true)
67          );
68          MatcherAssert.assertThat(
69              new ReleaseTag(repo, "1.62").allowed(),
70              Matchers.is(false)
71          );
72      }
73  
74      /**
75       * ReleaseTag can retrieve the latest release version in the repo.
76       * @throws Exception In case of error
77       */
78      @Test
79      public void getsReferenceVersion() throws Exception {
80          final Repo repo = new MkGithub().randomRepo();
81          final String latest = "2.2.1";
82          repo.releases().create("1.0");
83          repo.releases().create(latest);
84          repo.releases().create("3.0-beta");
85          MatcherAssert.assertThat(
86              new ReleaseTag(repo, "2.4").reference(),
87              Matchers.is(latest)
88          );
89      }
90  }
