33  import com.jcabi.github.Issue;
34  import com.jcabi.github.Repo;

36  import com.jcabi.matchers.XhtmlMatchers;
37  import java.net.URI;
38  import org.hamcrest.MatcherAssert;
39  import org.junit.Test;
40  import org.xembly.Directives;
41  import org.xembly.Xembler;
42  
43  /**
44   * Tests for {@link QnStop}.
45   *
46   * @author Krzysztof Krason (Krzysztof.Krason@gmail.com)
47   * @version $Id: 0968d54d482e6d47a843a2758a1e3401351bc8d3 $
48   */
49  public final class QnStopTest {
50      /**
51       * QnMerge can build a request.
52       * @throws Exception In case of error.
53       */
54      @Test
55      public void buildsRequest() throws Exception {
56          final Repo repo = new MkGithub().randomRepo();
57          final Issue issue = repo.issues().create("", "");
58          issue.comments().post("stop");
59          MatcherAssert.assertThat(
60              new Xembler(
61                  new Directives().add("request").append(
62                      new QnStop().understand(
63                          new Comment.Smart(issue.comments().get(1)), new URI("#")
64                      ).dirs()
65                  )
66              ).xml(),
67              XhtmlMatchers.hasXPaths(
68                  "/request[type='stop']",
69                  "/request/args[count(arg) = 0]"
70              )
71          );
72      }
73  }
