51  /**
52   * Test case for {@link SQSQueue}.
53   * @author Yegor Bugayenko (yegor@tpc2.com)
54   * @version $Id$
55   * @checkstyle ClassDataAbstractionCoupling (500 lines)
56   */
57  public final class SQSQueueTest {
58  
59      /**
60       * SQSQueue can save JSON.
61       * @throws Exception If some problem inside
62       */
63      @Test
64      public void savesJson() throws Exception {
65          final Coordinates work = new Coordinates.Simple(
66              new URN("urn:facebook:1"),
67              "some-test-rule",
68              new Time(Tv.MILLION)
69          );
70          final SQSClient client = Mockito.mock(SQSClient.class);
71          final AmazonSQS aws = Mockito.mock(AmazonSQS.class);
72          Mockito.doReturn(aws).when(client).get();
73          Mockito.doReturn(
74              new SendMessageResult().withMessageId("test")
75          ).when(aws).sendMessage(Mockito.any(SendMessageRequest.class));
76          Mockito.doReturn(new GetQueueAttributesResult())
77              .when(aws)
78              .getQueueAttributes(Mockito.any(GetQueueAttributesRequest.class));
79          final SQSQueue queue = new SQSQueue(client);
80          queue.push(work);
81          Mockito.verify(aws).sendMessage(
82              Mockito.argThat(
83                  Matchers.<SendMessageRequest>hasProperty(
84                      "messageBody",
85                      Matchers.equalTo(
86                          // @checkstyle LineLength (1 line)
87                          "{\"urn\":\"urn:facebook:1\",\"scheduled\":\"1970-01-01T00:16:40Z\",\"rule\":\"some-test-rule\"}"
88                      )
89                  )
90              )
91          );
92      }
93  
94      /**
95       * SQSQueue can restore JSON.
96       * @throws Exception If some problem inside
97       */
98      @Test
99      public void restoresJson() throws Exception {
100         final SQSClient client = Mockito.mock(SQSClient.class);
101         final AmazonSQS aws = Mockito.mock(AmazonSQS.class);
102         Mockito.doReturn(aws).when(client).get();
103         Mockito.doReturn(new GetQueueAttributesResult())
104             .when(aws)
105             .getQueueAttributes(Mockito.any(GetQueueAttributesRequest.class));
106         final SQSQueue queue = new SQSQueue(client);
107         Mockito.doReturn(
108             new ReceiveMessageResult().withMessages(
109                 new Message().withBody(
110                     // @checkstyle LineLength (1 line)
111                     "{\"urn\":\"urn:facebook:65\",\"rule\":\"test-rule\",\"scheduled\":\"1970-01-01T00:16:40Z\"}"
112                 )
113             )
114         ).when(aws).receiveMessage(Mockito.any(ReceiveMessageRequest.class));
115         final Coordinates work = queue.pull(1, TimeUnit.SECONDS);
116         MatcherAssert.assertThat(
117             work.owner(), Matchers.equalTo(new URN("urn:facebook:65"))
118         );
119         MatcherAssert.assertThat(work.rule(), Matchers.equalTo("test-rule"));
120         MatcherAssert.assertThat(
121             work.scheduled().toString(),
122             Matchers.equalTo(new Time(Tv.MILLION).toString())
123         );
124     }
