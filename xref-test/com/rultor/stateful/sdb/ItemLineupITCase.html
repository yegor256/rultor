76      /**
77       * ItemLineup can run code in parallel.
78       * @throws Exception If some problem inside
79       */
80      @Test
81      public void runsInParallel() throws Exception {
82          final Lineup lineup = this.lineup("ItemLineupITCase.txt");
83          final int threads = 10;
84          final CountDownLatch start = new CountDownLatch(1);
85          final AtomicInteger count = new AtomicInteger();
86          final ExecutorService svc =
87              Executors.newFixedThreadPool(threads, new VerboseThreads());
88          final Random rnd = new SecureRandom();
89          final Callable<?> callable = new Callable<Integer>() {
90              @Override
91              public Integer call() throws Exception {
92                  start.await();
93                  return lineup.exec(
94                      new Callable<Integer>() {
95                          @Override
96                          public Integer call() throws Exception {
97                              final int num = count.get();
98                              TimeUnit.MILLISECONDS.sleep(rnd.nextInt(Tv.TEN));
99                              count.set(num + 1);
100                             return count.get();
101                         }
102                     }
103                 );
104             }
105         };
106         for (int thread = 0; thread < threads; ++thread) {
107             svc.submit(callable);
108         }
109         start.countDown();
110         svc.shutdown();
111         MatcherAssert.assertThat(
112             svc.awaitTermination(2, TimeUnit.MINUTES),
113             Matchers.is(true)
114         );
115         MatcherAssert.assertThat(count.get(), Matchers.equalTo(threads));
116     }
117 
118     /**
119      * Get lineup to work with.
120      * @param name Name of item
121      * @return Lineup
122      * @throws Exception If some problem inside
123      */
124     private Lineup lineup(final String name) throws Exception {
125         Assume.assumeNotNull(ItemLineupITCase.KEY);
126         return new ItemLineup(

128             name,
129             new SDBClient.Simple(

132                 ItemLineupITCase.DOMAIN
133             )
134         );
135     }
