97      /**
98       * MongoStand can accept and return pulses.
99       * @throws Exception If some problem inside
100      */
101     @Test
102     public void managesDataInMongoDb() throws Exception {
103         final Stand stand = this.stand();
104         final Coordinates pulse = this.pulse();
105         stand.post(pulse, 2, "ADD 'tags'; ADD 'tag'; ADD 'label'; SET 'foo';");
106         stand.post(pulse, 1, "ADD 'test'; SET 'hello, world!';");
107         final Iterator<Pulse> pulses = stand.pulses().iterator();
108         MatcherAssert.assertThat(pulses.hasNext(), Matchers.is(true));
109         MatcherAssert.assertThat(
110             pulses.next().xembly(),
111             Matchers.allOf(
112                 Matchers.containsString("ADD 'test';"),
113                 Matchers.containsString("ADD 'tags';")
114             )
115         );
116     }

118     /**
119      * MongoStand can update the same pulse concurrently.
120      * @throws Exception If some problem inside
121      */
122     @Test
123     @SuppressWarnings("PMD.DoNotUseThreads")
124     public void updatesSameStandConcurrently() throws Exception {
125         final Stand stand = this.stand();
126         final Coordinates pulse = this.pulse();
127         final CountDownLatch start = new CountDownLatch(1);
128         final AtomicLong nano = new AtomicLong(1);
129         final Runnable runnable = new VerboseRunnable(
130             new Callable<Void>() {
131                 @Override
132                 public Void call() throws Exception {
133                     start.await();
134                     final long inc = nano.getAndIncrement();
135                     stand.post(
136                         pulse, inc,
137                         new Directives()
138                             .xpath(String.format("//a%d", inc - 1))
139                             .strict(1)
140                             .add(String.format("a%d", inc))
141                             .toString()
142                     );
143                     return null;
144                 }
145             }
146         );
147         final int threads = 5;
148         final ExecutorService svc = Executors.newFixedThreadPool(
149             threads, new VerboseThreads()
150         );
151         for (int thread = 0; thread < threads; ++thread) {
152             svc.submit(runnable);
153         }
154         start.countDown();
155         svc.shutdown();
156         MatcherAssert.assertThat(
157             svc.awaitTermination(Tv.TEN, TimeUnit.SECONDS),
158             Matchers.is(true)
159         );
160         final Document dom = DocumentBuilderFactory.newInstance()
161             .newDocumentBuilder().newDocument();
162         dom.appendChild(dom.createElement("a0"));
163         new Snapshot(
164             stand.pulses().tail(pulse).iterator().next().xembly()
165         ).apply(dom);
166         MatcherAssert.assertThat(
167             XhtmlMatchers.xhtml(dom),
168             XhtmlMatchers.hasXPath("/a0/a1/a2/a3/a4")
169         );
170     }
