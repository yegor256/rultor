125     /**
126      * AwsStands can add stands and list them.
127      * @throws Exception If some problem inside
128      */
129     @Test
130     public void addsStandsAndListsThem() throws Exception {
131         final URN owner = new URN("urn:test:6");
132         final Stands stands = new AwsStands(this.region, owner);
133         final String name = "test-stand";
134         stands.create(name);
135         MatcherAssert.assertThat(stands.contains(name), Matchers.is(true));
136         MatcherAssert.assertThat(stands.contains("abc"), Matchers.is(false));
137         MatcherAssert.assertThat(
138             Iterables.filter(
139                 stands,
140                 new Predicate<Stand>() {
141                     @Override
142                     public boolean apply(final Stand stand) {
143                         return stand.name().equals(name)
144                             && stand.owner().equals(owner);
145                     }
146                 }
147             ),
148             Matchers.<Stand>iterableWithSize(1)
149         );
150         final Stand stand = stands.get(name);
151         MatcherAssert.assertThat(stand.owner(), Matchers.equalTo(owner));
152     }
