65  /**
66   * Test case for {@link IndexRs}.

74      /**
75       * IndexRs can render front page.
76       * @throws Exception If some problem inside
77       * @checkstyle ExecutableStatementCount (6 lines)
78       */
79      @Test
80      @SuppressWarnings({ "unchecked", "PMD.CloseResource" })
81      public void rendersFrontPage() throws Exception {
82          // @checkstyle MultipleStringLiterals (1 line)
83          Manifests.inject("Rultor-Revision", "12345");
84          final IndexRs res = new IndexRs();
85          res.setUriInfo(new UriInfoMocker().mock());
86          res.setHttpHeaders(new HttpHeadersMocker().mock());
87          res.setSecurityContext(Mockito.mock(SecurityContext.class));
88          final User user = Mockito.mock(User.class);
89          final Stands stands = Mockito.mock(Stands.class);
90          Mockito.doReturn(stands).when(user).stands();
91          final Pulses flow = new Pulses.Row();
92          Mockito.doReturn(flow).when(stands).flow();
93          final Rules rules = Mockito.mock(Rules.class);
94          Mockito.doReturn(rules).when(user).rules();
95          Mockito.doReturn(Collections.emptyIterator())
96              .when(rules).iterator();
97          final Users users = Mockito.mock(Users.class);
98          Mockito.doReturn(user).when(users).get(Mockito.any(URN.class));
99          final Account account = Mockito.mock(Account.class);
100         Mockito.doReturn(account).when(user).account();
101         Mockito.doReturn(new Dollars(1)).when(account).balance();
102         res.setServletContext(
103             new ServletContextMocker().withAttribute(
104                 Users.class.getName(), users
105             ).mock()
106         );
107         final Response response = res.index();
108         MatcherAssert.assertThat(
109             JaxbConverter.the(response.getEntity()),
110             XhtmlMatchers.hasXPaths(
111                 "/page/version[name='1.0-SNAPSHOT']"
112             )
113         );
114     }

116     /**
117      * IndexRs generates only public pulses for anonymous users.
118      * @throws Exception If some problem inside
119      */
120     @Test
121     @SuppressWarnings({ "unchecked", "PMD.CloseResource" })
122     public void providesOnlyPublicPulsesForAnonymous() throws Exception {
123         // @checkstyle MultipleStringLiterals (1 line)
124         Manifests.inject("Rultor-Revision", "1");
125         final Users users = Mockito.mock(Users.class);
126         final String visible = "visible";
127         final String invisible = "invisible";
128         final String viurn = "urn:user:user1";
129         final String inviurn = "urn:user:user2";
130         final Pulses pulses = Mockito.mock(Pulses.class);
131         final Iterator<Pulse> iterator = Arrays.asList(
132             this.pulse(visible, viurn),
133             this.pulse(invisible, inviurn)
134         ).iterator();
135         Mockito.when(pulses.iterator()).thenReturn(iterator);
136         Mockito.when(users.flow()).thenReturn(pulses);
137         this.stand(users, visible, true);
138         this.stand(users, invisible, false);
139         final IndexRs res = this.prepare(users);
140         final Response response = res.index();
141         final String coords = "/page/pulses/pulse/coordinates[owner='%s']";
142         MatcherAssert.assertThat(
143             JaxbConverter.the(response.getEntity()),
144             XhtmlMatchers.hasXPaths(String.format(coords, viurn))
145         );
146         MatcherAssert.assertThat(
147             JaxbConverter.the(response.getEntity()),
148             Matchers.not(
149                 XhtmlMatchers.hasXPaths(String.format(coords, inviurn))
150             )
151         );
152     }

154     /**
155      * Prepare IndexRs to be used in tests.
156      * @param users Users to store in attributes.
157      * @return IndexRs with mocked data.
158      */
159     private IndexRs prepare(final Users users) {
160         final IndexRs res = new IndexRs();
161         res.setUriInfo(new UriInfoMocker().mock());
162         res.setHttpHeaders(new HttpHeadersMocker().mock());
163         res.setSecurityContext(Mockito.mock(SecurityContext.class));
164         res.setServletContext(
165             new ServletContextMocker()
166                 .withAttribute(Users.class.getName(), users)
167                 .withAttribute(Repo.class.getName(), new ClasspathRepo())
168                 .mock()
169         );
170         return res;
171     }
172 
173     /**
174      * Mock pulse.
175      * @param stand Stand name.
176      * @param urn User URN.
177      * @return Pulse
178      */
179     private Pulse pulse(final String stand, final String urn) {
180         final Pulse pulse = Mockito.mock(Pulse.class);
181         final Coordinates coords = Mockito.mock(Coordinates.class);
182         Mockito.when(coords.owner()).thenReturn(URN.create(urn));
183         Mockito.when(coords.rule()).thenReturn("");
184         Mockito.when(coords.scheduled()).thenReturn(new Time());
185         Mockito.when(pulse.coordinates()).thenReturn(coords);
186         Mockito.when(pulse.stand()).thenReturn(stand);
187         final Tags tags = Mockito.mock(Tags.class);
188         Mockito.when(tags.iterator())
189             .thenReturn(Collections.<Tag>emptyIterator());
190         Mockito.when(pulse.tags()).thenReturn(tags);
191         return pulse;
192     }

194     /**
195      * Add a mock stand.
196      * @param users Users that provide stands.
197      * @param name Name of the stand.
198      * @param visible If the stand is public.
199      */
200     private void stand(final Users users, final String name,
201         final boolean visible) {
202         final Stand stand = Mockito.mock(Stand.class);
203         final Spec acl;
204         if (visible) {
205             acl = new Spec.Simple("com.rultor.acl.OpenView()");
206         } else {
207             acl = new Spec.Simple("com.rultor.acl.Prohibited()");
208         }
209         Mockito.when(stand.acl()).thenReturn(acl);
210         Mockito.when(users.stand(name)).thenReturn(stand);
211     }
