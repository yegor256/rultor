62      /**
63       * GroupAppender can log slf4j messages.
64       * @throws Exception If some problem inside
65       */
66      @Test
67      public void logsMessages() throws Exception {
68          final Drain drain = Mockito.mock(Drain.class);
69          final GroupAppender appender =
70              new GroupAppender(new Time(), drain);
71          appender.setLayout(new PatternLayout("%m"));
72          final String text = "test message to see in log";
73          final Thread publisher = new Thread() {
74              @Override
75              public void run() {
76                  appender.append(
77                      new LoggingEvent(
78                          "",
79                          Logger.getLogger(this.getClass()),
80                          org.apache.log4j.Level.INFO,
81                          text,
82                          new IllegalArgumentException()
83                      )
84                  );
85              }
86          };
87          publisher.start();
88          publisher.join();
89          appender.close();
90          Mockito.verify(drain).append(
91              Mockito.argThat(
92                  Matchers.everyItem(
93                      Matchers.endsWith(
94                          " INFO test message to see in log"
95                      )
96                  )
97              )
98          );
99      }

129     /**
130      * GroupAppender can log in multiple threads.
131      * @throws Exception If some problem inside
132      */
133     @Test
134     public void logsMessagesInMultipleThreads() throws Exception {
135         final String text = "test message to see \u20ac";
136         final CountDownLatch start = new CountDownLatch(1);
137         final int threads = Tv.TEN;
138         final CountDownLatch done = new CountDownLatch(threads);
139         final CountDownLatch ready = new CountDownLatch(threads);
140         final Collection<GroupAppender> appenders =
141             new CopyOnWriteArrayList<GroupAppender>();
142         final Runnable runnable = new VerboseRunnable(
143             // @checkstyle AnonInnerLength (50 lines)
144             new Callable<Void>() {
145                 @Override
146                 public Void call() throws Exception {
147                     final Drain drain = Mockito.mock(Drain.class);
148                     final GroupAppender appender =
149                         new GroupAppender(new Time(), drain);
150                     appender.setLayout(new PatternLayout("%m%n"));
151                     appenders.add(appender);
152                     ready.countDown();
153                     assert start.await(1, TimeUnit.SECONDS);
154                     final LoggingEvent event = new LoggingEvent(
155                         "",
156                         Logger.getLogger(this.getClass()),
157                         org.apache.log4j.Level.INFO,
158                         text,
159                         new IllegalArgumentException()
160                     );
161                     for (GroupAppender app : appenders) {
162                         app.append(event);
163                         app.run();
164                     }
165                     Mockito.verify(drain).append(
166                         Mockito.argThat(Matchers.<String>iterableWithSize(1))
167                     );
168                     done.countDown();
169                     return null;
170                 }
171             }
172         );
173         final ExecutorService svc = Executors.newFixedThreadPool(
174             threads,
175             new ThreadFactory() {
176                 private final transient AtomicLong group = new AtomicLong();
177                 @Override
178                 public Thread newThread(final Runnable runnable) {
179                     return new Thread(
180                         new ThreadGroup(
181                             Long.toString(this.group.incrementAndGet())
182                         ),
183                         runnable,
184                         String.format("app-%d", this.group.get())
185                     );
186                 }
187             }
188         );
189         for (int thread = 0; thread < threads; ++thread) {
190             svc.submit(runnable);
191         }
192         MatcherAssert.assertThat(
193             ready.await(1, TimeUnit.SECONDS),
194             Matchers.is(true)
195         );
196         start.countDown();
197         MatcherAssert.assertThat(
198             done.await(1, TimeUnit.SECONDS),
199             Matchers.is(true)
200         );
201     }
