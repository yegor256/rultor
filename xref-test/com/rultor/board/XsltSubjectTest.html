43  public final class XsltSubjectTest {

45      /**
46       * Transforms subject via XSLT.
47       * @throws Exception In case of error.
48       */
49      @Test
50      public void transformsSubject() throws Exception {
51          final Bill bill = Mockito.mock(Bill.class);
52          Mockito
53              .doReturn("<subject><value>Subj</value></subject>")
54              .when(bill).subject();
55          MatcherAssert.assertThat(
56              new XsltSubject(
57                  StringUtils.join(
58                      "<?xml version=\"1.0\"?>",
59                      // @checkstyle LineLength (1 line)
60                      "<xsl:stylesheet xmlns:xsl=\"http://www.w3.org/1999/XSL/Transform\" version=\"1.0\">",
61                      "<xsl:template match=\"/subject\">",
62                      "<test><xsl:value-of select=\"value\"/></test>",
63                      "</xsl:template>",
64                      "</xsl:stylesheet>"
65                  ),
66                  bill
67              ).subject(),
68              Matchers.equalTo("<test>Subj</test>")
69          );
70      }
71
