45  /**
46   * Test case for {@link SES}.
47   * @author Yegor Bugayenko (yegor@tpc2.com)

50  public final class SESTest {
51  
52      /**
53       * SES can send emails.
54       * @throws Exception If some problem inside
55       */
56      @Test
57      public void sendsEmail() throws Exception {
58          final SESClient client = Mockito.mock(SESClient.class);
59          final AmazonSimpleEmailService aws =
60              Mockito.mock(AmazonSimpleEmailService.class);
61          Mockito.doReturn(aws).when(client).get();
62          Mockito.doReturn(new SendEmailResult()).when(aws)
63              .sendEmail(Mockito.any(SendEmailRequest.class));
64          final String subject = "hello, друг!";
65          final Billboard board = new SES(
66              new Bill.Simple(
67                  subject,
68                  "sender@rultor.com",
69                  Arrays.asList("recepient@rultor.com")
70              ),
71              client
72          );
73          board.announce(true);
74          Mockito.verify(aws).sendEmail(
75              Mockito.argThat(
76                  Matchers.<SendEmailRequest>hasProperty(
77                      "message",
78                      Matchers.allOf(
79                          Matchers.<Message>hasProperty(
80                              "subject",
81                              Matchers.<Content>hasProperty(
82                                  // @checkstyle MultipleStringLiterals (1 line)
83                                  "data",
84                                  Matchers.equalTo(subject)
85                              )
86                          ),
87                          Matchers.<Message>hasProperty(
88                              "body",
89                              Matchers.<Body>hasProperty(
90                                  "text",
91                                  Matchers.<Content>hasProperty(
92                                      "data",
93                                      XhtmlMatchers.hasXPath("/snapshot")
94                                  )
95                              )
96                          )
97                      )
98                  )
99              )
100         );
101     }
