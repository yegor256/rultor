47  /**
48   * Test case for {@link IncrementalBash}.
49   * @author Yegor Bugayenko (yegor@tpc2.com)
50   * @version $Id$
51   * @checkstyle ClassDataAbstractionCoupling (500 lines)
52   */
53  public final class IncrementalBashTest {
54  
55      /**
56       * IncrementalBash can run complex script with bash.
57       * @throws Exception If some problem inside
58       */
59      @Test
60      public void runsIncrementalBashScript() throws Exception {
61          final ByteArrayOutputStream stdout = new ByteArrayOutputStream();
62          final File dir = Files.createTempDir();
63          FileUtils.write(new File(dir, "a.txt"), "first\nsecond");
64          final int code = new IncrementalBash(
65              new Permanent(new ShellMocker.Bash(dir)),
66              Arrays.asList(
67                  "MSG='$A'; echo `date` $A; sleep 1;",
68                  "find . -name \"a.txt\" | grep txt | wc -l",
69                  "if [ -f a.txt ]; then echo 'exists!'; fi",
70                  "/usr/bin/broken-name"
71              )
72          ).exec(new ImmutableMap.Builder<String, Object>().build(), stdout);
73          MatcherAssert.assertThat(code, Matchers.not(Matchers.equalTo(0)));
74          MatcherAssert.assertThat(
75              XhtmlMatchers.xhtml(
76                  new Snapshot(
77                      new ByteArrayInputStream(stdout.toByteArray())
78                  ).dom()
79              ),
80              XhtmlMatchers.hasXPaths(
81                  "/snapshot/steps/step",
82                  // @checkstyle LineLength (5 lines)
83                  "//step[summary=\"`MSG='$A'; echo \\`date\\` $A; sleep 1;`\"]/start",
84                  "//step[summary='`/usr/bin/broken-name`']/exception",
85                  "//step/exception[stacktrace='bash: /usr/bin/broken-name: No such file or directory']",
86                  "//steps[count(step[level='INFO']) = 3]",
87                  "//steps[count(step[level='SEVERE']) = 1]",
88                  "//steps[count(step[start]) = 4]",
89                  "//steps[count(step[finish]) = 4]",
90                  "//steps[count(step[duration = '']) = 0]"
91              )
92          );
93      }
