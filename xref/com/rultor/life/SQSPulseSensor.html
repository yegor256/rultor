62  /**
63   * Sensor to pulses in SQS queue.
64   *
65   * @author Yegor Bugayenko (yegor@tpc2.com)
66   * @version $Id$
67   * @checkstyle ClassDataAbstractionCoupling (500 lines)
68   */
69  @Loggable(Loggable.DEBUG)
70  @EqualsAndHashCode(of = { "users", "client" })
71  @SuppressWarnings({ "PMD.DoNotUseThreads", "PMD.ExcessiveImports" })
72  public final class SQSPulseSensor implements Runnable, Closeable {
73  
74      /**
75       * JSON schema-based reader.
76       */
77      private static final NormJson NORM = new NormJson(
78          SQSPulseSensor.class.getResourceAsStream("pulse.json")
79      );
80  
81      /**
82       * How many threads to use.
83       */
84      private static final int THREADS =
85          Runtime.getRuntime().availableProcessors() * Tv.FIVE;
86  
87      /**
88       * Executor service.
89       */
90      private final transient ScheduledExecutorService service =
91          Executors.newScheduledThreadPool(
92              SQSPulseSensor.THREADS, new VerboseThreads()
93          );
94  
95      /**
96       * Users.
97       */
98      private final transient Users users;
99  
100     /**
101      * Repository for ACL instantiation.
102      */
103     private final transient Repo repo;
104 
105     /**
106      * Quartz queue client.
107      */
108     private final transient SQSClient client;
109 
110     /**
111      * Public ctor.
112      * @param usr Users
113      * @param rpo Repo
114      * @param clnt SQS client for quartz queue
115      */
116     protected SQSPulseSensor(final Users usr, final Repo rpo,
117         final SQSClient clnt) {
118         this.users = usr;
119         this.repo = rpo;
120         this.client = clnt;
121         final Runnable runnable = new VerboseRunnable(this, true, false);
122         for (int thread = 0; thread < SQSPulseSensor.THREADS; ++thread) {
123             this.service.scheduleWithFixedDelay(
124                 runnable, TimeUnit.SECONDS.toMillis(1), 1,
125                 TimeUnit.MILLISECONDS
126             );
127         }
128     }
129 
130     /**
131      * {@inheritDoc}
132      */
133     @Override
134     @SuppressWarnings({
135         "PMD.AvoidInstantiatingObjectsInLoops",
136         "PMD.AvoidCatchingGenericException"
137     })
138     public void run() {
139         final AmazonSQS aws = this.client.get();
140         final ReceiveMessageResult result = aws.receiveMessage(
141             new ReceiveMessageRequest()
142                 .withQueueUrl(this.client.url())
143                 .withWaitTimeSeconds(Tv.TWENTY)
144                 .withVisibilityTimeout(Tv.FIVE)
145                 .withMaxNumberOfMessages(Tv.TEN)
146         );
147         for (Message msg : result.getMessages()) {
148             try {
149                 this.post(SQSPulseSensor.NORM.readObject(msg.getBody()));
150             } catch (SecurityException ex) {
151                 Logger.info(this, ExceptionUtils.getRootCauseMessage(ex));
152             // @checkstyle IllegalCatch (1 line)
153             } catch (Stand.BrokenXemblyException ex) {
154                 Logger.warn(this, ExceptionUtils.getRootCauseMessage(ex));
155             } catch (NormJson.JsonException ex) {
156                 Logger.warn(this, ExceptionUtils.getRootCauseMessage(ex));
157             } finally {
158                 aws.deleteMessage(
159                     new DeleteMessageRequest()
160                         .withQueueUrl(this.client.url())
161                         .withReceiptHandle(msg.getReceiptHandle())
162                 );
163             }
164         }
165         aws.shutdown();
166     }
167 
168     /**
169      * {@inheritDoc}
170      */
171     @Override
172     public void close() throws IOException {
173         this.service.shutdown();
174     }

176     /**
177      * Post this JSON to the right stand.
178      * @param json JSON to process
179      * @throws Stand.BrokenXemblyException If fails
180      * @checkstyle RedundantThrows (4 lines)
181      */
182     private void post(final JsonObject json)
183         throws Stand.BrokenXemblyException {
184         final Stand stand = this.users.stand(json.getString("stand"));
185         final String key = json.getString("key");
186         if (!this.acl(stand).canPost(key)) {
187             throw new SecurityException(
188                 String.format(
189                     "access denied to `%s` for '%s'",
190                     stand.name(), key
191                 )
192             );
193         }
194         final JsonObject work = json.getJsonObject("work");
195         stand.post(
196             String.format(
197                 "%s:%s:%s",
198                 work.getString("owner"),
199                 work.getString("unit"),
200                 work.getString("scheduled")
201             ),
202             json.getJsonNumber("nano").longValue(),
203             json.getString("xembly")
204         );
205     }
206 
207     /**
208      * Get ACL of a stand.
209      * @param stand The stand
210      * @return ACL
211      */
212     private ACL acl(final Stand stand) {
213         try {
214             return ACL.class.cast(
215                 new Repo.Cached(this.repo, new User.Nobody(), stand.acl())
216                     .get()
217                     .instantiate(
218                         this.users,
219                         new Arguments(new Work.None(), new Wallet.Empty())
220                     )
221             );
222         } catch (SpecException ex) {
223             throw new IllegalStateException(ex);
224         }
225     }
