35  import com.amazonaws.services.sqs.model.ReceiveMessageRequest;
36  import com.amazonaws.services.sqs.model.ReceiveMessageResult;

38  import com.jcabi.aspects.Tv;
39  import com.jcabi.log.VerboseRunnable;
40  import com.jcabi.log.VerboseThreads;
41  import com.jcabi.urn.URN;

54  import com.rultor.tools.Time;
55  import java.io.Closeable;
56  import java.io.IOException;
57  import java.util.concurrent.Executors;
58  import java.util.concurrent.ScheduledExecutorService;
59  import java.util.concurrent.TimeUnit;
60  import javax.json.JsonObject;
61  import lombok.EqualsAndHashCode;

63  /**
64   * Sensor to pulses in SQS queue.
65   *
66   * @author Yegor Bugayenko (yegor@tpc2.com)
67   * @version $Id$
68   * @checkstyle ClassDataAbstractionCoupling (500 lines)
69   */
70  @Loggable(Loggable.DEBUG)
71  @EqualsAndHashCode(of = { "users", "client" })
72  @SuppressWarnings({ "PMD.DoNotUseThreads", "PMD.ExcessiveImports" })
73  public final class SQSPulseSensor implements Runnable, Closeable {
74  
75      /**
76       * JSON schema-based reader.
77       */
78      private static final NormJson NORM = new NormJson(
79          SQSPulseSensor.class.getResourceAsStream("pulse.json")
80      );
81  
82      /**
83       * How many threads to use.
84       */
85      private static final int THREADS = Math.min(
86          Runtime.getRuntime().availableProcessors() * Tv.FIVE,
87          Tv.TEN
88      );
89  
90      /**
91       * Executor service.
92       */
93      private final transient ScheduledExecutorService service =
94          Executors.newScheduledThreadPool(
95              SQSPulseSensor.THREADS, new VerboseThreads()
96          );
97  
98      /**
99       * Users.
100      */
101     private final transient Users users;
102 
103     /**
104      * Repository for ACL instantiation.
105      */
106     private final transient Repo repo;
107 
108     /**
109      * Quartz queue client.
110      */
111     private final transient SQSClient client;
112 
113     /**
114      * Public ctor.
115      * @param usr Users
116      * @param rpo Repo
117      * @param clnt SQS client for quartz queue
118      */
119     SQSPulseSensor(final Users usr, final Repo rpo,
120         final SQSClient clnt) {
121         this.users = usr;
122         this.repo = rpo;
123         this.client = clnt;
124         final Runnable runnable = new VerboseRunnable(this, true, false);
125         for (int thread = 0; thread < SQSPulseSensor.THREADS; ++thread) {
126             this.service.scheduleWithFixedDelay(
127                 runnable, TimeUnit.SECONDS.toMillis(1L), 1L,
128                 TimeUnit.MILLISECONDS
129             );
130         }
131     }
132 
133     @Override
134     @Loggable(value = Loggable.DEBUG, limit = Integer.MAX_VALUE)
135     public void run() {
136         final AmazonSQS aws = this.client.get();
137         try {
138             final ReceiveMessageResult result = aws.receiveMessage(
139                 new ReceiveMessageRequest()
140                     .withQueueUrl(this.client.url())
141                     .withWaitTimeSeconds(Tv.FIVE)
142                     .withVisibilityTimeout(Tv.FIVE)
143                     .withMaxNumberOfMessages(Tv.TEN)
144             );
145             for (final Message msg : result.getMessages()) {
146                 this.process(aws, msg);
147             }
148         } finally {
149             aws.shutdown();
150         }
151     }
152 
153     @Override
154     public void close() throws IOException {
155         this.service.shutdown();
156     }
157 
158     /**
159      * Process one message.
160      * @param aws SQS client
161      * @param msg Message from SQS
162      */
163     private void process(final AmazonSQS aws, final Message msg) {
164         try {
165             this.post(SQSPulseSensor.NORM.readObject(msg.getBody()));
166         } catch (final SecurityException ex) {
167             Exceptions.info(this, ex);
168         } catch (final Stand.BrokenXemblyException ex) {
169             throw new IllegalArgumentException(ex);
170         } catch (final NormJson.JsonException ex) {
171             throw new IllegalArgumentException(ex);
172         } finally {
173             aws.deleteMessage(
174                 new DeleteMessageRequest()
175                     .withQueueUrl(this.client.url())
176                     .withReceiptHandle(msg.getReceiptHandle())
177             );
178         }
179     }
180 
181     /**
182      * Post this JSON to the right stand.
183      * @param json JSON to process
184      * @throws Stand.BrokenXemblyException If fails
185      * @checkstyle RedundantThrows (5 lines)
186      */
187     private void post(final JsonObject json)
188         throws Stand.BrokenXemblyException {
189         final Stand stand = this.users.stand(json.getString("stand"));
190         final String key = json.getString("key");
191         if (!this.acl(stand).canPost(key)) {
192             throw new SecurityException(
193                 String.format(
194                     "access denied to `%s` for '%s'",
195                     stand.name(), key
196                 )
197             );
198         }
199         final JsonObject work = json.getJsonObject("work");
200         stand.post(
201             new Coordinates.Simple(
202                 URN.create(work.getString("owner")),
203                 work.getString("rule"),
204                 new Time(work.getString("scheduled"))
205             ),
206             json.getJsonNumber("nano").longValue(),
207             json.getString("xembly")
208         );
209     }
210 
211     /**
212      * Get ACL of a stand.
213      * @param stand The stand
214      * @return ACL
215      */
216     private ACL acl(final Stand stand) {
217         try {
218             return ACL.class.cast(
219                 this.repo.make(new User.Nobody(), stand.acl())
220                     .instantiate(
221                         this.users,
222                         new Arguments(
223                             new Coordinates.None(), new Wallet.Empty()
224                         )
225                     )
226             );
227         } catch (final SpecException ex) {
228             throw new IllegalStateException(ex);
229         }
230     }
