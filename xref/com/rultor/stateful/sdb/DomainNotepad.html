152     /**
153      * {@inheritDoc}
154      */
155     @Override
156     public boolean contains(final Object object) {
157         final long start = System.currentTimeMillis();
158         final GetAttributesResult result = this.client.get().getAttributes(
159             new GetAttributesRequest()
160                 .withDomainName(this.client.domain())
161                 .withItemName(this.name(object.toString()))
162                 .withConsistentRead(true)
163         );
164         this.wallet.charge(
165             Logger.format(
166                 // @checkstyle LineLength (1 line)
167                 "checked existence of AWS SimpleDB item from `%s` domain in %[ms]s",
168                 this.client.domain(),
169                 System.currentTimeMillis() - start
170             ),
171             new Dollars(Tv.FIVE)
172         );
173         return !result.getAttributes().isEmpty();
174     }

176     /**
177      * {@inheritDoc}
178      */
179     @Override
180     @RetryOnFailure
181     public Iterator<String> iterator() {
182         final String query = String.format(
183             "SELECT `%s` FROM `%s` WHERE `%s`='%s' AND `%s`='%s'",
184             DomainNotepad.ATTR_TEXT,
185             this.client.domain(),
186             DomainNotepad.ATTR_OWNER,
187             this.work.owner(),
188             DomainNotepad.ATTR_UNIT,
189             this.work.unit()
190         );
191         final long start = System.currentTimeMillis();
192         final SelectResult result = this.client.get().select(
193             new SelectRequest()
194                 .withConsistentRead(true)
195                 .withSelectExpression(query)
196         );
197         final Collection<String> items = new LinkedList<String>();
198         for (Item item : result.getItems()) {
199             final String text = item.getAttributes().get(0).getValue();
200             if (item.getName().equals(this.name(text))) {
201                 items.add(text);
202             }
203         }
204         this.wallet.charge(
205             Logger.format(
206                 // @checkstyle LineLength (1 line)
207                 "retrieved AWS SimpleDB %d item(s) from `%s` domain in %[ms]s (%d total)",
208                 items.size(), this.client.domain(),
209                 System.currentTimeMillis() - start, result.getItems().size()
210             ),
211             new Dollars(Tv.FIVE)
212         );
213         return items.iterator();
214     }

236     @Override
237     @RetryOnFailure
238     public boolean add(final String line) {
239         final long start = System.currentTimeMillis();
240         this.client.get().putAttributes(
241             new PutAttributesRequest()
242                 .withDomainName(this.client.domain())
243                 .withItemName(this.name(line))
244                 .withAttributes(
245                     new ReplaceableAttribute()
246                         .withName(DomainNotepad.ATTR_TEXT)
247                         .withValue(line)
248                         .withReplace(true),
249                     new ReplaceableAttribute()
250                         .withName(DomainNotepad.ATTR_OWNER)
251                         .withValue(this.work.owner().toString())
252                         .withReplace(true),
253                     new ReplaceableAttribute()
254                         .withName(DomainNotepad.ATTR_UNIT)
255                         .withValue(this.work.unit())
256                         .withReplace(true),
257                     new ReplaceableAttribute()
258                         .withName(DomainNotepad.ATTR_TIME)
259                         .withValue(new Time().toString())
260                         .withReplace(true)
261                 )
262         );
263         this.wallet.charge(
264             Logger.format(
265                 "added AWS SimpleDB item `%s` to `%s` domain in %[ms]s",
266                 this.name(line),
267                 this.client.domain(),
268                 System.currentTimeMillis() - start
269             ),
270             new Dollars(Tv.FIVE)
271         );
272         return true;
273     }

281         final long start = System.currentTimeMillis();
282         this.client.get().deleteAttributes(
283             new DeleteAttributesRequest()
284                 .withDomainName(this.client.domain())
285                 .withItemName(this.name(line.toString()))
286                 .withAttributes(
287                     new Attribute().withName(DomainNotepad.ATTR_OWNER),
288                     new Attribute().withName(DomainNotepad.ATTR_TEXT),
289                     new Attribute().withName(DomainNotepad.ATTR_TIME),
290                     new Attribute().withName(DomainNotepad.ATTR_UNIT)
291                 )
292         );
293         this.wallet.charge(
294             Logger.format(
295                 "removed AWS SimpleDB item `%s` from `%s` domain in %[ms]s",
296                 this.name(line.toString()),
297                 this.client.domain(),
298                 System.currentTimeMillis() - start
299             ),
300             new Dollars(Tv.FIVE)
301         );
302         return true;
303     }
