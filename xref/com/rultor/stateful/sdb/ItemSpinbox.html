32  import com.google.common.collect.ImmutableMap;
33  import com.jcabi.aspects.Immutable;
34  import com.jcabi.aspects.Loggable;
35  import com.jcabi.aspects.RetryOnFailure;
36  import com.jcabi.simpledb.Item;
37  import com.rultor.stateful.Spinbox;
38  import com.rultor.tools.Time;
39  import javax.validation.constraints.NotNull;
40  import lombok.EqualsAndHashCode;
41  import lombok.ToString;

43  /**
44   * Spinbox in SimpleDB item.
45   *
46   * @author Yegor Bugayenko (yegor@tpc2.com)
47   * @version $Id$
48   * @since 1.0
49   * @checkstyle ClassDataAbstractionCoupling (500 lines)
50   */
51  @Immutable
52  @ToString
53  @Loggable(Loggable.DEBUG)
54  @EqualsAndHashCode(of = "item")
55  public final class ItemSpinbox implements Spinbox {
56  
57      /**
58       * Attribute name.
59       */
60      private static final String COUNTER = "counter";
61  
62      /**
63       * Attribute name.
64       */
65      private static final String TIME = "time";
66  
67      /**
68       * SimpleDB item.
69       */
70      private final transient Item item;
71  
72      /**
73       * Public ctor.
74       * @param itm Item
75       */
76      public ItemSpinbox(
77          @NotNull(message = "item can't be NULL") final Item itm) {
78          this.item = itm;
79      }
80  
81      @Override
82      @RetryOnFailure(verbose = false)
83      public long add(final long value) {
84          String before = this.item.get(ItemSpinbox.COUNTER);
85          if (before == null) {
86              before = "0";
87          }
88          final long after = Long.parseLong(before) + value;
89          this.item.putAll(
90              new ImmutableMap.Builder<String, String>()
91                  .put(ItemSpinbox.COUNTER, Long.toString(after))
92                  .put(ItemSpinbox.TIME, new Time().toString())
93                  .build()
94          );
95          return after;
96      }
97  
98  }
