58  /**
59   * All users in Dynamo DB.
60   *
61   * @author Yegor Bugayenko (yegor@tpc2.com)
62   * @version $Id$
63   * @since 1.0
64   */
65  @Immutable
66  @ToString
67  @EqualsAndHashCode(of = "region")
68  @Loggable(Loggable.DEBUG)
69  public final class AwsUsers implements Users {
70  
71      /**
72       * Dynamo.
73       */
74      private final transient Region region;
75  
76      /**
77       * SQS client.
78       */
79      private final transient SQSClient client;
80  
81      /**
82       * Public ctor.
83       * @param reg AWS region
84       * @param sqs SQS client
85       */
86      public AwsUsers(final Region reg, final SQSClient sqs) {
87          final AmazonDynamoDB aws = reg.aws();
88          final DescribeTableResult result = aws.describeTable(
89              new DescribeTableRequest()
90                  // @checkstyle MultipleStringLiterals (1 line)
91                  .withTableName(reg.table(AwsRule.TABLE).name())
92          );
93          Logger.info(
94              AwsUsers.class, "Amazon DynamoDB is ready with %d rule(s)",
95              result.getTable().getItemCount()
96          );
97          this.region = reg;
98          this.client = sqs;
99      }
100 
101     /**
102      * {@inheritDoc}
103      */
104     @Override
105     @NotNull(message = "list of users is never NULL")
106     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
107     @RetryOnFailure(verbose = false)
108     public Iterator<User> iterator() {
109         final Collection<User> users = new HashSet<User>(0);
110         for (Item item : this.region.table(AwsRule.TABLE).frame()) {
111             users.add(
112                 new AwsUser(
113                     AwsUsers.this.region,
114                     AwsUsers.this.client,
115                     URN.create(item.get(AwsRule.HASH_OWNER).getS())
116                 )
117             );
118         }
119         return users.iterator();
120     }
121 
122     /**
123      * {@inheritDoc}
124      */
125     @Override
126     @NotNull(message = "User is never NULL")
127     public User get(@NotNull(message = "URN can't be empty") final URN urn) {
128         return new AwsUser(this.region, this.client, urn);
129     }
130 
131     /**
132      * {@inheritDoc}
133      */
134     @Override
135     @Cacheable(lifetime = Tv.FIVE, unit = TimeUnit.MINUTES)
136     @RetryOnFailure(verbose = false)
137     public Stand stand(final String stand) {
138         final Collection<Item> items = this.region.table(AwsStand.TABLE)
139             .frame()
140             .where(AwsStand.RANGE_STAND, stand);
141         if (items.isEmpty()) {
142             throw new NoSuchElementException(
143                 String.format("Stand `%s` doesn't exist", stand)
144             );
145         }
146         return new AwsStand(items.iterator().next());
147     }
148 
149     /**
150      * {@inheritDoc}
151      */
152     @Override
153     public Pulses flow() {
154         throw new UnsupportedOperationException();
155     }
