38  import com.jcabi.urn.URN;
39  import com.rultor.spi.Coordinates;

43  import java.util.concurrent.TimeUnit;
44  import javax.validation.constraints.NotNull;
45  import lombok.EqualsAndHashCode;
46  import lombok.ToString;

48  /**
49   * Single stand in Dynamo DB.
50   *
51   * @author Yegor Bugayenko (yegor@tpc2.com)
52   * @version $Id$
53   * @since 1.0
54   * @checkstyle ClassDataAbstractionCoupling (500 lines)
55   */
56  @Immutable
57  @ToString
58  @EqualsAndHashCode(of = "item")
59  @Loggable(Loggable.DEBUG)
60  final class AwsStand implements Stand {
61  
62      /**
63       * Dynamo DB table name.
64       */
65      public static final String TABLE = "stands";
66  
67      /**
68       * Dynamo DB table column.
69       */
70      public static final String HASH_OWNER = "owner";
71  
72      /**
73       * Dynamo DB table column.
74       */
75      public static final String RANGE_STAND = "stand";
76  
77      /**
78       * Dynamo DB table column.
79       */
80      public static final String FIELD_ACL = "acl";
81  
82      /**
83       * Dynamo DB table column.
84       */
85      public static final String FIELD_WIDGETS = "widgets";
86  
87      /**
88       * Item.
89       */
90      private final transient Item item;
91  
92      /**
93       * Public ctor.
94       * @param itm Item from Dynamo
95       */
96      AwsStand(final Item itm) {
97          this.item = itm;
98      }
99  
100     @Override
101     @Cacheable.FlushAfter
102     public void update(
103         @NotNull(message = "ACL spec can't be NULL") final Spec spec,
104         @NotNull(message = "widgets spec can't be NULL") final Spec widgets) {
105         this.item.put(
106             new Attributes()
107                 .with(AwsStand.FIELD_ACL, spec.asText())
108                 .with(AwsStand.FIELD_WIDGETS, widgets.asText())
109         );
110     }
111 
112     @Override
113     @NotNull(message = "ACL of a stand is never NULL")
114     @Cacheable(lifetime = Tv.FIVE, unit = TimeUnit.MINUTES)
115     public Spec acl() {
116         Spec spec;
117         if (this.item.has(AwsStand.FIELD_ACL)) {
118             spec = new Spec.Simple(this.item.get(AwsStand.FIELD_ACL).getS());
119         } else {
120             spec = new Spec.Simple("com.rultor.acl.Prohibited()");
121         }
122         return spec;
123     }
124 
125     @Override
126     @NotNull(message = "widgets of a stand is never NULL")
127     public Spec widgets() {
128         Spec spec;
129         if (this.item.has(AwsStand.FIELD_WIDGETS)) {
130             spec = new Spec.Simple(
131                 this.item.get(AwsStand.FIELD_WIDGETS).getS()
132             );
133         } else {
134             spec = new Spec.Simple("[]");
135         }
136         return spec;
137     }
138 
139     @Override
140     @Cacheable(lifetime = Tv.FIVE, unit = TimeUnit.MINUTES)
141     public String name() {
142         return this.item.get(AwsStand.RANGE_STAND).getS();
143     }
144 
145     @Override
146     @Cacheable(lifetime = Tv.FIVE, unit = TimeUnit.MINUTES)
147     public URN owner() {
148         return URN.create(this.item.get(AwsStand.HASH_OWNER).getS());
149     }
150 
151     @Override
152     public Pulses pulses() {
153         throw new UnsupportedOperationException();
154     }
155 
156     @Override
157     public void post(final Coordinates pulse, final long nano,
158         final String xembly) {
159         throw new UnsupportedOperationException();
160     }
161 
162 }
