35  import com.jcabi.dynamo.Attributes;
36  import com.jcabi.dynamo.Item;
37  import com.jcabi.urn.URN;
38  import com.rultor.aws.SQSClient;
39  import com.rultor.spi.Coordinates;
40  import com.rultor.spi.Rule;
41  import com.rultor.spi.Spec;

43  import javax.validation.constraints.NotNull;
44  import lombok.EqualsAndHashCode;
45  import lombok.ToString;

47  /**
48   * Single unit in Dynamo DB.
49   *
50   * @author Yegor Bugayenko (yegor@tpc2.com)
51   * @version $Id$
52   * @since 1.0
53   * @checkstyle ClassDataAbstractionCoupling (500 lines)
54   */
55  @Immutable
56  @ToString
57  @EqualsAndHashCode(of = { "item", "client" })
58  @Loggable(Loggable.DEBUG)
59  final class AwsRule implements Rule {
60  
61      /**
62       * Dynamo DB table name (the table has this name for historical
63       * reason, and we can't change it since there is no table renaming
64       * feature in DynamoDB).
65       */
66      public static final String TABLE = "units";
67  
68      /**
69       * Dynamo DB table column.
70       */
71      public static final String HASH_OWNER = "owner";
72  
73      /**
74       * Dynamo DB table column.
75       */
76      public static final String RANGE_NAME = "name";
77  
78      /**
79       * Dynamo DB table column.
80       */
81      public static final String FIELD_SPEC = "spec";
82  
83      /**
84       * Dynamo DB table column.
85       */
86      public static final String FIELD_DRAIN = "drain";
87  
88      /**
89       * Dynamo DB table column.
90       */
91      public static final String FIELD_FAILURE = "failure";
92  
93      /**
94       * Dynamo DB table column.
95       */
96      private static final String NO_FAILURE = "-";
97  
98      /**
99       * Item.
100      */
101     private final transient Item item;
102 
103     /**
104      * SQS client.
105      */
106     private final transient SQSClient client;
107 
108     /**
109      * Public ctor.
110      * @param sqs SQS client
111      * @param itm Item from Dynamo
112      */
113     protected AwsRule(final SQSClient sqs, final Item itm) {
114         this.client = sqs;
115         this.item = itm;
116     }
117 
118     /**
119      * {@inheritDoc}
120      */
121     @Override
122     @RetryOnFailure(verbose = false)
123     public void update(
124         @NotNull(message = "spec can't be NULL") final Spec spec,
125         @NotNull(message = "drain can't be NULL") final Spec drain) {
126         this.item.put(
127             new Attributes()
128                 .with(AwsRule.FIELD_SPEC, spec.asText())
129                 .with(AwsRule.FIELD_DRAIN, drain.asText())
130                 .with(AwsRule.FIELD_FAILURE, AwsRule.NO_FAILURE)
131         );
132     }
133 
134     /**
135      * {@inheritDoc}
136      */
137     @Override
138     @NotNull(message = "spec of a rule is never NULL")
139     @RetryOnFailure(verbose = false)
140     public Spec spec() {
141         Spec spec;
142         if (this.item.has(AwsRule.FIELD_SPEC)) {
143             spec = new Spec.Simple(this.item.get(AwsRule.FIELD_SPEC).getS());
144         } else {
145             spec = new Spec.Simple();
146         }
147         return spec;
148     }
149 
150     /**
151      * {@inheritDoc}
152      */
153     @Override
154     @RetryOnFailure(verbose = false)
155     public String name() {
156         return this.item.get(AwsRule.RANGE_NAME).getS();
157     }
158 
159     /**
160      * {@inheritDoc}
161      */
162     @Override
163     public Wallet wallet(final Coordinates work,
164         final URN taker, final String rule) {
165         return new SQSWallet(
166             this.client, work,
167             this.owner(), this.name(),
168             taker, rule
169         );
170     }
171 
172     /**
173      * {@inheritDoc}
174      */
175     @Override
176     @RetryOnFailure(verbose = false)
177     public Spec drain() {
178         Spec spec;
179         if (this.item.has(AwsRule.FIELD_DRAIN)) {
180             spec = new Spec.Simple(this.item.get(AwsRule.FIELD_DRAIN).getS());
181         } else {
182             spec = new Spec.Simple("com.rultor.drain.Trash()");
183         }
184         return spec;
185     }
186 
187     /**
188      * {@inheritDoc}
189      */
190     @Override
191     @RetryOnFailure(verbose = false)
192     public void failure(final String desc) {
193         this.item.put(
194             new Attributes()
195                 .with(AwsRule.FIELD_SPEC, this.spec().asText())
196                 .with(AwsRule.FIELD_DRAIN, this.drain().asText())
197                 .with(AwsRule.FIELD_FAILURE, desc)
198         );
199     }
200 
201     /**
202      * {@inheritDoc}
203      */
204     @Override
205     @RetryOnFailure(verbose = false)
206     public String failure() {
207         String failure;
208         if (this.item.has(AwsRule.FIELD_FAILURE)) {
209             failure = this.item.get(AwsRule.FIELD_FAILURE).getS();
210             if (failure.equals(AwsRule.NO_FAILURE)) {
211                 failure = "";
212             }
213         } else {
214             failure = "";
215         }
216         return failure;
217     }
218 
219     /**
220      * Owner of it.
221      * @return URN of the owner
222      */
223     @RetryOnFailure(verbose = false)
224     private URN owner() {
225         return URN.create(this.item.get(AwsRule.HASH_OWNER).getS());
226     }
