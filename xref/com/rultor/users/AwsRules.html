33  import com.jcabi.aspects.Immutable;
34  import com.jcabi.aspects.Loggable;
35  import com.jcabi.aspects.Tv;
36  import com.jcabi.dynamo.Attributes;

56  /**
57   * Rules in DynamoDB.
58   *
59   * @author Yegor Bugayenko (yegor@tpc2.com)
60   * @version $Id$
61   * @since 1.0
62   */
63  @Immutable
64  @ToString
65  @EqualsAndHashCode(of = { "region", "client", "owner" })
66  @Loggable(Loggable.DEBUG)
67  @SuppressWarnings("PMD.TooManyMethods")
68  final class AwsRules implements Rules {
69  
70      /**
71       * Dynamo.
72       */
73      private final transient Region region;
74  
75      /**
76       * SQS client.
77       */
78      private final transient SQSClient client;
79  
80      /**
81       * URN of the user.
82       */
83      private final transient URN owner;
84  
85      /**
86       * Public ctor.
87       * @param reg Region in Dynamo
88       * @param sqs SQS client
89       * @param urn URN of the user
90       */
91      AwsRules(final Region reg, final SQSClient sqs, final URN urn) {
92          this.region = reg;
93          this.client = sqs;
94          this.owner = urn;
95      }
96  
97      @Override
98      @NotNull(message = "list of rules of a user is never NULL")
99      public Iterator<Rule> iterator() {
100         final Iterator<Item> items = this.fetch().iterator();
101         return new Iterator<Rule>() {
102             @Override
103             public boolean hasNext() {
104                 return items.hasNext();
105             }
106             @Override
107             public Rule next() {
108                 return AwsRules.this.toRule(items.next());
109             }
110             @Override
111             public void remove() {
112                 throw new UnsupportedOperationException();
113             }
114         };
115     }
116 
117     @Override
118     @Cacheable.FlushAfter
119     public void create(
120         @NotNull(message = "rule name is mandatory when creating new rule")
121         @Pattern(
122             regexp = "[a-z][-a-z0-9]{2,}",
123             message = "Only numbers, letters, and dashes are allowed"
124         )
125         final String unt) {
126         if (this.contains(unt)) {
127             throw new IllegalArgumentException(
128                 String.format("Rule `%s` already exists", unt)
129             );
130         }
131         this.region.table(AwsRule.TABLE).put(
132             new Attributes()
133                 .with(AwsRule.HASH_OWNER, this.owner.toString())
134                 .with(AwsRule.RANGE_NAME, unt)
135                 .with(AwsRule.FIELD_SPEC, new Spec.Simple().asText())
136         );
137     }
138 
139     @Override
140     @Cacheable.FlushAfter
141     public void remove(@NotNull(message = "rule name is mandatory")
142         final String rule) {
143         final Iterator<Item> items = this.region.table(AwsRule.TABLE).frame()
144             .where(AwsRule.HASH_OWNER, this.owner.toString())
145             .where(AwsRule.RANGE_NAME, rule)
146             .through(new QueryValve())
147             .iterator();
148         if (!items.hasNext()) {
149             throw new NoSuchElementException(
150                 String.format("Rule `%s` not found", rule)
151             );
152         }
153         items.next();
154         items.remove();
155     }
156 
157     @Override
158     @Cacheable(lifetime = Tv.FIVE, unit = TimeUnit.MINUTES)
159     public Rule get(
160         @NotNull(message = "rule name can't be NULL")
161         @Pattern(regexp = ".+", message = "rule shouldn't be blank")
162         final String rule) {
163         final Collection<Item> items = this.region.table(AwsRule.TABLE)
164             .frame()
165             .where(AwsRule.HASH_OWNER, this.owner.toString())
166             .where(AwsRule.RANGE_NAME, rule)
167             .through(new QueryValve());
168         if (items.isEmpty()) {
169             throw new NoSuchElementException(
170                 String.format("Rule `%s` doesn't exist", rule)
171             );
172         }
173         return this.toRule(items.iterator().next());
174     }
175 
176     @Override
177     @Cacheable(lifetime = Tv.FIVE, unit = TimeUnit.MINUTES)
178     public boolean contains(
179         @NotNull(message = "rule name can't be NULL")
180         @Pattern(regexp = ".+", message = "rule shouldn't be blank")
181         final String rule) {
182         return !this.region.table(AwsRule.TABLE)
183             .frame()
184             .where(AwsRule.HASH_OWNER, this.owner.toString())
185             .where(AwsRule.RANGE_NAME, rule)
186             .through(new QueryValve())
187             .isEmpty();
188     }
189 
190     /**
191      * Fetch them all.
192      * @return All Rules
193      */
194     @Cacheable(lifetime = Tv.FIVE, unit = TimeUnit.MINUTES)
195     private Collection<Item> fetch() {
196         return this.region.table(AwsRule.TABLE)
197             .frame()
198             .where(AwsRule.HASH_OWNER, this.owner.toString())
199             .through(
200                 new QueryValve().withAttributesToGet(
201                     AwsRule.FIELD_SPEC,
202                     AwsRule.FIELD_DRAIN,
203                     AwsRule.FIELD_FAILURE
204                 )
205             );
206     }
207 
208     /**
209      * Marker to flush cash.
210      */
211     @Cacheable.FlushAfter
212     private void flush() {
213         // nothing to do here
214     }
215 
216     /**
217      * Convert item to rule.
218      * @param item Item from AWS
219      * @return Rule
220      */
221     private Rule toRule(final Item item) {
222         return new CacheableRule(new AwsRule(this.client, item));
223     }
224 
225     /**
226      * Rule that caches its calls.
227      */
228     @ToString
229     @EqualsAndHashCode(of = "origin")
230     @Loggable(Loggable.DEBUG)
231     private final class CacheableRule implements Rule {
232         /**
233          * Original rule.
234          */
235         private final transient Rule origin;
236         /**
237          * Ctor.
238          * @param rule Rule to wrap
239          */
240         CacheableRule(final Rule rule) {
241             this.origin = rule;
242         }
243         @Override
244         @Cacheable(lifetime = Tv.FIVE, unit = TimeUnit.MINUTES)
245         public String name() {
246             return this.origin.name();
247         }
248         @Override
249         @Cacheable.FlushAfter
250         public void update(final Spec spec, final Spec drain) {
251             this.origin.update(spec, drain);
252             AwsRules.this.flush();
253         }
254         @Override
255         @Cacheable(lifetime = Tv.FIVE, unit = TimeUnit.MINUTES)
256         public Spec spec() {
257             return this.origin.spec();
258         }
259         @Override
260         @Cacheable(lifetime = Tv.FIVE, unit = TimeUnit.MINUTES)
261         public Spec drain() {
262             return this.origin.drain();
263         }
264         @Override
265         @Cacheable.FlushAfter
266         public void failure(final String desc) {
267             this.origin.failure(desc);
268             AwsRules.this.flush();
269         }
270         @Override
271         @Cacheable(lifetime = Tv.FIVE, unit = TimeUnit.MINUTES)
272         public String failure() {
273             return this.origin.failure();
274         }
275         /**
276          * {@inheritDoc}
277          * @checkstyle RedundantThrows (5 lines)
278          */
279         @Override
280         public Wallet wallet(final Coordinates work, final URN taker,
281             final String rule) throws Wallet.NotEnoughFundsException {
282             return this.origin.wallet(work, taker, rule);
283         }
284     }
