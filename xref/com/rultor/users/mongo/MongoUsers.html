30  package com.rultor.users.mongo;

32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.aspects.Loggable;
34  import com.jcabi.urn.URN;
35  import com.rultor.spi.Pulses;
36  import com.rultor.spi.Stand;
37  import com.rultor.spi.User;
38  import com.rultor.spi.Users;
39  import java.util.Iterator;
40  import lombok.EqualsAndHashCode;
41  import lombok.ToString;
42  
43  /**
44   * Users with extra features from Mongo.
45   *
46   * @author Yegor Bugayenko (yegor@tpc2.com)
47   * @version $Id$
48   * @since 1.0
49   */
50  @Immutable
51  @ToString
52  @EqualsAndHashCode(of = { "mongo", "origin" })
53  @Loggable(Loggable.DEBUG)
54  public final class MongoUsers implements Users {
55  
56      /**
57       * Mongo container.
58       */
59      private final transient Mongo mongo;
60  
61      /**
62       * Original users.
63       */
64      private final transient Users origin;
65  
66      /**
67       * Public ctor.
68       * @param mng Mongo container
69       * @param users Users
70       */
71      public MongoUsers(final Mongo mng, final Users users) {
72          this.mongo = mng;
73          this.origin = users;
74      }
75  
76      @Override
77      public Iterator<User> iterator() {
78          final Iterator<User> iter = this.origin.iterator();
79          return new Iterator<User>() {
80              @Override
81              public boolean hasNext() {
82                  return iter.hasNext();
83              }
84              @Override
85              public User next() {
86                  return new MongoUser(MongoUsers.this.mongo, iter.next());
87              }
88              @Override
89              public void remove() {
90                  throw new UnsupportedOperationException();
91              }
92          };
93      }
94  
95      @Override
96      public User get(final URN name) {
97          return new MongoUser(this.mongo, this.origin.get(name));
98      }
99  
100     @Override
101     public Stand stand(final String name) {
102         return new MongoStand(this.mongo, this.origin.stand(name));
103     }
104 
105     @Override
106     public Pulses flow() {
107         return new MongoPulses(
108             this.mongo,
109             new Predicate.Any(),
110             new Predicate.Any()
111         );
112     }
113 
114 }
