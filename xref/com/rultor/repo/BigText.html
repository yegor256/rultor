45  /**
46   * Big text, without any formatting.
47   *
48   * @author Yegor Bugayenko (yegor@tpc2.com)
49   * @version $Id$
50   * @since 1.0
51   */
52  @Immutable
53  @ToString
54  @EqualsAndHashCode(of = "lines")
55  @Loggable(Loggable.DEBUG)
56  final class BigText implements Variable<String> {
57  
58      /**
59       * EOL.
60       */
61      private static final String EOL = "\n";
62  
63      /**
64       * Lines of text.
65       */
66      private final transient String[] lines;
67  
68      /**
69       * Public ctor.
70       * @param text Text to parse and encapsulate
71       */
72      protected BigText(final String text) {
73          this.lines = StringUtils.splitPreserveAllTokens(
74              StringUtils.stripStart(
75                  StringUtils.stripEnd(text, null),
76                  "\n\r"
77              ),
78              BigText.EOL
79          );
80          int min = Integer.MAX_VALUE;
81          for (int idx = 0; idx < this.lines.length; ++idx) {
82              this.lines[idx] = StringUtils.strip(this.lines[idx], "\r")
83                  .replaceAll("\\s+$", "")
84                  .replaceAll("\t", "    ");
85              if (this.lines[idx].isEmpty()) {
86                  continue;
87              }
88              min = Math.min(
89                  min,
90                  this.lines[idx].length()
91                  - this.lines[idx].replaceAll("^\\s+", "").length()
92              );
93          }
94          for (int idx = 0; idx < this.lines.length; ++idx) {
95              if (this.lines[idx].length() > min) {
96                  this.lines[idx] = this.lines[idx].substring(min);
97              } else {
98                  this.lines[idx] = StringUtils.repeat(' ', min);
99              }
100         }
101     }
102 
103     /**
104      * {@inheritDoc}
105      * @checkstyle RedundantThrows (8 lines)
106      */
107     @Override
108     @NotNull
109     public String instantiate(
110         @NotNull(message = "users can't be NULL") final Users users,
111         @NotNull(message = "arguments can't be NULL") final Arguments args)
112         throws SpecException {
113         return StringUtils.join(this.lines, BigText.EOL);
114     }
115 
116     /**
117      * {@inheritDoc}
118      */
119     @Override
120     public String asText() {
121         return new StringBuilder("\"\"\"\n")
122             .append(StringUtils.join(this.lines, BigText.EOL))
123             .append("\n\"\"\"")
124             .toString();
125     }
126 
127     /**
128      * {@inheritDoc}
129      */
130     @Override
131     public Map<Integer, String> arguments() {
132         return new ConcurrentHashMap<Integer, String>(0);
133     }
