53  /**
54   * Comments a new tag in Github.
55   *
56   * @author Yegor Bugayenko (yegor256@gmail.com)
57   * @version $Id: 07ca111b93cd8b02aa56565829fad0e9fb956839 $
58   * @since 1.31
59   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
60   */
61  @Immutable
62  @ToString
63  @EqualsAndHashCode(callSuper = false, of = "github")
64  @SuppressWarnings("PMD.ExcessiveImports")
65  public final class CommentsTag extends AbstractAgent {
66  
67      /**
68       * Message bundle.
69       */
70      private static final ResourceBundle PHRASES =
71          ResourceBundle.getBundle("phrases");
72  
73      /**
74       * Github.
75       */
76      private final transient Github github;
77  
78      /**
79       * Ctor.
80       * @param ghub Github client
81       */
82      public CommentsTag(final Github ghub) {
83          super(
84              "/talk/wire[github-repo and github-issue]",
85              "/talk/request[@id and type='release' and success='true']"
86          );
87          this.github = ghub;
88      }
89  
90      @Override
91      public Iterable<Directive> process(final XML xml) throws IOException {
92          final XML req = xml.nodes("/talk/request").get(0);
93          final Issue.Smart issue = new TalkIssues(this.github, xml).get();
94          final String tag = req.xpath("args/arg[@name='tag']/text()").get(0);
95          final Releases.Smart rels = new Releases.Smart(issue.repo().releases());
96          final URI home = new Home(xml).uri();
97          if (rels.exists(tag)) {
98              final Release.Smart rel = new Release.Smart(rels.find(tag));
99              rel.body(
100                 String.format(
101                     "%s\n\nSee also #%d and [build log](%s)",
102                     rel.body(), issue.number(), home
103                 )
104             );
105             issue.comments().post(
106                 String.format(
107                     CommentsTag.PHRASES.getString("CommentsTag.duplicate"),
108                     tag
109                 )
110             );
111             Logger.info(this, "duplicate tag %s commented", tag);
112         } else {
113             final Repo repo = issue.repo();
114             final Date prev = CommentsTag.previous(repo);
115             final Release.Smart rel = new Release.Smart(
116                 rels.create(tag.trim())
117             );
118             rel.name(issue.title());
119             rel.prerelease(true);
120             rel.body(
121                 String.format(
122                     // @checkstyle LineLength (1 line)
123                     "See #%d, release log:\n\n%s\n\nReleased by Rultor %s, see [build log](%s)",
124                     issue.number(),
125                     new CommitsLog(repo).build(prev, rel.publishedAt()),
126                     Manifests.read("Rultor-Version"), home
127                 )
128             );
129             Logger.info(this, "tag %s created and commented", tag);
130         }
131         return new Directives();
132     }
133 
134     /**
135      * Get previous release time.
136      * @param repo Repo in which to find the releases.
137      * @return Previous release time or start of epoch.
138      * @throws IOException In case of problem communicating with repo.
139      */
140     private static Date previous(final Repo repo) throws IOException {
141         Date prev = new Date(0L);
142         final Iterable<Release.Smart> releases =
143             new Smarts<>(repo.releases().iterate());
144         for (final Release.Smart rel : releases) {
145             if (rel.json().isNull("published_at")) {
146                 continue;
147             }
148             if (prev.before(rel.publishedAt())) {
149                 prev = rel.publishedAt();
150             }
151         }
152         return prev;
153     }
154 }
