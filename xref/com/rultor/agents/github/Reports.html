49  /**
50   * Posts merge results to Github pull request.
51   *
52   * @author Yegor Bugayenko (yegor256@gmail.com)
53   * @version $Id: e1c84c0112bf06d54a0793e0c5d039c76b44994d $
54   * @since 1.0
55   */
56  @Immutable
57  @ToString
58  @EqualsAndHashCode(callSuper = false, of = "github")
59  public final class Reports extends AbstractAgent {
60  
61      /**
62       * Message bundle.
63       */
64      private static final ResourceBundle PHRASES =
65          ResourceBundle.getBundle("phrases");
66  
67      /**
68       * Github.
69       */
70      private final transient Github github;
71  
72      /**
73       * Ctor.
74       * @param ghub Github client
75       */
76      public Reports(final Github ghub) {
77          super(
78              "/talk/wire[github-repo and github-issue]",
79              "/talk/request[@id and success]"
80          );
81          this.github = ghub;
82      }
83  
84      @Override
85      @SuppressWarnings("PMD.AvoidCatchingGenericException")
86      public Iterable<Directive> process(final XML xml) throws IOException {
87          final XML req = xml.nodes("/talk/request").get(0);
88          final Issue.Smart issue = new TalkIssues(this.github, xml).get();
89          final boolean success = Boolean.parseBoolean(
90              req.xpath("success/text()").get(0)
91          );
92          final URI home = new Home(xml).uri();
93          final String pattern;
94          if (success) {
95              pattern = "Reports.success";
96          } else {
97              pattern = "Reports.failure";
98          }
99          final int number = Integer.parseInt(req.xpath("@id").get(0));
100         final Comment.Smart comment = new Comment.Smart(
101             new SafeComment(
102                 Reports.origin(issue, number)
103             )
104         );
105         final StringBuilder message = new StringBuilder();
106         if (comment.body().contains("stop")) {
107             message.append(Reports.PHRASES.getString("Reports.stop-fails"))
108                 .append(' ');
109         }
110         message.append(
111             Logger.format(
112                 Reports.PHRASES.getString(pattern),
113                 home.toASCIIString(),
114                 Long.parseLong(req.xpath("msec/text()").get(0))
115             )
116         ).append(Reports.highlights(req));
117         if (!success) {
118             message.append(Reports.tail(req));
119         }
120         new Answer(comment).post(success, message.toString());
121         Logger.info(this, "issue #%d reported: %B", issue.number(), success);
122         return new Directives()
123             .xpath("/talk/request[success]")
124             .strict(1).remove();
125     }
126 
127     /**
128      * Get highlights.
129      * @param req Request
130      * @return Highlights
131      */
132     private static String highlights(final XML req) {
133         final List<String> highlights = req.xpath("highlights/text()");
134         final String text;
135         if (highlights.isEmpty()) {
136             text = "";
137         } else {
138             text = String.format("\n\n%s", highlights.get(0));
139         }
140         return text;
141     }
142 
143     /**
144      * Get tail.
145      * @param req Request
146      * @return Tail
147      */
148     private static String tail(final XML req) {
149         final List<String> tail = req.xpath("tail/text()");
150         final String text;
151         if (tail.isEmpty()) {
152             text = "";
153         } else {
154             text = String.format("\n\n```\n%s\n```", tail.get(0));
155         }
156         return text;
157     }
158 
159     /**
160      * Get a comment we're answering to.
161      * @param issue The issue
162      * @param number Its number
163      * @return Comment
164      */
165     private static Comment.Smart origin(final Issue.Smart issue,
166         final int number) {
167         final Comment comment;
168         if (number == 1) {
169             comment = new FirstComment(issue);
170         } else {
171             comment = issue.comments().get(number);
172         }
173         return new Comment.Smart(comment);
174     }
175 
176 }
