57  /**
58   * Starts talk when I'm mentioned in a Github issue.
59   *
60   * @author Yegor Bugayenko (yegor@teamed.io)
61   * @version $Id: 6fc947194ebb303181024a2949e2c1f78df8ac28 $
62   * @since 1.0
63   */
64  @Immutable
65  @ToString
66  @EqualsAndHashCode(of = "github")
67  public final class StartsTalks implements SuperAgent {
68  
69      /**
70       * Github.
71       */
72      private final transient Github github;
73  
74      /**
75       * Ctor.
76       * @param ghub Github client
77       */
78      public StartsTalks(final Github ghub) {
79          this.github = ghub;
80      }
81  
82      @Override
83      @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
84      public void execute(final Talks talks) throws IOException {
85          final String since = new Time(
86              DateUtils.addMinutes(new Date(), -Tv.THREE)
87          ).iso();
88          final Request req = this.github.entry()
89              .uri().path("/notifications").back();
90          final Iterable<JsonObject> events = new RtPagination<>(
91              req.uri().queryParam("participating", "true")
92                  .queryParam("since", since)
93                  .queryParam("all", Boolean.toString(true))
94                  .back(),
95              RtPagination.COPYING
96          );
97          final Collection<String> names = new LinkedList<>();
98          for (final JsonObject event : events) {
99              final String reason = event.getString("reason");
100             if ("mention".equals(reason)) {
101                 names.add(this.activate(talks, event));
102             }
103         }
104         req.uri()
105             .queryParam("last_read_at", since).back()
106             .method(Request.PUT)
107             .body().set("{}").back()
108             .fetch()
109             .as(RestResponse.class)
110             .assertStatus(HttpURLConnection.HTTP_RESET);
111         Logger.info(
112             this, "%d new notification(s): %[list]s",
113             names.size(), names
114         );
115     }
116 
117     /**
118      * Activate talk.
119      * @param talks Talks
120      * @param event Event
121      * @return Name of the talk activated
122      * @throws IOException If fails
123      */
124     private String activate(final Talks talks, final JsonObject event)
125         throws IOException {
126         final Coordinates coords = this.coords(event);
127         final Issue issue = this.github.repos().get(coords).issues().get(
128             Integer.parseInt(
129                 StringUtils.substringAfterLast(
130                     event.getJsonObject("subject").getString("url"),
131                     "/"
132                 )
133             )
134         );
135         final String name = String.format("%s#%d", coords, issue.number());
136         if (!talks.exists(name)) {
137             talks.create(coords.toString(), name);
138         }
139         final Talk talk = talks.get(name);
140         talk.modify(
141             new Directives()
142                 .xpath("/talk").attr("later", Boolean.toString(true))
143                 .xpath("/talk[not(wire)]")
144                 .add("wire").add("href")
145                 .set(new Issue.Smart(issue).htmlUrl().toString())
146                 .up()
147                 .add("github-repo").set(coords.toString())
148                 .up()
149                 .add("github-issue")
150                 .set(Integer.toString(issue.number()))
151         );
152         talk.active(true);
153         Logger.info(
154             this, "talk %s#%d activated as %s",
155             coords, issue.number(), name
156         );
157         return talk.name();
158     }
159 
160     /**
161      * Get coordinates from JSON.
162      * @param event Event
163      * @return Coords
164      */
165     private Coordinates coords(final JsonObject event) {
166         return new Coordinates.Simple(
167             event.getJsonObject("repository").getString("full_name")
168         );
169     }

171 }
