30  package com.rultor.agents.docker;
31  
32  import com.github.dockerjava.api.DockerClient;
33  import com.github.dockerjava.api.command.CreateContainerResponse;
34  import com.github.dockerjava.api.model.ExposedPort;
35  import com.github.dockerjava.api.model.Ports;
36  import com.github.dockerjava.core.DockerClientBuilder;
37  import com.github.dockerjava.core.command.BuildImageResultCallback;
38  import com.rultor.agents.shells.PfShell;
39  import com.rultor.spi.Profile;
40  import java.io.File;
41  import java.io.IOException;
42  import java.io.StringWriter;
43  import java.util.Collection;
44  import java.util.Collections;
45  import java.util.concurrent.ConcurrentHashMap;
46  import org.apache.commons.io.IOUtils;
47  
48  /**
49   * Starts a Docker Container containing a Docker daemon and SSHD.
50   * @author Armin Braun (me@obrown.io)
51   * @version $Id: 32e6797ba8bc5ce016b82731f7cf9fcb1b030e10 $
52   * @since 1.63
53   */
54  public final class StartsDockerDaemon implements AutoCloseable {
55  
56      /**
57       * Profile used.
58       */
59      private final transient Profile profile;
60  
61      /**
62       * Docker client.
63       */
64      private final transient DockerClient client;
65  
66      /**
67       * Docker Containers created.
68       */
69      private final transient Collection<CreateContainerResponse> containers;
70  
71      /**
72       * Ctor.
73       * @param prof Current Profile
74       */
75      public StartsDockerDaemon(final Profile prof) {
76          this.profile = prof;
77          this.client = DockerClientBuilder.getInstance().build();
78          this.containers = Collections.newSetFromMap(
79              // @checkstyle MagicNumber (1 line)
80              new ConcurrentHashMap<CreateContainerResponse, Boolean>(1, 0.9f, 1)
81          );
82      }
83  
84      /**
85       * Sets up a PfShell in which Rultor can function.
86       * @return PfShell loaded with credentials for new Rultor runner container
87       * @throws IOException on failure
88       */
89      public PfShell shell() throws IOException {
90          final ExposedPort ssh = ExposedPort.tcp(22);
91          final Ports ports = new Ports();
92          ports.bind(ssh, Ports.Binding(null));
93          final CreateContainerResponse container = this.client
94              .createContainerCmd(this.build()).withExposedPorts(
95                  ssh
96              ).withPortBindings(ports).exec();
97          this.containers.add(container);
98          this.client.startContainerCmd(container.getId()).exec();
99          return new PfShell(
100             this.profile,
101             this.client.infoCmd().exec().getName(),
102             this.client.inspectContainerCmd(container.getId())
103                 .exec().getNetworkSettings().getPorts().getBindings().values()
104                 .iterator().next()[0].getHostPort(),
105             "root",
106             this.key(container)
107         );
108     }
109 
110     @Override
111     public void close() throws IOException {
112         for (final CreateContainerResponse container : this.containers) {
113             this.client.killContainerCmd(container.getId()).exec();
114             this.client.removeContainerCmd(container.getId()).exec();
115         }
116         this.client.close();
117     }
118 
119     /**
120      * Retrieves SSH private key needed to connect to a given container.
121      * @param container Container from which to get the SSH key
122      * @return SSH private key
123      * @throws IOException on failure
124      */
125     private String key(final CreateContainerResponse container)
126         throws IOException {
127         final StringWriter writer = new StringWriter();
128         IOUtils.copy(
129             this.client.copyFileFromContainerCmd(
130                 container.getId(), "/root/.ssh/id_rsa"
131             ).exec(),
132             writer
133         );
134         final String key = writer.toString();
135         return key.substring(key.indexOf('-'), key.lastIndexOf('-') + 1);
136     }
137 
138     /**
139      * Builds a fresh Rultor runner base image on the Docker daemon.
140      * @return Image ID of the Rultor runner base image
141      */
142     private String build() {
143         return this.client.buildImageCmd(
144             new File(this.getClass().getResource("image").getPath())
145         ).exec(new BuildImageResultCallback()).awaitImageId();
146     }
147 
148 }
