62  @Immutable
63  @ToString
64  @EqualsAndHashCode(callSuper = false, of = "profile")
65  public final class StartsRequest extends AbstractAgent {
66  
67      /**
68       * Command types not requiring _head.sh.
69       */
70      private static final Collection<String> HEADLESS =
71          Collections.singleton("stop");
72      /**
73       * Default port value to be used with Decrypt.
74       */
75      private static final String DEFAULT_PORT = "80";
76      /**
77       * HTTP proxy port system property key.
78       */
79      private static final String PORT_KEY = "http.proxyPort";
80      /**
81       * HTTP proxy host system property key.
82       */
83      private static final String HOST_KEY = "http.proxyHost";
84  
85      /**
86       * Profile.
87       */
88      private final transient Profile profile;
89  
90      /**
91       * Ctor.
92       * @param prof Profile
93       */
94      public StartsRequest(final Profile prof) {
95          super(
96              "/talk/request[@id and type and not(success)]",
97              "/talk[not(daemon)]"
98          );
99          this.profile = prof;
100     }
101 
102     @Override
103     public Iterable<Directive> process(final XML xml) throws IOException {
104         final XML req = xml.nodes("//request").get(0);
105         final String type = req.xpath("type/text()").get(0);
106         final String hash = req.xpath("@id").get(0);
107         String script;
108         try {
109             script = this.script(
110                 req, type, xml.xpath("/talk/@name").get(0)
111             );
112             Logger.info(this, "request %s/%s started", type, hash);
113         } catch (final Profile.ConfigException ex) {
114             script = Logger.format(
115                 "cat <<EOT\n%[exception]s\nEOT\nexit -1", ex
116             );
117         }
118         return new Directives().xpath("/talk")
119             .add("daemon")
120             .attr("id", hash)
121             .add("title").set(type).up()
122             .add("script").set(script);
123     }
124 
125     /**
126      * Make a script.
127      * @param req Request
128      * @param type Its type
129      * @param name Name of talk
130      * @return Script
131      * @throws IOException If fails
132      */
133     @SuppressWarnings("unchecked")
134     private String script(final XML req, final String type, final String name)
135         throws IOException {
136         return Joiner.on('\n').join(
137             Iterables.concat(
138                 Iterables.transform(
139                     Sets.union(
140                         this.vars(req, type).entrySet(),
141                         Sets.newHashSet(
142                             Maps.immutableEntry(
143                                 "container",
144                                 name.replaceAll("[^a-zA-Z0-9_.-]", "_")
145                             )
146                         )
147                     ),
148                     new Function<Map.Entry<String, String>, String>() {
149                         @Override
150                         public String apply(
151                             final Map.Entry<String, String> input) {
152                             return String.format(
153                                 "%s=%s", input.getKey(),
154                                 input.getValue()
155                             );
156                         }
157                     }
158                 ),
159                 Collections.singleton(this.asRoot()),
160                 Iterables.filter(
161                     Collections.singleton(
162                         IOUtils.toString(
163                             this.getClass().getResourceAsStream("_head.sh"),
164                             CharEncoding.UTF_8
165                         )
166                     ),
167                     new Predicate<String>() {
168                         @Override
169                         public boolean apply(final String input) {
170                             return !StartsRequest.HEADLESS.contains(type);
171                         }
172                     }
173                 ),
174                 this.decryptor().commands(),
175                 Collections.singleton(
176                     IOUtils.toString(
177                         this.getClass().getResourceAsStream(
178                             String.format("%s.sh", type)
179                         ),
180                         CharEncoding.UTF_8
181                     )
182                 )
183             )
184         );
185     }
186 
187     /**
188      * Obtain proxy settings and create a Decrypt instance.
189      * @return Decrypt instance.
190      */
191     private Decrypt decryptor() {
192         return new Decrypt(
193             this.profile,
194             System.getProperty(HOST_KEY, ""),
195             Integer.parseInt(System.getProperty(PORT_KEY, DEFAULT_PORT))
196         );
197     }
198 
199     /**
200      * Get start script for as_root config.
201      * @return Script
202      * @throws IOException If fails
203      * @since 1.37
204      */
205     private String asRoot() throws IOException {
206         return String.format(
207             "as_root=%b",
208             !this.profile.read().nodes(
209                 "/p/entry[@key='docker']/entry[@key='as_root' and .='true']"
210             ).isEmpty()
211         );
212     }
213 
214     /**
215      * Get variables from script.
216      * @param req Request
217      * @param type Its type
218      * @return Vars
219      * @throws IOException If fails
220      */
221     private Map<String, String> vars(final XML req, final String type)
222         throws IOException {
223         final ImmutableMap.Builder<String, String> vars =
224             new ImmutableMap.Builder<String, String>();
225         for (final XML arg : req.nodes("args/arg")) {
226             vars.put(arg.xpath("@name").get(0), arg.xpath("text()").get(0));
227         }
228         final DockerRun docker = new DockerRun(
229             this.profile, String.format("/p/entry[@key='%s']", type)
230         );
231         vars.put("vars", docker.envs(vars.build()));
232         vars.put(
233             "image",
234             new Profile.Defaults(this.profile).text(
235                 "/p/entry[@key='docker']/entry[@key='image']",
236                 "yegor256/rultor"
237             )
238         );
239         vars.put(
240             "directory",
241             new Profile.Defaults(this.profile).text(
242                 "/p/entry[@key='docker']/entry[@key='directory']",
243                 ""
244             )
245         );
246         vars.put("scripts", docker.script());
247         return vars.build();
248     }
