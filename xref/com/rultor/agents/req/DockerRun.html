52  /**
53   * Docker run command.
54   *
55   * @author Yegor Bugayenko (yegor@teamed.io)
56   * @version $Id: fd36b97e3f303f0b630fd5885aac708fd9a34b1e $
57   * @since 1.0
58   * @checkstyle MultipleStringLiteralsCheck (500 lines)
59   */
60  @Immutable
61  @ToString
62  @EqualsAndHashCode(of = { "profile", "xpath" })
63  @SuppressWarnings("PMD.TooManyMethods")
64  final class DockerRun {
65  
66      /**
67       * Profile.
68       */
69      private final transient Profile profile;
70  
71      /**
72       * Node name.
73       */
74      private final transient String xpath;
75  
76      /**
77       * Ctor.
78       * @param prof Profile
79       * @param path Node name in profile XML, XPath
80       */
81      DockerRun(final Profile prof, final String path) {
82          this.profile = prof;
83          this.xpath = path;
84      }
85  
86      /**
87       * Make a script to run.
88       * @return Script
89       * @throws IOException If fails
90       */
91      public String script() throws IOException {
92          final Iterable<String> uninstall;
93          if (this.profile.read().nodes("/p/entry[@key='uninstall']").isEmpty()) {
94              uninstall = Collections.emptyList();
95          } else {
96              uninstall = Iterables.concat(
97                  Lists.newArrayList("function", "clean_up()", "{"),
98                  DockerRun.scripts(
99                      this.profile.read(), "/p/entry[@key='uninstall']"
100                 ),
101                 Lists.newArrayList("}", ";"),
102                 Lists.newArrayList("trap", "clean_up", "EXIT", ";")
103             );
104         }
105         return DockerRun.enlist(
106             Iterables.concat(
107                 uninstall,
108                 DockerRun.scripts(
109                     this.profile.read(), "/p/entry[@key='install']"
110                 ),
111                 DockerRun.scripts(this.node(), "entry[@key='script']")
112             )
113         );
114     }
115 
116     /**
117      * Make a list of env vars for docker.
118      * @param extra Extra vars
119      * @return Envs
120      * @throws IOException If fails
121      */
122     public String envs(final Map<String, String> extra) throws IOException {
123         return DockerRun.enlist(
124             Iterables.concat(
125                 DockerRun.envs(this.profile.read(), "/p/entry[@key='env']"),
126                 DockerRun.envs(this.node(), "entry[@key='env']"),
127                 Collections2.transform(
128                     extra.entrySet(),
129                     new Function<Map.Entry<String, String>, String>() {
130                         @Override
131                         public String apply(
132                             final Map.Entry<String, String> ent) {
133                             return String.format(
134                                 "--env=%s=%s", ent.getKey(), ent.getValue()
135                             );
136                         }
137                     }
138                 )
139             )
140         );
141     }
142 
143     /**
144      * Get xpath.
145      * @return XML
146      * @throws IOException If fails
147      */
148     private XML node() throws IOException {
149         final XML node;
150         final Collection<XML> nodes = this.profile.read().nodes(this.xpath);
151         if (nodes.isEmpty()) {
152             node = new XMLDocument("<x/>");
153         } else {
154             node = nodes.iterator().next();
155         }
156         return node;
157     }
158 
159     /**
160      * Make a list for bash.
161      * @param items Items
162      * @return Text for bash
163      */
164     private static String enlist(final Iterable<String> items) {
165         return String.format(
166             "( %s )",
167             Joiner.on(' ').join(
168                 Iterables.transform(
169                     items,
170                     new Function<String, String>() {
171                         @Override
172                         public String apply(final String input) {
173                             return SSH.escape(input);
174                         }
175                     }
176                 )
177             )
178         );
179     }
180 
181     /**
182      * Get items from XML.
183      * @param xml The XML
184      * @param path The XPath
185      * @return Items
186      */
187     private static Iterable<String> scripts(final XML xml, final String path) {
188         final Collection<String> items = new LinkedList<String>();
189         if (!xml.nodes(path).isEmpty()) {
190             final XML node = xml.nodes(path).get(0);
191             if (node.nodes("item").isEmpty()) {
192                 items.addAll(DockerRun.lines(node));
193             } else {
194                 for (final String cmd : node.xpath("item/text()")) {
195                     items.add(cmd.trim());
196                 }
197             }
198         }
199         final Collection<String> scripts = new LinkedList<String>();
200         for (final String item : items) {
201             scripts.add(neutralize(item));
202             scripts.add(";");
203         }
204         return scripts;
205     }
206 
207     /**
208      * Is hash character inside double or single quotes.
209      * @param item String to check.
210      * @param pos Position of the hash.
211      * @return If hash is in quotes.
212      */
213     private static boolean inquotes(final String item, final int pos) {
214         return StringUtils.countMatches(item.substring(0, pos), "\"") % 2 == 1
215             || StringUtils.countMatches(item.substring(0, pos), "'") % 2 == 1;
216     }
217 
218     /**
219      * Neutralize comment contained in the script line.
220      * @param item Script element.
221      * @return Script element with invisible comment
222      */
223     private static String neutralize(final String item) {
224         final int start = item.indexOf('#');
225         final String result;
226         if (start == 0 || start > 0 && item.charAt(start - 1) != '\\'
227             && !DockerRun.inquotes(item, start)) {
228             result = new StringBuilder(item.substring(0, start))
229                 .append('`')
230                 .append(item.substring(start))
231                 .append('`')
232                 .toString();
233         } else {
234             result = item;
235         }
236         return result;
237     }
238 
239     /**
240      * Environments from XML by XPath.
241      * @param xml The XML
242      * @param path The XPath
243      * @return Items
244      */
245     private static Iterable<String> envs(final XML xml, final String path) {
246         final Collection<String> envs = new LinkedList<>();
247         if (!xml.nodes(path).isEmpty()) {
248             final XML node = xml.nodes(path).get(0);
249             final Collection<String> parts;
250             if (node.nodes("item").iterator().hasNext()) {
251                 parts = node.xpath("item/text()");
252             } else if (node.nodes("entry").iterator().hasNext()) {
253                 parts = new LinkedList<>();
254                 for (final XML env : node.nodes("./entry")) {
255                     parts.add(
256                         String.format(
257                             "%s=%s", env.xpath("@key").get(0),
258                             env.xpath("text()").get(0)
259                         )
260                     );
261                 }
262             } else {
263                 parts = DockerRun.lines(node);
264             }
265             envs.addAll(
266                 Collections2.transform(
267                     parts,
268                     new Function<String, String>() {
269                         @Override
270                         public String apply(final String input) {
271                             return String.format("--env=%s", input);
272                         }
273                     }
274                 )
275             );
276         }
277         return envs;
278     }
279 
280     /**
281      * Get lines from a single XML node.
282      * @param node Node to get text() from
283      * @return Lines found
284      */
285     private static Collection<String> lines(final XML node) {
286         final Collection<String> lines = new LinkedList<String>();
287         if (node.node().hasChildNodes()) {
288             lines.addAll(
289                 Collections2.transform(
290                     Arrays.asList(
291                         StringUtils.split(node.xpath("text()").get(0), '\n')
292                     ),
293                     new Function<String, String>() {
294                         @Override
295                         public String apply(final String line) {
296                             return line.trim();
297                         }
298                     }
299                 )
300             );
301         }
302         return lines;
303     }
