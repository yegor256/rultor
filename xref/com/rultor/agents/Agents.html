113 @Immutable
114 @ToString
115 @EqualsAndHashCode(of = { "github", "sttc" })
116 @SuppressWarnings("PMD.ExcessiveImports")
117 public final class Agents {
118 
119     /**
120      * Github client.
121      */
122     private final transient Github github;
123 
124     /**
125      * Sttc client.
126      */
127     private final transient Sttc sttc;
128 
129     /**
130      * Ctor.
131      * @param ghub Github client
132      * @param stc Sttc client
133      */
134     public Agents(final Github ghub, final Sttc stc) {
135         this.github = ghub;
136         this.sttc = stc;
137     }
138 
139     /**
140      * Create super agent, starter.
141      * @return The starter
142      * @throws IOException If fails
143      */
144     public SuperAgent starter() throws IOException {
145         return new SuperAgent.Iterative(
146             new Array<>(
147                 new StartsTalks(this.github),
148                 new Invitations(this.github),
149                 new IndexesRequests(),
150                 new DockerExec(
151                     new SSH(
152                         // @checkstyle MagicNumber (1 line)
153                         "b3.rultor.com", 22,
154                         "rultor",
155                         IOUtils.toString(
156                             this.getClass().getResourceAsStream("rultor.key"),
157                             CharEncoding.UTF_8
158                         )
159                     ),
160                     "rmi.sh"
161                 )
162             )
163         );
164     }
165 
166     /**
167      * Create super agent, closer.
168      * @return The closer
169      * @throws IOException If fails
170      */
171     public SuperAgent closer() throws IOException {
172         return new SuperAgent.Iterative(
173             new Array<>(
174                 new UnlocksRepo(this.sttc.locks(), this.github),
175                 new DeactivatesTalks()
176             )
177         );
178     }
179 
180     /**
181      * Create it for a talk.
182      * @param talk Talk itself
183      * @param profile Profile
184      * @return The agent
185      * @throws IOException If fails
186      */
187     public Agent agent(final Talk talk, final Profile profile)
188         throws IOException {
189         final Locks locks = this.sttc.locks();
190         final Question question = new QnSince(
191             // @checkstyle MagicNumber (1 line)
192             49092213,
193             new QnNotSelf(
194                 new QnReferredTo(
195                     this.github.users().self().login(),
196                     new QnParametrized(
197                         new QnFollow(
198                             new QnFirstOf(
199                                 new QnIfContains(
200                                     "config", new QnConfig(profile)
201                                 ),
202                                 new QnIfContains("status", new QnStatus(talk)),
203                                 new QnIfContains("version", new QnVersion()),
204                                 new QnIfContains("hello", new QnHello()),
205                                 new QnIfContains(
206                                     "stop",
207                                     new QnAskedBy(
208                                         profile,
209                                         Agents.commanders("stop"),
210                                         new QnStop()
211                                     )
212                                 ),
213                                 new QnIfCollaborator(
214                                     new QnAlone(
215                                         talk, locks,
216                                         Agents.commands(profile)
217                                     )
218                                 ),
219                                 new QnIamLost()
220                             )
221                         )
222                     )
223                 )
224             )
225         );
226         return new Agent.Iterative(
227             new SanitizesDaemon(),
228             new WipesDaemon(),
229             new Understands(
230                 this.github,
231                 new QnSafe(question)
232             ),
233             new StartsRequest(profile),
234             new RegistersShell(
235                 profile,
236                 // @checkstyle MagicNumber (1 line)
237                 "b3.rultor.com", 22,
238                 "rultor",
239                 IOUtils.toString(
240                     this.getClass().getResourceAsStream("rultor.key"),
241                     CharEncoding.UTF_8
242                 )
243             ),
244             new StartsDaemon(profile),
245             new KillsDaemon(TimeUnit.HOURS.toMinutes(2L)),
246             new StopsDaemon(),
247             new EndsDaemon(),
248             new EndsRequest(),
249             new SafeAgent(
250                 new Tweets(
251                     this.github,
252                     new OAuthTwitter(
253                         Manifests.read("Rultor-TwitterKey"),
254                         Manifests.read("Rultor-TwitterSecret"),
255                         Manifests.read("Rultor-TwitterToken"),
256                         Manifests.read("Rultor-TwitterTokenSecret")
257                     )
258                 )
259             ),
260             new SafeAgent(
261                 new HnUpdates(
262                     this.github,
263                     new HttpHackerNews(
264                         Manifests.read("Rultor-HNUser"),
265                         Manifests.read("Rultor-HNPassword")
266                     )
267                 )
268             ),
269             new CommentsTag(this.github),
270             new ReleaseBinaries(this.github, profile),
271             new Dephantomizes(this.github),
272             new Reports(this.github),
273             new RemovesShell(),
274             new ArchivesDaemon(
275                 new ReRegion(
276                     new Region.Simple(
277                         Manifests.read("Rultor-S3Key"),
278                         Manifests.read("Rultor-S3Secret")
279                     )
280                 ).bucket(Manifests.read("Rultor-S3Bucket"))
281             ),
282             new Publishes(profile),
283             new SafeAgent(new Stars(this.github))
284         );
285     }
286 
287     /**
288      * Handle main commands.
289      * @param profile Profile to uuse
290      * @return Array of questions.
291      */
292     private static Question commands(final Profile profile) {
293         return new QnByArchitect(
294             profile,
295             "/p/entry[@key='architect']/item/text()",
296             new QnFirstOf(
297                 new QnIfContains(
298                     "unlock",
299                     new QnUnlock()
300                 ),
301                 new QnIfContains(
302                     "lock",
303                     new QnLock()
304                 ),
305                 new QnIfContains(
306                     "merge",
307                     new QnAskedBy(
308                         profile,
309                         Agents.commanders("merge"),
310                         new QnIfPull(new QnIfUnlocked(new QnMerge()))
311                     )
312                 ),
313                 new QnIfContains(
314                     "deploy",
315                     new QnAskedBy(
316                         profile,
317                         Agents.commanders("deploy"),
318                         new QnDeploy()
319                     )
320                 ),
321                 new QnIfContains(
322                     "release",
323                     new QnAskedBy(
324                         profile,
325                         Agents.commanders("release"),
326                         new QnRelease()
327                     )
328                 )
329             )
330         );
331     }
332 
333     /**
334      * XPath for commanders.
335      * @param entry Entry
336      * @return XPath
337      */
338     private static String commanders(final String entry) {
339         return String.format(
340             "/p/entry[@key='%s']/entry[@key='commanders']/item/text()",
341             entry
342         );
343     }
