55  /**
56   * Talks in Dynamo.
57   *
58   * @author Yegor Bugayenko (yegor@teamed.io)
59   * @version $Id: 66132c07941730282683ae61005c5929c79a98eb $
60   * @since 1.0
61   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
62   */
63  @Immutable
64  @ToString
65  @EqualsAndHashCode(of = "region")
66  @SuppressWarnings("PMD.TooManyMethods")
67  public final class DyTalks implements Talks {
68  
69      /**
70       * Table name.
71       */
72      public static final String TBL = "talks";
73  
74      /**
75       * Index name.
76       * @checkstyle MultipleStringLiteralsCheck (3 lines)
77       */
78      public static final String IDX_ACTIVE = "active";
79  
80      /**
81       * Index name.
82       * @since 1.3
83       */
84      public static final String IDX_NUMBERS = "numbers";
85  
86      /**
87       * Index name.
88       * @since 1.23
89       */
90      public static final String IDX_SIBLINGS = "siblings";
91  
92      /**
93       * Talk unique name.
94       */
95      public static final String HASH = "name";
96  
97      /**
98       * Its number.
99       * @since 1.3
100      */
101     public static final String ATTR_NUMBER = "number";
102 
103     /**
104      * Name of the repo it belongs to.
105      * @since 1.23
106      */
107     public static final String ATTR_REPO = "repo";
108 
109     /**
110      * Is it active (1) or archived (0).
111      */
112     public static final String ATTR_ACTIVE = "active";
113 
114     /**
115      * XML of the talk.
116      */
117     public static final String ATTR_XML = "xml";
118 
119     /**
120      * XML of the talk, gzip.
121      */
122     public static final String ATTR_XML_ZIP = "zipxml";
123 
124     /**
125      * When updated.
126      */
127     public static final String ATTR_UPDATED = "updated";
128 
129     /**
130      * Region we're in.
131      */
132     private final transient Region region;
133 
134     /**
135      * Counter of talks.
136      */
137     private final transient Counter counter;
138 
139     /**
140      * Public ctor.
141      * @param reg Region
142      * @param cnt Counter of talks
143      */
144     public DyTalks(final Region reg, final Counter cnt) {
145         this.region = reg;
146         this.counter = cnt;
147     }
148 
149     @Override
150     public boolean exists(final long number) {
151         return this.region.table(DyTalks.TBL)
152             .frame()
153             .through(
154                 new QueryValve()
155                     .withLimit(1)
156                     .withIndexName(DyTalks.IDX_NUMBERS)
157                     .withConsistentRead(false)
158             )
159             .where(DyTalks.ATTR_NUMBER, Conditions.equalTo(number))
160             .iterator().hasNext();
161     }
162 
163     @Override
164     public Talk get(final long number) {
165         return new DyTalk(
166             this.region.table(DyTalks.TBL)
167                 .frame()
168                 .through(
169                     new QueryValve()
170                         .withLimit(1)
171                         .withIndexName(DyTalks.IDX_NUMBERS)
172                         .withConsistentRead(false)
173                         .withSelect(Select.SPECIFIC_ATTRIBUTES)
174                         .withAttributesToGet(DyTalks.HASH, DyTalks.ATTR_NUMBER)
175                 )
176                 .where(DyTalks.ATTR_NUMBER, Conditions.equalTo(number))
177                 .iterator().next()
178         );
179     }
180 
181     @Override
182     public boolean exists(final String name) {
183         return this.region.table(DyTalks.TBL)
184             .frame()
185             .through(new QueryValve().withLimit(1))
186             .where(DyTalks.HASH, name)
187             .iterator().hasNext();
188     }
189 
190     @Override
191     public Talk get(final String name) {
192         return new DyTalk(
193             this.region.table(DyTalks.TBL)
194                 .frame()
195                 .through(
196                     new QueryValve()
197                         .withLimit(1)
198                         .withAttributesToGet(DyTalks.ATTR_NUMBER)
199                 )
200                 .where(DyTalks.HASH, name)
201                 .iterator().next()
202         );
203     }
204 
205     @Override
206     public void delete(final String name) {
207         Iterables.removeIf(
208             this.region.table(DyTalks.TBL)
209                 .frame()
210                 .through(new QueryValve().withLimit(1))
211                 .where(DyTalks.HASH, name),
212             Predicates.alwaysTrue()
213         );
214     }
215 
216     @Override
217     public void create(final String repo, final String name)
218         throws IOException {
219         final long number = this.counter.incrementAndGet(1L);
220         this.region.table(DyTalks.TBL).put(
221             new Attributes()
222                 .with(DyTalks.HASH, name)
223                 .with(DyTalks.ATTR_ACTIVE, Boolean.toString(true))
224                 .with(DyTalks.ATTR_REPO, repo)
225                 .with(DyTalks.ATTR_NUMBER, number)
226                 .with(DyTalks.ATTR_UPDATED, System.currentTimeMillis())
227                 .with(
228                     DyTalks.ATTR_XML,
229                     String.format("<talk name='%s' number='%d'/>", name, number)
230                 )
231         );
232     }
233 
234     @Override
235     public Iterable<Talk> active() {
236         return Iterables.transform(
237             this.region.table(DyTalks.TBL)
238                 .frame()
239                 .through(
240                     new QueryValve()
241                         .withIndexName(DyTalks.IDX_ACTIVE)
242                         .withConsistentRead(false)
243                         .withSelect(Select.SPECIFIC_ATTRIBUTES)
244                         .withAttributesToGet(DyTalks.HASH, DyTalks.ATTR_NUMBER)
245                 )
246                 .where(DyTalks.ATTR_ACTIVE, Boolean.toString(true)),
247             new Function<Item, Talk>() {
248                 @Override
249                 public Talk apply(final Item input) {
250                     return new DyTalk(input);
251                 }
252             }
253         );
254     }
255 
256     @Override
257     public Iterable<Talk> recent() {
258         return Iterables.limit(
259             Iterables.filter(
260                 Iterables.transform(
261                     this.region.table(DyTalks.TBL)
262                         .frame()
263                         .through(
264                             new QueryValve()
265                                 .withIndexName(DyTalks.IDX_ACTIVE)
266                                 .withScanIndexForward(false)
267                                 .withConsistentRead(false)
268                                 .withLimit(Tv.FIVE)
269                                 .withSelect(Select.ALL_PROJECTED_ATTRIBUTES)
270                         )
271                         .where(
272                             DyTalks.ATTR_ACTIVE, Boolean.toString(false)
273                         ),
274                     new Function<Item, Talk>() {
275                         @Override
276                         public Talk apply(final Item input) {
277                             return new DyTalk(input);
278                         }
279                     }
280                 ),
281                 new Predicate<Talk>() {
282                     @Override
283                     public boolean apply(final Talk talk) {
284                         try {
285                             return !talk.read().nodes(
286                                 "/talk[@public='true']"
287                             ).isEmpty();
288                         } catch (final IOException ex) {
289                             throw new IllegalStateException(ex);
290                         }
291                     }
292                 }
293             ),
294             Tv.FIVE
295         );
296     }
297 
298     @Override
299     public Iterable<Talk> siblings(final String repo, final Date since) {
300         return Iterables.transform(
301             this.region.table(DyTalks.TBL)
302                 .frame()
303                 .through(
304                     new QueryValve()
305                         .withIndexName(DyTalks.IDX_SIBLINGS)
306                         .withScanIndexForward(false)
307                         .withConsistentRead(false)
308                         .withLimit(Tv.TWENTY)
309                         .withSelect(Select.ALL_PROJECTED_ATTRIBUTES)
310                 )
311                 .where(DyTalks.ATTR_REPO, repo)
312                 .where(
313                     DyTalks.ATTR_UPDATED,
314                     new Condition()
315                         .withComparisonOperator(ComparisonOperator.LT)
316                         .withAttributeValueList(
317                             new AttributeValue().withN(
318                                 Long.toString(since.getTime())
319                             )
320                         )
321                 ),
322             new Function<Item, Talk>() {
323                 @Override
324                 public Talk apply(final Item input) {
325                     return new DyTalk(input);
326                 }
327             }
328         );
329     }
