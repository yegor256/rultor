59  /**
60   * Drain in a single S3 object.
61   *
62   * @author Yegor Bugayenko (yegor@tpc2.com)
63   * @version $Id$
64   * @since 1.0
65   * @checkstyle ClassDataAbstractionCoupling (500 lines)
66   */
67  @Immutable
68  @ToString
69  @EqualsAndHashCode(of = { "client", "key" })
70  @Loggable(Loggable.DEBUG)
71  public final class ObjectDrain implements Drain {
72  
73      /**
74       * S3 client.
75       */
76      private final transient S3Client client;
77  
78      /**
79       * S3 object name.
80       */
81      private final transient String key;
82  
83      /**
84       * Public ctor.
85       * @param clnt S3 client
86       * @param name S3 object name
87       */
88      public ObjectDrain(
89          @NotNull(message = "S3 client can't be NULL") final S3Client clnt,
90          @NotNull(message = "object name can't be NULL") final String name) {
91          this.client = clnt;
92          Validate.matchesPattern(
93              name, "([^/]+/)*[^/]+", "invalid S3 object name '%s'", name
94          );
95          this.key = name;
96      }

98      /**
99       * {@inheritDoc}
100      */
101     @Override
102     public Pageable<Time, Time> pulses() {
103         return new Pageable.Array<Time>();
104     }

106     /**
107      * {@inheritDoc}
108      */
109     @Override
110     public void append(final Iterable<String> lines) throws IOException {
111         final AmazonS3 aws = this.client.get();
112         final InputStream body;
113         long size;
114         if (aws.listObjects(this.client.bucket(), this.key)
115             .getObjectSummaries().isEmpty()) {
116             body = IOUtils.toInputStream("");
117             size = 0;
118         } else {
119             final S3Object object =
120                 aws.getObject(this.client.bucket(), this.key);
121             body = object.getObjectContent();
122             size = object.getObjectMetadata().getContentLength();
123         }
124         final byte[] suffix = new StringBuilder(StringUtils.join(lines, "\n"))
125             .append('\n')
126             .toString()
127             .getBytes(CharEncoding.UTF_8);
128         Validate.isTrue(suffix.length > 0, "empty input");
129         size += suffix.length;
130         final ObjectMetadata meta = new ObjectMetadata();
131         meta.setContentEncoding(CharEncoding.UTF_8);
132         meta.setContentType(MediaType.TEXT_PLAIN);
133         meta.setContentLength(size);
134         try {
135             final PutObjectResult result = aws.putObject(
136                 this.client.bucket(),
137                 this.key,
138                 new SequenceInputStream(
139                     body,
140                     new ByteArrayInputStream(suffix)
141                 ),
142                 meta
143             );
144             Logger.info(
145                 this,
146                 "`%s` saved %s to S3, etag=`%s`",
147                 this.key,
148                 FileUtils.byteCountToDisplaySize(size),
149                 result.getETag()
150             );
151         } catch (AmazonS3Exception ex) {
152             throw new IOException(
153                 String.format(
154                     "failed to flush %s to `%s`: %s",
155                     this.key,
156                     this.client.bucket(),
157                     ex
158                 ),
159                 ex
160             );
161         }
162     }

164     /**
165      * {@inheritDoc}
166      */
167     @Override
168     public InputStream read() throws IOException {
169         final AmazonS3 aws = this.client.get();
170         InputStream stream;
171         try {
172             if (aws.listObjects(this.client.bucket(), this.key)
173                 .getObjectSummaries().isEmpty()) {
174                 throw new IOException(
175                     String.format(
176                         "S3 object %s not found in %s",
177                         this.key,
178                         this.client.bucket()
179                     )
180                 );
181             }
182             final S3Object object =
183                 aws.getObject(this.client.bucket(), this.key);
184             Logger.info(
185                 this,
186                 "'%s' ready for loading from S3, size=%s, etag=%s",
187                 this.key,
188                 FileUtils.byteCountToDisplaySize(
189                     object.getObjectMetadata().getContentLength()
190                 ),
191                 object.getObjectMetadata().getETag()
192             );
193             stream = new SequenceInputStream(
194                 IOUtils.toInputStream(
195                     String.format(
196                         "ObjectDrain: etag='%s', size=%s\n\n",
197                         object.getObjectMetadata().getETag(),
198                         FileUtils.byteCountToDisplaySize(
199                             object.getObjectMetadata().getContentLength()
200                         )
201                     ),
202                     CharEncoding.UTF_8
203                 ),
204                 new ObjectDrain.Wrap(object.getObjectContent(), aws)
205             );
206         } catch (AmazonS3Exception ex) {
207             throw new IOException(
208                 String.format(
209                     "failed to read %s from %s: %s",
210                     this.key,
211                     this.client.bucket(),
212                     ex
213                 ),
214                 ex
215             );
216         }
217         return new SequenceInputStream(
218             IOUtils.toInputStream(
219                 String.format(
220                     "ObjectDrain: key='%s', client='%s'\n",
221                     this.key, this.client
222                 ),
223                 CharEncoding.UTF_8
224             ),
225             stream
226         );
227     }

229     /**
230      * Wrap around exiting InputStream, to keep AWS object apart from
231      * garbage collector, in order not to loose HTTP connection.
232      */
233     private static final class Wrap extends BufferedInputStream {
234         /**
235          * Encapsulated AmazonS3 client.
236          */
237         private final transient AmazonS3 aws;
238         /**
239          * Public ctor.
240          * @param stream Origin input stream
241          * @param amazon AmazonS3
242          */
243         protected Wrap(final InputStream stream, final AmazonS3 amazon) {
244             super(stream);
245             this.aws = amazon;
246         }
247     }
