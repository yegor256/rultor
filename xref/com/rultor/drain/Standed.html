58  /**
59   * Mirrored to a web {@link Stand}.
60   *
61   * @author Yegor Bugayenko (yegor@tpc2.com)
62   * @version $Id$
63   * @since 1.0
64   * @checkstyle ClassDataAbstractionCoupling (500 lines)
65   */
66  @Immutable
67  @EqualsAndHashCode(of = { "origin", "work", "stand", "key" })
68  @Loggable(Loggable.DEBUG)
69  public final class Standed implements Drain {
70  
71      /**
72       * Work we're in.
73       */
74      private final transient Work work;
75  
76      /**
77       * Original drain.
78       */
79      private final transient Drain origin;
80  
81      /**
82       * Name of stand.
83       */
84      private final transient String stand;
85  
86      /**
87       * Secret key of it.
88       */
89      private final transient String key;
90  
91      /**
92       * Public ctor.
93       * @param wrk Work we're in
94       * @param name Name of stand
95       * @param secret Secret key of the stand
96       * @param drain Main drain
97       * @checkstyle ParameterNumber (8 lines)
98       */
99      public Standed(
100         @NotNull(message = "work can't be NULL") final Work wrk,
101         @NotNull(message = "name of stand can't be NULL") final String name,
102         @NotNull(message = "key can't be NULL") final String secret,
103         @NotNull(message = "drain can't be NULL") final Drain drain) {
104         this.work = wrk;
105         this.stand = name;
106         this.key = secret;
107         this.origin = drain;
108     }
109 
110     /**
111      * {@inheritDoc}
112      */
113     @Override
114     public String toString() {
115         return String.format(
116             "%s standed at `%s`",
117             this.origin, this.stand
118         );
119     }
120 
121     /**
122      * {@inheritDoc}
123      */
124     @Override
125     public Pageable<Time, Time> pulses() throws IOException {
126         return this.origin.pulses();
127     }
128 
129     /**
130      * {@inheritDoc}
131      */
132     @Override
133     public void append(final Iterable<String> lines) throws IOException {
134         for (String line : lines) {
135             if (XemblyLine.existsIn(line)) {
136                 try {
137                     this.send(XemblyLine.parse(line).xembly());
138                 } catch (XemblySyntaxException ex) {
139                     Logger.warn(this, ExceptionUtils.getRootCauseMessage(ex));
140                 }
141             }
142         }
143         this.origin.append(lines);
144     }
145 
146     /**
147      * {@inheritDoc}
148      */
149     @Override
150     public InputStream read() throws IOException {
151         return new SequenceInputStream(
152             IOUtils.toInputStream(
153                 String.format(
154                     "Standed: stand='%s'\n",
155                     this.stand
156                 ),
157                 CharEncoding.UTF_8
158             ),
159             this.origin.read()
160         );
161     }
162 
163     /**
164      * Send line to stand.
165      * @param xembly The xembly script
166      * @throws IOException If fails
167      */
168     private void send(final String xembly) throws IOException {
169         final StringWriter writer = new StringWriter();
170         Json.createGenerator(writer)
171             .writeStartObject()
172             .write("nano", System.nanoTime())
173             .write("stand", this.stand)
174             .write("key", this.key)
175             .write("xembly", xembly)
176             .writeStartObject("work")
177             .write("owner", this.work.owner().toString())
178             .write("unit", this.work.unit())
179             .write("scheduled", this.work.scheduled().toString())
180             .writeEnd()
181             .writeEnd()
182             .close();
183         final String body = String.format(
184             "Action=SendMessage&Version=2011-10-01&MessageBody=%s",
185             URLEncoder.encode(writer.toString(), CharEncoding.UTF_8)
186         );
187         try {
188             RestTester
189                 .start(Stand.QUEUE)
190                 .header(HttpHeaders.CONTENT_ENCODING, CharEncoding.UTF_8)
191                 .header(
192                     HttpHeaders.CONTENT_LENGTH,
193                     body.getBytes(CharEncoding.UTF_8).length
194                 )
195                 .header(
196                     HttpHeaders.CONTENT_TYPE,
197                     MediaType.APPLICATION_FORM_URLENCODED
198                 )
199                 .post("sending one line to stand SQS queue", body)
200                 .assertStatus(HttpURLConnection.HTTP_OK);
201         } catch (AssertionError ex) {
202             throw new IOException(ex);
203         }
204     }
