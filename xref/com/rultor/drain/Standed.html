200     @Override
201     public void append(final Iterable<String> lines) throws IOException {
202         final Iterable<List<String>> batches = Iterables.partition(
203             FluentIterable.from(lines)
204                 .filter(
205                     new Predicate<String>() {
206                         @Override
207                         public boolean apply(final String line) {
208                             return XemblyLine.existsIn(line);
209                         }
210                     }
211                 )
212                 .transform(
213                     new Function<String, String>() {
214                         @Override
215                         public String apply(final String line) {
216                             try {
217                                 return XemblyLine.parse(line).xembly();
218                             } catch (SyntaxException ex) {
219                                 Exceptions.warn(this, ex);
220                             }
221                             return null;
222                         }
223                     }
224                 )
225                 .filter(Predicates.notNull()), Standed.MAX
226         );
227         for (List<String> batch : batches) {
228             this.send(batch);
229         }
230         this.origin.append(lines);
231     }

247     /**
248      * Send lines to stand.
249      * @param xemblies The xembly scripts
250      * @throws IOException If fails
251      */
252     private void send(final Iterable<String> xemblies) throws IOException {
253         this.exec.get().submit(
254             new VerboseRunnable(
255                 new Callable<Void>() {
256                     @Override
257                     public Void call() throws Exception {
258                         Standed.this.send(Standed.this.body(xemblies));
259                         return null;
260                     }
261                 },
262                 true, false
263             )
264         );
265     }

267     /**
268      * Send the message.
269      * @param body POST request body
270      * @throws IOException If fails
271      */
272     @RetryOnFailure(verbose = false, attempts = Tv.TEN)
273     private void send(final String body) throws IOException {
274         final Collection<String> missed = Standed.this.enqueue(
275             this.entry.get(), body
276         );
277         if (!missed.isEmpty()) {
278             throw new IOException(
279                 String.format(
280                     "Problem with sending of %d message(s): %s",
281                     missed.size(), StringUtils.join(missed, ", ")
282                 )
283             );
284         }
285     }

287     /**
288      * Create POST request body.
289      * @param xemblies The xembly scripts
290      * @return POST request body
291      * @throws IOException If fails
292      */
293     private String body(final Iterable<String> xemblies) throws IOException {
294         final List<String> msgs = new ArrayList<String>(0);
295         for (String xembly : xemblies) {
296             msgs.add(this.json(xembly));
297         }
298         final StringBuilder body = new StringBuilder()
299             .append("Action=SendMessageBatch")
300             .append("&Version=2011-10-01");
301         for (int idx = 0; idx < msgs.size(); ++idx) {
302             final int ent = idx + 1;
303             body.append('&')
304                 .append("SendMessageBatchRequestEntry.")
305                 .append(ent)
306                 .append(".Id=")
307                 .append(ent)
308                 .append("&SendMessageBatchRequestEntry.")
309                 .append(ent)
310                 .append(".MessageBody=")
311                 .append(URLEncoder.encode(msgs.get(idx), CharEncoding.UTF_8));
312         }
313         return body.toString();
314     }

316     /**
317      * Put messages on SQS.
318      * @param client HTTP client to use.
319      * @param body Body of the POST.
320      * @return List of message IDs that were not enqueued.
321      * @throws UnsupportedEncodingException When unable to find UTF-8 encoding.
322      */
323     private List<String> enqueue(final Request client, final String body)
324         throws UnsupportedEncodingException {
325         try {
326             return client.header(
327                 HttpHeaders.CONTENT_ENCODING, CharEncoding.UTF_8
328             )
329                 .header(
330                     HttpHeaders.CONTENT_LENGTH,
331                     body.getBytes(CharEncoding.UTF_8).length
332                 )
333                 .header(
334                     HttpHeaders.CONTENT_TYPE,
335                     MediaType.APPLICATION_FORM_URLENCODED
336                 )
337                 .method(Request.POST)
338                 .body()
339                 .set(body)
340                 .back()
341                 .fetch()
342                 .as(RestResponse.class)
343                 .assertStatus(HttpURLConnection.HTTP_OK)
344                 .as(XmlResponse.class)
345                 .xml()
346                 // @checkstyle LineLength (1 line)
347                 .xpath("/SendMessageBatchResponse/BatchResultError/BatchResultErrorEntry/Id/text()");
348         } catch (IOException ex) {
349             throw new IllegalStateException(ex);
350         }
351     }

353     /**
354      * Create a JSON representation of xembly.
355      * @param xembly Xembly to change into JSON.
356      * @return JSON of xembly.
357      */
358     private String json(final String xembly) {
359         final StringWriter writer = new StringWriter();
360         final long nano = System.nanoTime() * Tv.HUNDRED
361             + Standed.RND.nextInt(Tv.HUNDRED);
362         Json.createGenerator(writer)
363             .writeStartObject()
364             .write("stand", this.stand)
365             .write("nano", nano)
366             .write("key", this.key)
367             .write("xembly", xembly)
368             .writeStartObject("work")
369             .write("owner", this.work.owner().toString())
370             .write("rule", this.work.rule())
371             .write("scheduled", this.work.scheduled().toString())
372             .writeEnd()
373             .writeEnd()
374             .close();
375         return writer.toString();
376     }
