55  /**
56   * Temporary in memory.
57   *
58   * @author Yegor Bugayenko (yegor@tpc2.com)
59   * @version $Id$
60   * @since 1.0
61   * @checkstyle ClassDataAbstractionCoupling (500 lines)
62   */
63  @Immutable
64  @EqualsAndHashCode(of = { "work", "marker" })
65  @Loggable(Loggable.DEBUG)
66  @SuppressWarnings({ "PMD.DoNotUseThreads", "PMD.TooManyMethods" })
67  public final class Temporary implements Drain {
68  
69      /**
70       * Maximum lifetime allowed, in milliseconds.
71       */
72      private static final long LIFETIME = TimeUnit.HOURS.toMillis(2);
73  
74      /**
75       * All in-memory buffers.
76       */
77      private static final ConcurrentMap<Temporary, Temporary.Buffer> BUFFERS =
78          new ConcurrentHashMap<Temporary, Temporary.Buffer>(0);
79  
80      /**
81       * Cleaner.
82       */
83      private static final Runnable CLEANER = new Temporary.Cleaner();
84  
85      /**
86       * Work we're in.
87       */
88      private final transient Work work;
89  
90      /**
91       * Optional marker.
92       */
93      private final transient String marker;
94  
95      /**
96       * Public ctor.
97       * @param wrk Work we're in
98       * @param mrk Optional marker
99       */
100     public Temporary(
101         @NotNull(message = "work can't be NULL") final Work wrk,
102         @NotNull(message = "marker can't be NULL") final String mrk) {
103         assert Temporary.CLEANER != null;
104         this.work = wrk;
105         this.marker = mrk;
106     }
107 
108     /**
109      * Public ctor.
110      * @param wrk Work we're in
111      */
112     public Temporary(final Work wrk) {
113         this(wrk, "def");
114     }
115 
116     /**
117      * {@inheritDoc}
118      */
119     @Override
120     public String toString() {
121         return String.format("temporary marked as `%s`", this.marker);
122     }
123 
124     /**
125      * {@inheritDoc}
126      */
127     @Override
128     public Pulses pulses() throws IOException {
129         final Collection<Time> times = new LinkedList<Time>();
130         for (Temporary client : Temporary.BUFFERS.keySet()) {
131             if (this.similar(client)) {
132                 times.add(client.work.started());
133             }
134         }
135         return new Pulses.Array(times);
136     }
137 
138     /**
139      * {@inheritDoc}
140      */
141     @Override
142     public void append(final Iterable<String> lines) throws IOException {
143         final Temporary.Buffer buffer;
144         synchronized (Temporary.BUFFERS) {
145             if (!Temporary.BUFFERS.containsKey(this)) {
146                 Temporary.BUFFERS.put(
147                     this,
148                     new Temporary.Buffer()
149                 );
150             }
151             buffer = Temporary.BUFFERS.get(this);
152         }
153         buffer.append(lines);
154     }
155 
156     /**
157      * {@inheritDoc}
158      */
159     @Override
160     public InputStream read() throws IOException {
161         final Temporary.Buffer buffer = Temporary.BUFFERS.get(this);
162         if (buffer == null) {
163             throw new IOException(
164                 String.format(
165                     "temporary buffer is absent for '%s' in '%s'",
166                     this.marker,
167                     this.work
168                 )
169             );
170         }
171         return new SequenceInputStream(
172             IOUtils.toInputStream(
173                 Logger.format(
174                     "Temporary: marker='%s', work='%s', buffer='%s'\n",
175                     this.marker,
176                     this.work,
177                     buffer
178                 )
179             ),
180             buffer.read()
181         );
182     }
183 
184     /**
185      * Thread-safe buffer to the real drain.
186      */
187     private final class Buffer {
188         /**
189          * When was is started.
190          */
191         private final transient Long start = System.currentTimeMillis();
192         /**
193          * Buffered data.
194          */
195         private final transient ByteArrayOutputStream data =
196             new ByteArrayOutputStream();
197         /**
198          * {@inheritDoc}
199          */
200         @Override
201         public String toString() {
202             return Logger.format(
203                 "age=%[ms]s, size=%d",
204                 System.currentTimeMillis() - this.start,
205                 this.data.size()
206             );
207         }
208         /**
209          * Append lines to the buffer.
210          * @param lines Lines to append
211          */
212         public void append(final Iterable<String> lines) {
213             synchronized (this.start) {
214                 final PrintWriter writer = new PrintWriter(this.data);
215                 for (String line : lines) {
216                     writer.println(line);
217                 }
218                 writer.flush();
219                 writer.close();
220             }
221         }
222         /**
223          * Read it as a stream.
224          * @return The stream
225          */
226         public InputStream read() {
227             return new ByteArrayInputStream(this.data.toByteArray());
228         }
229         /**
230          * Expired already?
231          * @return TRUE if it is too old
232          */
233         public boolean expired() {
234             return System.currentTimeMillis() - this.start
235                 > Temporary.LIFETIME;
236         }
237     }
238 
239     /**
240      * This drain is similar to the one provided?
241      * @param drain Drain to compare with
242      * @return TRUE if they are similar
243      */
244     private boolean similar(final Temporary drain) {
245         return this.work.owner().equals(drain.work.owner())
246             && this.work.unit().equals(drain.work.unit())
247             && this.marker.equals(drain.marker);
248     }
249 
250     /**
251      * Flush.
252      */
253     @ScheduleWithFixedDelay(delay = 1, unit = TimeUnit.SECONDS)
254     private static final class Cleaner implements Runnable {
255         @Override
256         public void run() {
257             for (Temporary client : Temporary.BUFFERS.keySet()) {
258                 if (Temporary.BUFFERS.get(client).expired()) {
259                     Temporary.BUFFERS.remove(client);
260                 }
261             }
262         }
263     }
