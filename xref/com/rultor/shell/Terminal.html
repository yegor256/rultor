33  import com.jcabi.aspects.Loggable;
34  import com.jcabi.log.Logger;
35  import java.io.ByteArrayOutputStream;
36  import java.io.IOException;
37  import java.util.logging.Level;
38  import javax.validation.constraints.NotNull;
39  import lombok.EqualsAndHashCode;
40  import lombok.ToString;
41  import org.apache.commons.codec.CharEncoding;
42  import org.apache.commons.io.IOUtils;
43  import org.apache.commons.io.output.TeeOutputStream;

45  /**
46   * Simplified terminal.
47   *
48   * @author Yegor Bugayenko (yegor@tpc2.com)
49   * @version $Id$
50   * @since 1.0
51   */
52  @Immutable
53  @ToString
54  @EqualsAndHashCode(of = "shell")
55  @Loggable(Loggable.DEBUG)
56  public final class Terminal {
57  
58      /**
59       * Shell to use.
60       */
61      private final transient Shell shell;
62  
63      /**
64       * Public ctor.
65       * @param shl Shell to use
66       */
67      public Terminal(@NotNull(message = "shell can't be NULL") final Shell shl) {
68          this.shell = shl;
69      }
70  
71      /**
72       * Escape argument.
73       * @param arg Argument
74       * @return Escaped
75       * @todo #34 This implementation is extremely bad
76       */
77      public static String escape(@NotNull(message = "argument can't be NULL")
78          final String arg) {
79          return arg
80              .replace("\\", "\\\\")
81              .replace("\"", "\\\"")
82              .replace("$", "\\$")
83              .replace("!", "\"'!'\"")
84              .replace("`", "\\`");
85      }
86  
87      /**
88       * Quotate argument.
89       * @param arg Argument
90       * @return Quotated
91       */
92      public static String quotate(@NotNull(message = "argument can't be NULL")
93          final String arg) {
94          return new StringBuilder()
95              .append('"')
96              .append(arg)
97              .append('"')
98              .toString();
99      }
100 
101     /**
102      * Run this command and return its output (fail on non-zero exit code).
103      * @param command The command
104      * @return Output stream
105      * @throws IOException If some IO problem inside
106      */
107     @Loggable(value = Loggable.DEBUG, limit = Integer.MAX_VALUE)
108     public String exec(@NotNull(message = "command can't be NULL")
109         final String command) throws IOException {
110         return this.exec(command, "");
111     }
112 
113     /**
114      * Run this command and return its output (fail on non-zero exit code).
115      * @param command The command
116      * @param stdin Input stream
117      * @return Output stream
118      * @throws IOException If some IO problem inside
119      */
120     @Loggable(value = Loggable.DEBUG, limit = Integer.MAX_VALUE)
121     public String exec(
122         @NotNull(message = "command can't be NULL") final String command,
123         @NotNull(message = "stdin can't be NULL") final String stdin)
124         throws IOException {
125         final ByteArrayOutputStream stdout = new ByteArrayOutputStream();
126         final ByteArrayOutputStream stderr = new ByteArrayOutputStream();
127         final int code = this.shell.exec(
128             command,
129             IOUtils.toInputStream(stdin, CharEncoding.UTF_8),
130             new TeeOutputStream(stdout, Logger.stream(Level.INFO, this)),
131             new TeeOutputStream(stderr, Logger.stream(Level.WARNING, this))
132         );
133         if (code != 0) {
134             throw new IOException(
135                 Logger.format(
136                     // @checkstyle LineLength (1 line)
137                     "non-zero exit code %d after executing `%s` at %s, stdout='%[text]s', stderr='%s'",
138                     code, command, this.shell,
139                     stdout.toString(CharEncoding.UTF_8),
140                     stderr.toString(CharEncoding.UTF_8)
141                 )
142             );
143         }
144         return stdout.toString(CharEncoding.UTF_8);
145     }
