147      * Constructor.

149      * @param prt Port of server
150      * @param user Login
151      * @param priv Private SSH key
152      * @checkstyle ParameterNumberCheck (6 lines)

154     public SSHChannel(
155         @NotNull(message = "IP address can't be NULL") final InetAddress adr,
156         final int prt,
157         @NotNull(message = "user name can't be NULL") final String user,
158         @NotNull(message = "private key can't be NULL") final PrivateKey priv) {
159         this.addr = adr.getHostAddress();
160         Validate.matchesPattern(
161             this.addr,

163             "Invalid IP address of the server `%s`",
164             this.addr
165         );
166         this.login = user;
167         Validate.notEmpty(this.login, "user name can't be empty");
168         this.key = priv;
169         this.port = prt;

186     /**
187      * {@inheritDoc}
188      * @checkstyle ParameterNumber (10 lines)
189      */
190     @Override
191     @Loggable(value = Loggable.DEBUG, limit = 1, unit = TimeUnit.HOURS)
192     public int exec(@NotNull(message = "command can't be NULL")
193         final String command,
194         @NotNull(message = "stdin can't be NULL") final InputStream stdin,
195         @NotNull(message = "stdout can't be NULL") final OutputStream stdout,
196         @NotNull(message = "stderr can't be NULL") final OutputStream stderr)
197         throws IOException {
198         try {
199             final Session session = this.session();
200             try {
201                 final ChannelExec channel = ChannelExec.class.cast(
202                     session.openChannel("exec")
203                 );
204                 channel.setErrStream(stderr, false);
205                 channel.setOutputStream(stdout, false);
206                 channel.setInputStream(stdin, false);
207                 channel.setCommand(command);
208                 channel.connect();
209                 Logger.info(this, "$ %s", command);
210                 return this.exec(channel, session);
211             } finally {
212                 session.disconnect();
213             }
214         } catch (JSchException ex) {
215             throw new IOException(ex);
216         }
217     }

235     /**
236      * Exec this channel and return its exit code.
237      * @param channel The channel to exec
238      * @param session The session
239      * @return Exit code (zero in case of success)
240      * @throws IOException If fails
241      */
242     private int exec(final ChannelExec channel, final Session session)
243         throws IOException {
244         try {
245             return this.code(channel, session);
246         } finally {
247             channel.disconnect();
248         }
249     }

251     /**
252      * Wait until it's done and return its code.
253      * @param exec The channel
254      * @param session The session
255      * @return The exit code
256      * @throws IOException If some IO problem inside
257      */
258     @SuppressWarnings("PMD.AvoidCatchingGenericException")
259     private int code(final ChannelExec exec, final Session session)
260         throws IOException {
261         while (!exec.isClosed()) {
262             try {
263                 session.sendKeepAliveMsg();
264             // @checkstyle IllegalCatch (1 line)
265             } catch (Exception ex) {
266                 throw new IOException(ex);
267             }
268             try {
269                 TimeUnit.SECONDS.sleep(1);
270             } catch (InterruptedException ex) {
271                 Thread.currentThread().interrupt();
272                 throw new IOException(ex);
273             }
274         }
275         return exec.getExitStatus();
276     }

278     /**
279      * Create and return a session, connected.
280      * @return JSch session
281      * @throws IOException If some IO problem inside
282      */
283     @RetryOnFailure(
284         attempts = Tv.TEN,
285         delay = 1,
286         unit = TimeUnit.MINUTES,
287         verbose = false
288     )
289     private Session session() throws IOException {
290         try {
291             JSch.setConfig("StrictHostKeyChecking", "no");
292             JSch.setLogger(SSHChannel.LOGGER);
293             final JSch jsch = new JSch();
294             final File file = this.key.asFile();
295             jsch.setHostKeyRepository(SSHChannel.REPO);
296             jsch.addIdentity(file.getAbsolutePath());
297             Logger.info(
298                 this,
299                 "Opening SSH session to %s@%s:%s (%d bytes in RSA key)...",
300                 this.login, this.addr, this.port,
301                 file.length()
302             );
303             final Session session = jsch.getSession(
304                 this.login, this.addr, this.port
305             );
306             session.setServerAliveInterval((int) TimeUnit.SECONDS.toMillis(1));
307             session.setServerAliveCountMax(Tv.MILLION);
308             session.connect();
309             return session;
310         } catch (JSchException ex) {
311             throw new IOException(ex);
312         }
313     }
