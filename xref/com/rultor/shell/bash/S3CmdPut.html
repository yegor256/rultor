45  /**
46   * Put file(s) using s3cmd command line tool.
47   *
48   * @author Yegor Bugayenko (yegor@tpc2.com)
49   * @version $Id$
50   * @since 1.0
51   * @see <a href="http://s3tools.org/s3cmd">s3cmd</a>
52   */
53  @Immutable
54  @ToString
55  @EqualsAndHashCode(of = { "name", "path", "bucket", "prefix", "key", "secret" })
56  @Loggable(Loggable.DEBUG)
57  public final class S3CmdPut implements Sequel {
58  
59      /**
60       * Tag name.
61       */
62      private final transient String name;
63  
64      /**
65       * Path to files, like "./alpha/files/*.txt".
66       */
67      private final transient String path;
68  
69      /**
70       * S3 bucket.
71       */
72      private final transient String bucket;
73  
74      /**
75       * S3 prefix in the bucket.
76       */
77      private final transient String prefix;
78  
79      /**
80       * S3 key.
81       */
82      private final transient String key;
83  
84      /**
85       * S3 secret.
86       */
87      private final transient String secret;
88  
89      /**
90       * Content type.
91       */
92      private final transient String contentType;
93  
94      /**
95       * Encoding.
96       */
97      private final transient String encoding;

99      /**
100      * Public ctor.
101      * @param label Name of the product to discover
102      * @param pth Path to use
103      * @param bkt Bucket name
104      * @param pfx Prefix in S3 bucket
105      * @param akey S3 authorization key
106      * @param scrt S3 authorization secret

108      */
109     public S3CmdPut(final String label, final String pth, final String bkt,

112     }

114     /**
115      * Public ctor.
116      * @param label Name of the product to discover
117      * @param pth Path to use
118      * @param bkt Bucket name
119      * @param pfx Prefix in S3 bucket
120      * @param akey S3 authorization key
121      * @param scrt S3 authorization secret
122      * @param type Content type
123      * @param enc Encoding
124      * @checkstyle ParameterNumber (8 lines)
125      */
126     public S3CmdPut(
127         @NotNull(message = "name can't be NULL") final String label,
128         @NotNull(message = "path can't be NULL") final String pth,
129         @NotNull(message = "bucket can't be NULL") final String bkt,
130         @NotNull(message = "prefix can't be NULL") final String pfx,
131         @NotNull(message = "key can't be NULL") final String akey,
132         @NotNull(message = "secret can't be NULL") final String scrt,
133         @NotNull(message = "content type can't be NULL") final String type,
134         @NotNull(message = "encoding can't be NULL") final String enc) {
135         Validate.notBlank(label, "S3 label can't be empty");
136         this.name = label;
137         Validate.notBlank(bkt, "S3 bucket can't be empty");
138         this.bucket = bkt;
139         Validate.notBlank(pth, "search path can't be empty");
140         this.path = pth;
141         this.prefix = pfx;
142         Validate.matchesPattern(
143             akey, "[0-9A-Z]{20}",
144             "wrong format of AWS key '%s'", akey
145         );
146         this.key = akey;
147         Validate.matchesPattern(
148             scrt, "[0-9A-Za-z=/\\+]{40}",
149             "wrong format of AWS secret '%s'", scrt
150         );
151         this.secret = scrt;
152         Validate.notBlank(type, "content type can't be empty");
153         this.contentType = type;
154         Validate.notBlank(enc, "encoding can't be empty");
155         this.encoding = enc;
156     }

158     @Override
159     public void exec(final Shell shell) throws IOException {
160         final String dir = FilenameUtils.getFullPathNoEndSeparator(this.path);
161         Validate.notBlank(dir, "directory part of path is empty");
162         final String mask = FilenameUtils.getName(this.path);
163         Validate.notBlank(mask, "file part of path is empty");
164         final String url = String.format(
165             "http://%s.s3.amazonaws.com/%s", this.bucket, this.prefix
166         );
167         final int files = new Terminal(shell).exec(
168             new StringBuilder()
169                 .append("CONFIG=$(mktemp /tmp/s3cmdput-XXXX)")
170                 .append(" && cat > $CONFIG")
171                 .append(" && HEAD=")
172                 .append(
173                     Terminal.quotate(
174                         Terminal.escape(
175                             String.format(
176                                 "s3://%s/%s",
177                                 this.bucket,
178                                 this.prefix
179                             )
180                         )
181                     )
182                 )
183                 .append(" && cd ")
184                 .append(Terminal.quotate(Terminal.escape(dir)))
185                 .append(" && if ls -U ")
186                 .append(mask)
187                 .append(" >/dev/null 2>&1; then")
188                 .append(" FILES=$(find ")
189                 .append(mask)
190                 .append(" -type f)")
191                 .append(" && for f in $FILES; do")
192                 .append(" s3cmd --config=$CONFIG put $f \"$HEAD$f\" >/dev/null")
193                 .append(" && echo $f; done;")
194                 .append(" else echo No files found by mask ")
195                 .append(Terminal.quotate(Terminal.escape(mask)))
196                 .append(" 1>&2; fi;")
197                 .append("rm -f $CONFIG")
198                 .toString(),
199             new StringBuilder()
200                 .append("[default]\n")
201                 .append("access_key=").append(this.key).append('\n')
202                 .append("secret_key=").append(this.secret).append('\n')
203                 .append("encoding=").append(this.encoding).append('\n')
204                 .append("mime-type=").append(this.contentType).append('\n')
205                 .toString()
206         ).split("\n").length;
207         if (mask.contains("*")) {
208             final String href = String.format("%sindex.html", url);
209             this.tag(String.format("see [%d files](%s)", files, href), href);
210         } else {
211             final String href = String.format("%s%s", url, mask);
212             this.tag(String.format("see [%s](%s) file", mask, href), href);
213         }
214     }

216     /**
217      * Log a tag.
218      * @param desc Markdown description
219      * @param href URL of content
220      * @throws IOException If fails
221      */
222     private void tag(final String desc, final String href) throws IOException {
223         new TagLine(this.name)
224             .markdown(desc)
225             .fine(true)
226             .attr("href", href)
227             .log();
228     }
