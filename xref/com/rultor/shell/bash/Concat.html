52  /**
53   * Smart concatenator of bash commands.
54   *
55   * @author Yegor Bugayenko (yegor@tpc2.com)
56   * @version $Id$
57   * @since 1.0
58   * @checkstyle ClassDataAbstractionCoupling (500 lines)
59   */
60  @Immutable
61  @ToString
62  @EqualsAndHashCode(of = "commands")
63  @Loggable(Loggable.DEBUG)
64  public final class Concat implements Proxy<String> {
65  
66      /**
67       * Escaped Xembly mark.
68       */
69      private static final String ESCAPED_MARK =
70          Concat.escape(XemblyLine.MARK);
71  
72      /**
73       * Number of lines to take from head of stderr.
74       */
75      private static final int HEAD = 25;
76  
77      /**
78       * Number of lines to take from tail of stderr.
79       */
80      private static final int TAIL = 100;
81  
82      /**
83       * Bash commands to execute, one by one in provided order.
84       */
85      private final transient Array<Vext> commands;
86  
87      /**
88       * Public ctor.
89       * @param cmds Bash commands to encapsulate
90       */
91      @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
92      public Concat(
93          @NotNull(message = "list of commands can't be NULL")
94          final Collection<String> cmds) {
95          final Collection<Vext> vexts = new ArrayList<Vext>(cmds.size());
96          for (String cmd : cmds) {
97              vexts.add(new Vext(cmd));
98          }
99          this.commands = new Array<Vext>(vexts);
100     }

102     /**
103      * {@inheritDoc}
104      */
105     @Override
106     public String object() {
107         return this.script().velocity();
108     }

110     /**
111      * Make a script using custom arguments provided.
112      *
113      * <p>The method returns a Vext script that is a composition of
114      * all commands encapsulated. This script when executed through BASH
115      * outputs Xembly log lines that, when parsed, produce a {@link Snapshot}
116      * with steps. Every step will have a summary, a log level, and
117      * an exception if its exit code is not zero.
118      *
119      * @return Bash script ready for execution
120      */
121     private Vext script() {
122         final StringBuilder script = new StringBuilder()
123             .append("#set($dollar='$')")
124             .append("set -o pipefail;\n")
125             .append("set +o histexpand;\n")
126             .append("ESCAPE=")
127             .append(Terminal.quotate(Terminal.escape(Concat.escape())))
128             .append(';').append('\n');
129         for (Vext cmd : this.commands) {
130             script.append(this.script(cmd));
131         }
132         return new Vext(script.toString());
133     }

135     /**
136      * Make a script of one BASH command.
137      *
138      * <p>The method converts one command in Vext format
139      * to a script ready for execution in bash.
140      *
141      * @param cmd Command in Vext format
142      * @return Bash script ready for execution
143      * @see http://stackoverflow.com/questions/18665603
144      */
145     private String script(final Vext cmd) {
146         final String uid = String.format("bash-%d", System.nanoTime());
147         final String velocity = StringUtils.strip(cmd.velocity(), " ;");
148         final String command = cmd.velocity();
149         return new StringBuilder()
150             .append("echo; echo ${dollar} ")
151             .append(Terminal.quotate(Terminal.escape(command)))
152             .append(';').append('\n')
153             .append(
154                 this.echo(
155                     new Directives()
156                         .xpath("/snapshot")
157                         .addIf("steps")
158                         .add("step")
159                         .attr("id", uid)
160                         .add("start")
161                         .set("${dollar}(date  -u +%Y-%m-%dT%H:%M:%SZ)")
162                 )
163             )
164             .append(';').append('\n')
165             .append(this.echo(uid, command))
166             .append(";\nSTART=${dollar}(date +%s%N | tr -d N);\n")
167             .append("STDERR=${dollar}(mktemp /tmp/bash-XXXX);\n")
168             .append("{ ")
169             .append(velocity)
170             .append(
171                 "; } 2> >(tee ${dollar}STDERR);\n"
172             )
173             .append("CODE=${dollar}?;\n")
174             .append("sync; wait;\n")
175             .append("FINISH=${dollar}(date +%s%N | tr -d N);\n")
176             .append("if [ ${dollar}CODE = 0 ]; then\n  ")
177             .append(
178                 this.echo(
179                     new Directives()
180                         .xpath(this.xpath(uid))
181                         // @checkstyle MultipleStringLiterals (1 line)
182                         .add("level")
183                         .set(Level.INFO.toString())
184                 )
185             )
186             .append(";\nelse\n")
187             .append("  ERRLEN=${dollar}(cat ${dollar}STDERR | wc -l);\n")
188             .append("  if [ ${dollar}ERRLEN -gt ")
189             .append(Concat.HEAD + Concat.TAIL)
190             .append(" ]; then\n")
191             .append("    HEAD=${dollar}(head -")
192             .append(Concat.HEAD)
193             .append(" ${dollar}STDERR | eval ${dollar}ESCAPE);\n")
194             .append("    BREAK='... '${dollar}((ERRLEN - ")
195             .append(Concat.HEAD + Concat.TAIL)
196             .append("))' lines skipped ...&#10;';\n")
197             .append("    TAIL=${dollar}(tail -")
198             .append(Concat.TAIL)
199             .append(" ${dollar}STDERR | eval ${dollar}ESCAPE); \n")
200             .append("    MSG=${dollar}HEAD${dollar}BREAK${dollar}TAIL;\n")
201             .append("  else\n")
202             // @checkstyle LineLength (1 line)
203             .append("    MSG=${dollar}(cat ${dollar}STDERR | eval ${dollar}ESCAPE);\n")
204             .append("  fi;\n")
205             .append(
206                 this.echo(
207                     new Directives()
208                         .xpath(this.xpath(uid))
209                         .add("level")
210                         .set(Level.SEVERE.toString())
211                         .up()
212                         .add("exception")
213                         .add("cause")
214                         .set("exit code ${dollar}CODE")
215                         .up()
216                         .add("stacktrace")
217                         .set("${dollar}MSG")
218                 )
219             )
220             .append(";\nfi;\n")
221             .append(
222                 this.echo(
223                     new Directives()
224                         .xpath(this.xpath(uid))
225                         .add("finish")
226                         .set("${dollar}(date -u +%Y-%m-%dT%H:%M:%SZ)")
227                         .up()
228                         .add("duration")
229                         .set("${dollar}(((FINISH-START)/1000000))")
230                 )
231             )
232             .append(";\n")
233             .append("rm -f ${dollar}STDERR;\n")
234             .append("if [ ${dollar}CODE != 0 ]; then\n  ")
235             .append("exit ${dollar}CODE;\nfi;\n\n")
236             .toString();
237     }

239     /**
240      * Make bash command that ECHO Xembly log line with directives.
241      *
242      * <p>The method creates a bash ECHO command that outputs Xembly
243      * directives provided, properly escaping them before.
244      *
245      * @param dirs Xembly directives to be echoed
246      * @return Bash command
247      */
248     private String echo(final Directives dirs) {
249         return String.format(
250             "echo -e \"%s\"",
251             new XemblyLine(dirs)
252                 .toString()
253                 // @checkstyle MultipleStringLiterals (2 lines)
254                 .replace("\\", "\\\\\\")
255                 .replace("\"", "\\\"")
256                 .replace(XemblyLine.MARK, Concat.ESCAPED_MARK)
257         );
258     }
259 
260     /**
261      * Echo step summary in Xembly format.
262      *
263      * <p>The method creates a bash command that echos step summary
264      * in Xembly format. The Xembly directives echoed will add summary
265      * to the previously created step, finding it by its unique ID
266      * provided as the first parameter.
267      *
268      * @param uid Unique ID of the step
269      * @param summary Summary to add to the step
270      * @return Bash command
271      */
272     private String echo(final String uid, final String summary) {
273         final String xembly = new XemblyLine(
274             new Directives()
275                 .xpath(this.xpath(uid))
276                 .add("summary")
277                 .set(summary.replaceAll("([_*`\\\\])", "\\\\$1"))
278         ).toString();
279         return String.format(
280             "echo -e '%s'",
281             xembly.replace("'", "\\x27").replace(
282                 XemblyLine.MARK, Concat.ESCAPED_MARK
283             )
284         );
285     }

287     /**
288      * Escape Unicode chars for bash command.
289      * @param text Original text
290      * @return Escaped
291      */
292     private static String escape(final String text) {
293         final StringBuilder out = new StringBuilder();
294         for (byte chr : text.getBytes(Charsets.UTF_8)) {
295             out.append("\\x").append(String.format("%X", chr));
296         }
297         return out.toString();
298     }

300     /**
301      * XPath to find the step.
302      * @param name Unique ID of the step
303      * @return XPath for xembly
304      */
305     private String xpath(final String name) {
306         return String.format("/snapshot/steps/step[@id='%s']", name);
307     }

309     /**
310      * Escaping bash script.
311      *
312      * <p>The method creates a bash command that escapes sensitive characters
313      * in its standard input stream (stdin) and outputs a clean text as
314      * its output stream (stdout).
315      *
316      * @return Bash script
317      */
318     @Cacheable(forever = true)
319     private static String escape() {
320         final ImmutableMap<String, String> pairs =
321             new ImmutableMap.Builder<String, String>()
322                 .put("\\&", "&amp;")
323                 .put("'\"'\"'", "&apos;")
324                 .put("\"", "&quot;")
325                 .put("<", "&lt;")
326                 .put(">", "&gt;")
327                 .build();
328         final StringBuilder script = new StringBuilder().append("cat");
329         for (Map.Entry<String, String> pair : pairs.entrySet()) {
330             script.append(" | LANG=C sed -e 's/")
331                 .append(pair.getKey())
332                 .append("/\\")
333                 .append(pair.getValue())
334                 .append("/g'");
335         }
336         return script.append(" | awk 1 ORS='&#10;'")
337             .append(" | LANG=C tr -cd '[:print:]'")
338             .toString();
339     }
