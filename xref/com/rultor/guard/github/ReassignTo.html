32  import com.jcabi.aspects.Immutable;
33  import com.jcabi.aspects.Loggable;
34  import com.jcabi.log.Logger;
35  import com.rultor.snapshot.Step;
36  import java.io.IOException;
37  import javax.validation.constraints.NotNull;
38  import lombok.EqualsAndHashCode;
39  import lombok.ToString;
40  import org.eclipse.egit.github.core.Issue;
41  import org.eclipse.egit.github.core.PullRequest;
42  import org.eclipse.egit.github.core.User;
43  import org.eclipse.egit.github.core.client.RequestException;
44  import org.eclipse.egit.github.core.service.IssueService;
45  
46  /**
47   * Reassigns to user.
48   * @author Bharath Bolisetty (bharathbolisetty@gmail.com)
49   * @author Yegor Bugayenko (yegor@tpc2.com)
50   * @version $Id$
51   */
52  @Immutable
53  @ToString
54  @EqualsAndHashCode(of = "user")
55  @Loggable(Loggable.DEBUG)
56  public final class ReassignTo implements Approval {
57  
58      /**
59       * Login to reassign to.
60       */
61      private final transient String user;
62  
63      /**
64       * Public ctor.
65       * @param login Login to assign to
66       */
67      public ReassignTo(
68          @NotNull(message = "login can not be null") final String login) {
69          this.user = login;
70      }
71  
72      @Override
73      @Step("assigned pull request #${args[0].number} to `${args[0].user.login}`")
74      public boolean has(final PullRequest request, final Github client,
75          final Github.Repo repo) throws IOException {
76          final IssueService svc = new IssueService(client.client());
77          final Issue issue = svc.getIssue(repo, request.getNumber());
78          User assignee = issue.getAssignee();
79          if (assignee == null) {
80              assignee = new User();
81              assignee.setLogin("");
82          }
83          if (!assignee.getLogin().equals(this.user)) {
84              assignee.setLogin(this.user);
85              issue.setAssignee(assignee);
86              try {
87                  svc.editIssue(repo, issue);
88              } catch (final RequestException ex) {
89                  if ("Server Error (500)".equals(ex.getMessage())) {
90                      Logger.warn(this, "strange error on Github side: %s", ex);
91                  } else {
92                      throw ex;
93                  }
94              }
95          }
96          return true;
97      }
98  
99  }
