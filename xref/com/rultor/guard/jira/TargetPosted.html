32  import com.google.common.collect.Iterables;
33  import com.jcabi.aspects.Immutable;
34  import com.jcabi.aspects.Loggable;
35  import com.rultor.ext.jira.JiraComment;
36  import com.rultor.ext.jira.JiraIssue;
37  import com.rultor.guard.MergeRequest;
38  import com.rultor.scm.Branch;
39  import java.io.IOException;
40  import java.net.URI;
41  import java.util.regex.Matcher;
42  import java.util.regex.Pattern;
43  import lombok.EqualsAndHashCode;
44  import lombok.ToString;
45  
46  /**
47   * Uni-direct merging.
48   *
49   * @author Yegor Bugayenko (yegor@tpc2.com)
50   * @version $Id$
51   * @since 1.0
52   */
53  @Immutable
54  @ToString
55  @EqualsAndHashCode(of = { "msg", "uri" })
56  @Loggable(Loggable.DEBUG)
57  public final class TargetPosted implements Refinement {
58  
59      /**
60       * Destination branch pattern.
61       */
62      private final transient String msg;
63  
64      /**
65       * URI of destination.
66       */
67      private final transient String uri;
68  
69      /**
70       * Public ctor.
71       * @param ptn Pattern for the message to search
72       * @param scm URI of SCM
73       */
74      public TargetPosted(final String ptn, final String scm) {
75          this.msg = ptn;
76          this.uri = scm;
77      }
78  
79      @Override
80      public MergeRequest refine(final MergeRequest request,
81          final JiraIssue issue) {
82          // @checkstyle AnonInnerLength (50 lines)
83          return new MergeRequest() {
84              @Override
85              public String name() {
86                  return request.name();
87              }
88              @Override
89              public Branch source() {
90                  return request.source();
91              }
92              @Override
93              public Branch destination() {
94                  return new Branch.Passive(
95                      URI.create(TargetPosted.this.uri),
96                      TargetPosted.this.branch(issue)
97                  );
98              }
99              @Override
100             public void started() throws IOException {
101                 TargetPosted.this.branch(issue);
102                 request.started();
103             }
104             @Override
105             public void accept() throws IOException {
106                 request.accept();
107             }
108             @Override
109             public void reject() throws IOException {
110                 request.reject();
111             }
112         };
113     }
114     /**
115      * Get destination branch from the issue.
116      * @param issue The issue
117      * @return Destination branch name
118      */
119     private String branch(final JiraIssue issue) {
120         final Iterable<JiraComment> comments = issue.comments();
121         if (Iterables.isEmpty(comments)) {
122             issue.revert(
123                 // @checkstyle LineLength (1 line)
124                 "Please, tell me which branch you want it to be merge into"
125             );
126             throw new IllegalStateException(
127                 String.format(
128                     "destination branch is not specified in %s", issue
129                 )
130             );
131         }
132         final String body = comments.iterator().next().body();
133         final Matcher matcher = Pattern.compile(this.msg).matcher(body);
134         if (!matcher.matches()) {
135             issue.revert(
136                 String.format(
137                     "I didn't get what you said here: \"%s\"", body
138                 )
139             );
140             throw new IllegalStateException(
141                 String.format(
142                     "message is not parseable: '%s'", body
143                 )
144             );
145         }
146         return matcher.group(1);
147     }
