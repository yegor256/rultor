50  /**
51   * On pull request.
52   *
53   * @author Yegor Bugayenko (yegor@tpc2.com)
54   * @version $Id$
55   * @since 1.0
56   */
57  @Immutable
58  @EqualsAndHashCode(of = { "requests", "busy", "batch" })
59  @Loggable(Loggable.DEBUG)
60  public final class OnPullRequest implements Instance {
61  
62      /**
63       * All available pull requests.
64       */
65      private final transient MergeRequests requests;
66  
67      /**
68       * List of requests we're busy with at the moment.
69       */
70      private final transient ConcurrentNotepad busy;
71  
72      /**
73       * Batch to execute.
74       */
75      private final transient Batch batch;
76  
77      /**
78       * Public ctor.
79       * @param rqsts Requests
80       * @param ntp Notepad
81       * @param btch Batch to use
82       */
83      public OnPullRequest(
84          @NotNull(message = "requests can't be NULL") final MergeRequests rqsts,
85          @NotNull(message = "notepad can't be NULL") final ConcurrentNotepad ntp,
86          @NotNull(message = "batch can't be NULL") final Batch btch) {
87          this.requests = rqsts;
88          this.busy = ntp;
89          this.batch = btch;
90      }
91  
92      /**
93       * {@inheritDoc}
94       */
95      @Override
96      @Loggable(value = Loggable.DEBUG, limit = Integer.MAX_VALUE)
97      public void pulse() throws Exception {
98          final Iterator<MergeRequest> iterator = this.requests.iterator();
99          while (iterator.hasNext()) {
100             final MergeRequest request = iterator.next();
101             if (!this.busy.addIfAbsent(request.name())) {
102                 continue;
103             }
104             try {
105                 this.merge(request);
106             } finally {
107                 this.busy.remove(request.name());
108             }
109         }
110     }
111 
112     /**
113      * {@inheritDoc}
114      */
115     @Override
116     public String toString() {
117         return Logger.format(
118             "on pull request from %s executes %s and track it in %s",
119             this.requests,
120             this.batch,
121             this.busy
122         );
123     }
124 
125     /**
126      * Merge this pull request.
127      * @param request The request to merge
128      * @return TRUE if success
129      * @throws IOException If IO problem
130      */
131     @Tag("merge")
132     @Step(
133         before = "building merge request ${args[0].name()}",
134         // @checkstyle LineLength (1 line)
135         value = "merge request ${args[0].name()} #if($result)built successfully#{else}failed to build#end"
136     )
137     private boolean merge(final MergeRequest request) throws IOException {
138         final String tag = "on-pull-request";
139         final Snapshot snapshot = new Build(tag, this.batch).exec(
140             new ImmutableMap.Builder<String, Object>()
141                 .putAll(request.params())
142                 .build()
143         );
144         final boolean success = this.success(snapshot, tag);
145         if (success) {
146             request.accept(snapshot);
147         } else {
148             request.reject(snapshot);
149         }
150         return success;
151     }
152 
153     /**
154      * Was it a successful merge?
155      * @param snapshot Snapshot received
156      * @param tag Tag to look for
157      * @return TRUE if success
158      */
159     private boolean success(final Snapshot snapshot, final String tag) {
160         boolean success = false;
161         try {
162             success ^= snapshot.xml()
163                 .nodes(String.format("//tag[label='%s' and level='FINE']", tag))
164                 .isEmpty();
165         } catch (ImpossibleModificationException ex) {
166             Logger.warn(this, ExceptionUtils.getRootCauseMessage(ex));
167         }
168         return success;
169     }
