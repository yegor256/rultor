50  /**
51   * RESTful Unit.
52   *
53   * @author Yegor Bugayenko (yegor@tpc2.com)
54   * @version $Id$
55   * @since 1.0
56   */
57  @Immutable
58  @ToString
59  @EqualsAndHashCode(of = { "home", "token" })
60  @Loggable(Loggable.DEBUG)
61  final class RestUnit implements Unit {
62  
63      /**
64       * Home URI.
65       */
66      private final transient String home;
67  
68      /**
69       * Authentication token.
70       */
71      private final transient String token;
72  
73      /**
74       * Public ctor.
75       * @param uri URI of home page
76       * @param auth Authentication token
77       */
78      protected RestUnit(final String uri, final String auth) {
79          this.home = uri;
80          this.token = auth;
81      }
82  
83      /**
84       * {@inheritDoc}
85       */
86      @Override
87      public void update(final Spec spec) {
88          try {
89              RestTester.start(UriBuilder.fromUri(this.home))
90                  .header(HttpHeaders.ACCEPT, MediaType.APPLICATION_XML)
91                  .header(HttpHeaders.AUTHORIZATION, this.token)
92                  .get(String.format("preparing for #spec(%s)", spec))
93                  .assertStatus(HttpURLConnection.HTTP_OK)
94                  .rel("/page/links/link[@rel='save']/@href")
95                  .header(HttpHeaders.ACCEPT, MediaType.APPLICATION_XML)
96                  .post(
97                      String.format("#spec(%s)", spec.asText()),
98                      String.format(
99                          "spec=%s",
100                         URLEncoder.encode(spec.asText(), CharEncoding.UTF_8)
101                     )
102                 )
103                 .assertStatus(HttpURLConnection.HTTP_SEE_OTHER);
104         } catch (UnsupportedEncodingException ex) {
105             throw new IllegalStateException(ex);
106         }
107     }
108 
109     /**
110      * {@inheritDoc}
111      */
112     @Override
113     public Spec spec() {
114         return new Spec.Simple(
115             RestTester.start(UriBuilder.fromUri(this.home))
116                 .header(HttpHeaders.ACCEPT, MediaType.APPLICATION_XML)
117                 .header(HttpHeaders.AUTHORIZATION, this.token)
118                 .get("#spec()")
119                 .assertStatus(HttpURLConnection.HTTP_OK)
120                 .xpath("/page/unit/spec/text()")
121                 .get(0)
122         );
123     }
124 
125     /**
126      * {@inheritDoc}
127      */
128     @Override
129     public String name() {
130         return RestTester.start(UriBuilder.fromUri(this.home))
131             .header(HttpHeaders.ACCEPT, MediaType.APPLICATION_XML)
132             .header(HttpHeaders.AUTHORIZATION, this.token)
133             .get("#name()")
134             .assertStatus(HttpURLConnection.HTTP_OK)
135             .xpath("/page/unit/name/text()")
136             .get(0);
137     }
138 
139     /**
140      * {@inheritDoc}
141      */
142     @Override
143     public Wallet wallet(final Work work, final URN urn, final String unit) {
144         throw new UnsupportedOperationException();
145     }
