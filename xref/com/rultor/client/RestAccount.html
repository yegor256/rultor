43  import javax.ws.rs.core.UriBuilder;
44  import lombok.EqualsAndHashCode;
45  import lombok.ToString;

47  /**
48   * RESTful account.
49   *
50   * @author Yegor Bugayenko (yegor@tpc2.com)
51   * @version $Id$
52   * @since 1.0
53   */
54  @Immutable
55  @ToString
56  @EqualsAndHashCode(of = { "home", "token" })
57  @Loggable(Loggable.DEBUG)
58  final class RestAccount implements Account {
59  
60      /**
61       * Home URI.
62       */
63      private final transient String home;
64  
65      /**
66       * Authentication token.
67       */
68      private final transient String token;
69  
70      /**
71       * Public ctor, with custom entry point.
72       * @param entry Entry point (URI)
73       * @param tkn Token
74       */
75      protected RestAccount(
76          @NotNull(message = "URI can't be NULL") final URI entry,
77          @NotNull(message = "token can't be NULL") final String tkn) {
78          this.home = entry.toString();
79          this.token = tkn;
80      }
81  
82      /**
83       * {@inheritDoc}
84       */
85      @Override
86      public Dollars balance() {
87          return Dollars.valueOf(
88              RestTester.start(UriBuilder.fromUri(this.home))
89                  .header(HttpHeaders.ACCEPT, MediaType.APPLICATION_XML)
90                  .header(HttpHeaders.AUTHORIZATION, this.token)
91                  .get("#balance()")
92                  .assertStatus(HttpURLConnection.HTTP_OK)
93                  .xpath("/page/balance/text()")
94                  .get(0)
95          );
96      }
97  
98      /**
99       * {@inheritDoc}
100      */
101     @Override
102     public Sheet sheet() {
103         return new RestSheet(
104             URI.create(
105                 RestTester.start(UriBuilder.fromUri(this.home))
106                     .header(HttpHeaders.ACCEPT, MediaType.APPLICATION_XML)
107                     .header(HttpHeaders.AUTHORIZATION, this.token)
108                     .get("#sheet()")
109                     .assertStatus(HttpURLConnection.HTTP_OK)
110                     .xpath("/page/links/link[@rel='account']/@href")
111                     .get(0)
112             ),
113             this.token
114         );
115     }
116 
117     /**
118      * {@inheritDoc}
119      */
120     @Override
121     public void fund(final Dollars amount, final String details) {
122         throw new UnsupportedOperationException();
123     }
124 
125 }
