47  /**
48   * JVM statistics to render in HTTP.

50   * @author Yegor Bugayenko (yegor@tpc2.com)

54  @Immutable
55  @EqualsAndHashCode

57  final class Statistics {

59      /**
60       * {@inheritDoc}
61       */
62      @Override
63      public String toString() {
64          return new StringBuilder()
65              .append(this.jvm())
66              .append(CharUtils.LF)
67              .append(this.runtime())
68              .append(CharUtils.LF)
69              .append(this.system())
70              .append(CharUtils.LF).append(CharUtils.LF)
71              .append(this.memory())
72              .append(CharUtils.LF).append(CharUtils.LF)
73              .append(this.locks())
74              .append(CharUtils.LF).append(CharUtils.LF)
75              .append(this.deadlocks())
76              .append(CharUtils.LF).append(CharUtils.LF)
77              .append(this.threads())
78              .toString();
79      }

169     /**
170      * Shows all locks.
171      * @return Plain text to render
172      */
173     @NotNull
174     private String locks() {
175         final StringBuilder text = new StringBuilder();
176         final ThreadMXBean mbean = ManagementFactory.getThreadMXBean();
177         for (ThreadInfo info
178             : mbean.getThreadInfo(mbean.getAllThreadIds(), true, true)) {
179             if (text.length() > 0) {
180                 text.append(CharUtils.LF);
181             }
182             text.append(String.format("%s: ", info.getThreadName()));
183             for (LockInfo lock : info.getLockedSynchronizers()) {
184                 text.append(
185                     String.format(
186                         "%s/%x ",
187                         lock.getClassName(),
188                         lock.getIdentityHashCode()
189                     )
190                 );
191             }
192             for (MonitorInfo monitor : info.getLockedMonitors()) {
193                 text.append(
194                     String.format(
195                         "\n   %s#%s[%d]",
196                         monitor.getLockedStackFrame().getClassName(),
197                         monitor.getLockedStackFrame().getMethodName(),
198                         monitor.getLockedStackFrame().getLineNumber()
199                     )
200                 );
201             }
202         }
203         return text.toString();
204     }

206     /**
207      * Shows all deadlocks.
208      * @return Plain text to render
209      */
210     @NotNull
211     private String deadlocks() {
212         final StringBuilder text = new StringBuilder();
213         final ThreadMXBean mbean = ManagementFactory.getThreadMXBean();
214         long[] deadlocks = mbean.findDeadlockedThreads();
215         if (deadlocks == null) {
216             deadlocks = new long[] {};
217         }
218         for (long thread : deadlocks) {
219             if (text.length() > 0) {
220                 text.append(CharUtils.LF);
221             }
222             final ThreadInfo info = mbean.getThreadInfo(thread);
223             text.append(
224                 String.format(
225                     "%s %s %s",
226                     info.getThreadName(),
227                     info.getLockName(),
228                     info.getLockOwnerName(),
229                     info.getLockInfo().getClassName()
230                 )
231             );
232             for (MonitorInfo monitor : info.getLockedMonitors()) {
233                 text.append(
234                     String.format(
235                         "\n  %d %s#%s[%d]",
236                         monitor.getLockedStackDepth(),
237                         monitor.getLockedStackFrame().getClassName(),
238                         monitor.getLockedStackFrame().getMethodName(),
239                         monitor.getLockedStackFrame().getLineNumber()
240                     )
241                 );
242             }
243             for (LockInfo lock : info.getLockedSynchronizers()) {
244                 text.append(
245                     String.format(
246                         "\n  %s %x",
247                         lock.getClassName(),
248                         lock.getIdentityHashCode()
249                     )
250                 );
251             }
252         }
253         return text.toString();
254     }

256     /**
257      * All details about threads.
258      * @return Plain text to render
259      */
260     @NotNull
261     @SuppressWarnings("PMD.DoNotUseThreads")
262     private String threads() {
263         final StringBuilder text = new StringBuilder();
264         for (Map.Entry<Thread, StackTraceElement[]> entry
265             : Thread.getAllStackTraces().entrySet()) {
266             if (text.length() > 0) {
267                 text.append(CharUtils.LF);
268             }
269             final Thread thread = entry.getKey();
270             text.append(
271                 String.format(
272                     "%s %s %s alive=%B daemon=%B interrupted=%B",
273                     thread.getName(),
274                     thread.getState(),
275                     thread.getClass().getName(),
276                     thread.isAlive(),
277                     thread.isDaemon(),
278                     thread.isInterrupted()
279                 )
280             );
281             if (thread.getState().equals(Thread.State.BLOCKED)) {
282                 for (StackTraceElement line : entry.getValue()) {
283                     text.append(
284                         String.format(
285                             "\n  %s#%s[%d]",
286                             line.getClassName(),
287                             line.getMethodName(),
288                             line.getLineNumber()
289                         )
290                     );
291                 }
292             }
293         }
294         return text.toString();
295     }
