50  /**
51   * One job to do.
52   *
53   * @author Yegor Bugayenko (yegor@tpc2.com)
54   * @version $Id$
55   * @since 1.0
56   */
57  @Immutable
58  @ToString
59  @Loggable(Loggable.DEBUG)
60  @EqualsAndHashCode(of = { "work", "repo", "users" })
61  @SuppressWarnings("PMD.DoNotUseThreads")
62  final class Job {
63  
64      /**
65       * Work to do.
66       */
67      private final transient Coordinates work;
68  
69      /**
70       * Repo.
71       */
72      private final transient Repo repo;
73  
74      /**
75       * Users.
76       */
77      private final transient Users users;
78  
79      /**
80       * Public ctor.
81       * @param wrk Work to do
82       * @param rep Repo
83       * @param usrs Users
84       */
85      protected Job(final Coordinates wrk, final Repo rep, final Users usrs) {
86          this.work = wrk;
87          this.repo = rep;
88          this.users = usrs;
89      }
90  
91      /**
92       * Process given work.
93       * @param decor Decorator to use
94       */
95      @Loggable(value = Loggable.DEBUG, limit = Integer.MAX_VALUE)
96      @SuppressWarnings("PMD.AvoidCatchingGenericException")
97      public void process(final Job.Decor decor) {
98          final User owner = this.users.get(this.work.owner());
99          final Rule rule = owner.rules().get(this.work.rule());
100         if (rule.failure().isEmpty()) {
101             try {
102                 this.make(owner, rule, decor).pulse();
103             } catch (SpecException ex) {
104                 rule.failure(Exceptions.stacktrace(ex));
105             // @checkstyle IllegalCatch (1 line)
106             } catch (Exception ex) {
107                 throw new IllegalArgumentException(ex);
108             }
109         }
110     }
111 
112     /**
113      * Make an instance to run.
114      * @param owner Owner of it
115      * @param rule The rule to use
116      * @param decor Decorator to use
117      * @return Instance
118      * @throws SpecException If incorrect spec
119      * @checkstyle RedundantThrows (4 lines)
120      */
121     private Instance make(final User owner, final Rule rule,
122         final Job.Decor decor) throws SpecException {
123         final Variable<?> var = this.var(owner, rule.spec());
124         Instance instance = Instance.EMPTY;
125         if (var.arguments().isEmpty()) {
126             final Arguments args = new Arguments(
127                 this.work,
128                 new XemblyReportingWallet(new OwnWallet(this.work, rule))
129             );
130             final Object inst = var.instantiate(this.users, args);
131             final Object drain = this.var(owner, rule.drain())
132                 .instantiate(this.users, args);
133             if (inst instanceof Instance && drain instanceof Drain) {
134                 instance = new ThreadGroupSpy(
135                     this.work,
136                     new WithSpec(
137                         rule.spec(),
138                         decor.decorate(Instance.class.cast(inst))
139                     ),
140                     Drain.class.cast(drain)
141                 );
142             }
143         }
144         return instance;
145     }
146 
147     /**
148      * Make a variable.
149      * @param owner Owner of the spec
150      * @param spec Spec itself
151      * @return Variable
152      * @throws SpecException If fails
153      * @checkstyle RedundantThrows (5 lines)
154      */
155     private Variable<?> var(final User owner, final Spec spec)
156         throws SpecException {
157         return this.repo.make(owner, spec);
158     }
159 
160     /**
161      * Decorator of an instance.
162      */
163     public interface Decor {
164         /**
165          * Decorate this given instance.
166          * @param instance Instance to decorate
167          * @return Decorated one
168          */
169         Instance decorate(Instance instance);
170     }
