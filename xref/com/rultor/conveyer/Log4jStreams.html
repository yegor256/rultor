53  /**
54   * Streams from Log4j.
55   *
56   * @author Yegor Bugayenko (yegor@tpc2.com)
57   * @version $Id$
58   * @since 1.0
59   * @checkstyle ClassDataAbstractionCoupling (500 lines)
60   */
61  @EqualsAndHashCode(callSuper = false, of = { "groups", "buffers" })
62  final class Log4jStreams extends AppenderSkeleton implements Streams {
63  
64      /**
65       * Thread groups to keys.
66       */
67      private final transient ConcurrentMap<ThreadGroup, String> groups =
68          new ConcurrentHashMap<ThreadGroup, String>(0);
69  
70      /**
71       * Keys to buffers.
72       */
73      private final transient ConcurrentMap<String, CircularBuffer> buffers =
74          new ConcurrentHashMap<String, CircularBuffer>(0);
75  
76      /**
77       * Public ctor.
78       */
79      protected Log4jStreams() {
80          super();
81          Logger.getRootLogger().addAppender(this);
82          this.setLayout(new PatternLayout("%p %m%n"));
83      }
84  
85      /**
86       * {@inheritDoc}
87       */
88      @Override
89      public void close() {
90          Logger.getRootLogger().removeAppender(this);
91      }
92  
93      /**
94       * {@inheritDoc}
95       */
96      @Override
97      public String register() {
98          final String key = DigestUtils.md5Hex(
99              RandomStringUtils.random(Tv.FIFTY)
100         );
101         final ThreadGroup group = Thread.currentThread().getThreadGroup();
102         if (this.groups.put(group, key) != null) {
103             throw new IllegalStateException("call unregister() first");
104         }
105         this.buffers.put(key, new CircularBuffer(Tv.FIFTY * Tv.THOUSAND));
106         return key;
107     }
108 
109     /**
110      * {@inheritDoc}
111      */
112     @Override
113     public void unregister(final String key) {
114         final ThreadGroup group = Thread.currentThread().getThreadGroup();
115         if (!this.groups.remove(group, key)) {
116             throw new IllegalStateException("call register() first");
117         }
118         this.buffers.remove(key);
119     }
120 
121     /**
122      * {@inheritDoc}
123      */
124     @Override
125     public InputStream stream(final String key) throws IOException {
126         final ThreadGroup group = ImmutableBiMap.copyOf(this.groups)
127             .inverse().get(key);
128         final InputStream stream;
129         if (group == null) {
130             stream = IOUtils.toInputStream(
131                 String.format("key '%s' is absent", key),
132                 CharEncoding.UTF_8
133             );
134         } else {
135             final CircularBuffer buffer = this.buffers.get(key);
136             if (buffer == null) {
137                 throw new IllegalStateException(
138                     String.format("buffer is absent for key '%s'", key)
139                 );
140             }
141             stream = new SequenceInputStream(
142                 IOUtils.toInputStream(
143                     String.format(
144                         "Listening to %s, key=%s, threads=%d...\n\n",
145                         group, key, group.activeCount()
146                     ),
147                     CharEncoding.UTF_8
148                 ),
149                 new InputStream() {
150                     @Override
151                     public int read() throws IOException {
152                         return Log4jStreams.this.read(buffer);
153                     }
154                 }
155             );
156         }
157         return stream;
158     }
159 
160     /**
161      * {@inheritDoc}
162      */
163     @Override
164     public String interrupt(final String key) {
165         final ThreadGroup group = ImmutableBiMap.copyOf(this.groups)
166             .inverse().get(key);
167         final String response;
168         if (group == null) {
169             response = String.format("key '%s' not found", key);
170         } else {
171             group.interrupt();
172             response = String.format("%s interrupted", group);
173         }
174         return response;
175     }
176 
177     /**
178      * {@inheritDoc}
179      */
180     @Override
181     protected void append(final LoggingEvent event) {
182         final String key = this.groups.get(
183             Thread.currentThread().getThreadGroup()
184         );
185         if (key != null) {
186             final CircularBuffer buffer = this.buffers.get(key);
187             if (buffer != null) {
188                 final byte[] bytes;
189                 try {
190                     bytes = this.getLayout().format(event)
191                         .getBytes(CharEncoding.UTF_8);
192                 } catch (UnsupportedEncodingException ex) {
193                     throw new IllegalStateException(ex);
194                 }
195                 for (byte data : bytes) {
196                     buffer.write(data);
197                 }
198             }
199         }
200     }
201 
202     /**
203      * {@inheritDoc}
204      */
205     @Override
206     public String toString() {
207         final StringBuilder text = new StringBuilder()
208             .append(this.getClass().getCanonicalName())
209             .append('\n')
210             .append(this.groups.size())
211             .append(" running thread group(s):\n");
212         for (Map.Entry<ThreadGroup, String> entry : this.groups.entrySet()) {
213             text.append(entry.getKey())
214                 .append(": ")
215                 .append(entry.getValue().substring(0, Tv.FIVE))
216                 .append("...\n");
217         }
218         return text.toString();
219     }

221     /**
222      * {@inheritDoc}
223      */
224     @Override
225     public boolean requiresLayout() {
226         return true;
227     }

229     /**
230      * Read next byte from the buffer.
231      * @param buffer The buffer to read from
232      * @return The byte
233      */
234     private byte read(final CircularBuffer buffer) {
235         while (buffer.isEmpty()) {
236             try {
237                 TimeUnit.SECONDS.sleep(1);
238             } catch (InterruptedException ex) {
239                 Thread.currentThread().interrupt();
240                 throw new IllegalStateException(ex);
241             }
242         }
243         return buffer.read();
244     }
