30  package com.rultor.spi;
31  
32  import com.jcabi.aspects.Immutable;
33  
34  /**
35   * Tick.
36   *
37   * @author Yegor Bugayenko (yegor256@gmail.com)
38   * @version $Id: 6d8564d51191cb37d6492fcf11892cb0582c0637 $
39   * @since 1.52
40   */
41  @Immutable
42  public final class Tick {
43  
44      /**
45       * When was it started.
46       */
47      private final transient long when;
48  
49      /**
50       * Duration.
51       */
52      private final transient long msec;
53  
54      /**
55       * Talks processed or -1.
56       */
57      private final transient int talks;
58  
59      /**
60       * Ctor.
61       * @param date When
62       * @param duration Duration in msec
63       * @param total Total processed or negative if failed
64       */
65      public Tick(final long date, final long duration,
66          final int total) {
67          this.when = date;
68          this.msec = duration;
69          this.talks = total;
70      }
71  
72      /**
73       * Time of start.
74       * @return Time of start
75       */
76      public long start() {
77          return this.when;
78      }
79  
80      /**
81       * Duration in msec.
82       * @return Duration
83       */
84      public long duration() {
85          return this.msec;
86      }
87  
88      /**
89       * Total processed or negative.
90       * @return Number of talks
91       */
92      public int total() {
93          return this.talks;
94      }
95  
96  }
