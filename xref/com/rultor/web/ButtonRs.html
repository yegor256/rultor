72       * Instance that retrieves XML from application.
73       */
74      private static final Build DEFAULT_BUILD = new Build() {
75          @Override
76          public String info(final URI uri) {
77              try {
78                  return new JdkRequest(uri)
79                      .header(
80                          HttpHeaders.ACCEPT,
81                          MediaType.APPLICATION_XML_UTF_8.toString()
82                  )
83                      .fetch()
84                      .as(RestResponse.class)
85                      .body();
86              } catch (IOException ex) {
87                  throw new IllegalStateException(ex);
88              }
89          }
90      };

131     /**
132      * Draw PNG image with build stats.
133      * @param rule Rule to use.
134      * @return Image generated.
135      * @throws Exception In case of problems generating image.
136      */
137     @GET
138     @Path("{rule:[\\w\\-]+}.png")
139     @Produces("image/png")
140     public Response pngButton(@PathParam("rule") final String rule)
141         throws Exception {
142         final PNGTranscoder transcoder = new PNGTranscoder();
143         transcoder.addTranscodingHint(
144             PNGTranscoder.KEY_WIDTH, (float) Tv.HUNDRED
145         );
146         transcoder.addTranscodingHint(
147             PNGTranscoder.KEY_HEIGHT, (float) Tv.FIFTY
148         );
149         final ByteArrayOutputStream png = new ByteArrayOutputStream();
150         transcoder.transcode(
151             new TranscoderInput(IOUtils.toInputStream(this.svg(rule))),
152             new TranscoderOutput(png)
153         );
154         return Response
155             .ok(png.toByteArray(), MediaType.PNG.toString())
156             .build();
157     }

159     /**

165     @GET

167     @Produces("image/svg+xml")

169         throws Exception {

172             .build();
173     }

175     /**
176      * Create SVG from build.
177      * @param rule Rule to use.
178      * @return String with SVG.
179      * @throws Exception In case of transformation error.
180      */
181     private String svg(final String rule) throws Exception {
182         return new XSLT(
183             new StreamSource(
184                 IOUtils.toInputStream(
185                     StringUtils.defaultIfBlank(
186                         this.build.info(
187                             UriBuilder.fromUri(this.uriInfo().getBaseUri())
188                                 .segment("s", this.stand).build()
189                         ),
190                         "<page></page>"
191                     )
192                 )
193             ),
194             new StreamSource(
195                 IOUtils.toInputStream(
196                     String.format(
197                         IOUtils.toString(
198                             this.getClass().getResourceAsStream(
199                                 "button.xsl"
200                             )
201                         ),
202                         rule
203                     )
204                 )
205             )
206         ).xml();
207     }

209     /**
210      * Register a new font.
211      * @param env Environment to register in.
212      * @param file Font file to register.
213      */
214     private void font(final GraphicsEnvironment env, final String file) {
215         try {
216             env.registerFont(
217                 Font.createFont(
218                     Font.TRUETYPE_FONT,
219                     UriBuilder.fromPath(file)
220                         .build().toURL().openStream()
221                 )
222             );
223         } catch (FontFormatException ex) {
224             throw new IllegalStateException(ex);
225         } catch (IOException ex) {
226             throw new IllegalStateException(ex);
227         }
228     }
