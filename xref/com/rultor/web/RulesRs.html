50  /**
51   * List of rules.
52   *
53   * @author Yegor Bugayenko (yegor@tpc2.com)
54   * @version $Id$
55   * @since 1.0
56   * @checkstyle MultipleStringLiterals (500 lines)
57   */
58  @Path("/rules")
59  @Loggable(Loggable.DEBUG)
60  public final class RulesRs extends BaseRs {
61  
62      /**
63       * Get entrance page JAX-RS response.
64       * @return The JAX-RS response
65       */
66      @GET
67      @Path("/")
68      public Response index() {
69          return new PageBuilder()
70              .stylesheet("/xsl/rules.xsl")
71              .build(EmptyPage.class)
72              .init(this)
73              .append(new Breadcrumbs().with("self", "rules").bundle())
74              .append(this.mine())
75              .link(new Link("create", "./create"))
76              .render()
77              .build();
78      }
79  
80      /**
81       * Create new empty rule.
82       * @param name Name of the rule to create
83       * @return The JAX-RS response
84       */
85      @POST
86      @Path("/create")
87      public Response create(@NotNull(message = "rule name is mandatory")
88          @FormParam("name") final String name) {
89          if (this.user().rules().contains(name)) {
90              throw this.flash().redirect(
91                  this.uriInfo().getRequestUri(),
92                  String.format("Rule `%s` already exists", name),
93                  Level.WARNING
94              );
95          }
96          this.user().rules().create(name);
97          throw this.flash().redirect(
98              this.uriInfo().getBaseUriBuilder()
99                  .clone()
100                 .path(RuleRs.class)
101                 .build(name),
102             String.format("Rule `%s` successfully created", name),
103             Level.INFO
104         );
105     }
106 
107     /**
108      * Re-save all rules in one go (internal feature, not for public users).
109      * @return Plain text output
110      */
111     @GET
112     @Path("/resave")
113     @Produces(MediaType.TEXT_PLAIN)
114     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
115     public String resave() {
116         final StringBuilder text = new StringBuilder();
117         for (final Rule rule : this.user().rules()) {
118             final String name = rule.name();
119             text.append(name).append(": ");
120             try {
121                 rule.update(
122                     new Spec.Strict(
123                         rule.spec().asText(), this.repo(), this.user(),
124                         this.users(), this.coordinates(name), Object.class
125                     ),
126                     new Spec.Strict(
127                         rule.drain().asText(), this.repo(), this.user(),
128                         this.users(), this.coordinates(name), Object.class
129                     )
130                 );
131                 text.append("OK");
132             } catch (final SpecException ex) {
133                 text.append('\n').append(Exceptions.stacktrace(ex));
134             }
135             text.append("\n\n");
136         }
137         return text.toString();
138     }
139 
140     /**
141      * All my rules.
142      * @return Collection of JAXB rules
143      */
144     private JaxbBundle mine() {
145         return new JaxbBundle("rules").add(
146             new JaxbBundle.Group<Rule>(this.user().rules()) {
147                 @Override
148                 public JaxbBundle bundle(final Rule rule) {
149                     return RulesRs.this.rule(rule);
150                 }
151             }
152         );
153     }
154 
155     /**
156      * Convert rule to JaxbBundle.
157      * @param rule The rule
158      * @return Bundle
159      */
160     private JaxbBundle rule(final Rule rule) {
161         return new JaxbBundle("rule")
162             .add("name", rule.name())
163             .up()
164             .add("failure", rule.failure())
165             .up()
166             .link(
167                 new Link(
168                     // @checkstyle MultipleStringLiterals (1 line)
169                     "remove",
170                     this.uriInfo().getBaseUriBuilder()
171                         .clone()
172                         .path(RuleRs.class)
173                         .path(RuleRs.class, "remove")
174                         .build(rule.name())
175                 )
176             )
177             .link(
178                 new Link(
179                     "edit",
180                     this.uriInfo().getBaseUriBuilder()
181                         .clone()
182                         .path(RuleRs.class)
183                         .build(rule.name())
184                 )
185             )
186             .link(
187                 new Link(
188                     "drain",
189                     this.uriInfo().getBaseUriBuilder()
190                         .clone()
191                         .path(DrainRs.class)
192                         .build(rule.name())
193                 )
194             );
195     }
