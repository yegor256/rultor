66      @GET

69          return new PageBuilder()
70              .stylesheet("/xsl/receipts.xsl")
71              .build(EmptyPage.class)
72              .init(this)
73              .append(
74                  this.receipts(
75                      Iterables.limit(this.user().receipts(), Tv.TEN)

78              .link(

112     /**
113      * Load them all in plain text.
114      * @return The JAX-RS response
115      */
116     @GET
117     @Path("/text")
118     @Produces(MediaType.TEXT_PLAIN)
119     public String text() {
120         final StringBuilder text = new StringBuilder();
121         for (Receipt receipt : this.user().receipts()) {
122             text.append(receipt.date())
123                 .append('\t')
124                 .append(receipt.dollars().points())
125                 .append('\t')
126                 .append(receipt.payer())
127                 .append('\t')
128                 .append(receipt.beneficiary())
129                 .append('\t')
130                 .append(receipt.unit())
131                 .append('\t')
132                 .append(receipt.details())
133                 .append(CharUtils.LF);
134         }
135         return text.toString();
136     }

138     /**
139      * Load them all in JSON.
140      * @return The JAX-RS response
141      */
142     @GET
143     @Path("/json")
144     @Produces(MediaType.APPLICATION_JSON)
145     public String json() {
146         final JsonArrayBuilder json = Json.createArrayBuilder();
147         for (Receipt receipt : this.user().receipts()) {
148             json.add(
149                 Json.createObjectBuilder()
150                     .add("date", receipt.date().toString())
151                     .add("points", receipt.dollars().points())
152                     .add("payer", receipt.payer().toString())
153                     .add("beneficiary", receipt.beneficiary().toString())
154                     .add("unit", receipt.unit())
155                     .add("details", receipt.details())
156             );
157         }
158         final StringWriter writer = new StringWriter();
159         Json.createWriter(writer).writeArray(json.build());
160         return writer.toString();
161     }

174     /**
175      * All receipts of the user.
176      * @param receipts All receipts to show
177      * @return Collection of JAXB units
178      */
179     private JaxbBundle receipts(final Iterable<Receipt> receipts) {
180         return new JaxbBundle("receipts").add(
181             new JaxbBundle.Group<Receipt>(receipts) {
182                 @Override
183                 public JaxbBundle bundle(final Receipt rcpt) {
184                     return ReceiptsRs.this.receipt(rcpt);
185                 }
186             }
187         );
188     }
189 
190     /**
191      * Convert receipt to JaxbBundle.
192      * @param receipt Receipt to render
193      * @return Bundle
194      */
195     private JaxbBundle receipt(final Receipt receipt) {
196         return new JaxbBundle("receipt")
197             .add("amount", receipt.dollars().toString())
198             .up()
199             .add("details", receipt.details())
200             .up()
201             .add("payer", receipt.payer().toString())
202             .up()
203             .add("beneficiary", receipt.beneficiary().toString())
204             .up()
205             .add("date", receipt.date().toString())
206             .up()
207             .add("unit", receipt.unit())
208             .up();
209     }
