175             .build(EmptyPage.class)

181                         .clone()
182                         .path(AclRs.class)

187                 new Link(
188                     "pulse-open",
189                     this.uriInfo().getBaseUriBuilder()
190                         .clone()
191                         .path(StandRs.class)
192                         .queryParam(StandRs.QUERY_OPEN, "")
193                         .build(this.name)
194                 )
195             )
196             .link(
197                 new Link(
198                     "collapse",
199                     this.self(new ArrayList<String>(0))
200                 )
201             );
202         if (this.auth().identity().equals(Identity.ANONYMOUS)) {
203             page = page.append(
204                 new Breadcrumbs()
205                     .with("home")
206                     .with("self", this.name)
207                     .bundle()
208             );
209         } else {
210             page = page.append(
211                 new Breadcrumbs()
212                     .with("stands")
213                     .with("edit", this.name)
214                     .with("collapse", "stand")
215                     .bundle()
216             );
217         }
218         Pulses pulses = this.stand().pulses();
219         for (String tag : this.tags) {
220             pulses = pulses.query().withTag(tag).fetch();
221         }
222         return page
223             .append(new JaxbBundle("stand", this.name))
224             .append(
225                 new JaxbBundle("filters").add(
226                     new JaxbBundle.Group<String>(this.tags) {
227                         @Override
228                         public JaxbBundle bundle(final String tag) {
229                             return new JaxbBundle("filter", tag);
230                         }
231                     }
232                 )
233             )
234             .append(this.widgets(this.widgets(this.stand().widgets())))
235             .append(this.pulses(pulses.iterator(), Tv.TWENTY))
236             .render()
237             .build();
238     }

240     /**
241      * Get snapshot HTML for a pulse.
242      * @param uid Unique identifier of a pulse
243      * @return The JAX-RS response
244      */
245     @GET
246     @Path("/fetch")
247     @Produces(MediaType.TEXT_HTML)
248     public Response fetch(@QueryParam(StandRs.QUERY_ID) final String uid) {
249         final Coordinates coords = Coordinates.Simple.valueOf(uid);
250         Response resp;
251         try {
252             resp = Response.ok().entity(
253                 new XSLT(
254                     this.render(
255                         new JaxbBundle("div"),
256                         this.stand().pulses().tail(coords).iterator().next()
257                     ).element(),
258                     this.getClass().getResourceAsStream("fetch.xsl")
259                 ).xml()
260             ).build();
261         } catch (TransformerException ex) {
262             resp = Response.serverError().entity(
263                 Exceptions.stacktrace(ex)
264             ).build();
265         } catch (IOException ex) {
266             resp = Response.serverError().entity(
267                 Exceptions.stacktrace(ex)
268             ).build();
269         }
270         return resp;
271     }

273     /**
274      * Get stand.
275      * @return The stand
276      */
277     private Stand stand() {
278         final Stand stand;
279         try {
280             stand = this.users().stand(this.name);
281         } catch (NoSuchElementException ex) {
282             throw this.flash().redirect(this.uriInfo().getBaseUri(), ex);
283         }
284         if (!stand.owner().equals(this.auth().identity().urn())
285             && !this.acl(stand).canView(this.auth().identity().urn())) {
286             throw this.flash().redirect(
287                 this.uriInfo().getBaseUri(),
288                 String.format("access denied to stand `%s`", this.name),
289                 Level.WARNING
290             );
291         }
292         return stand;
293     }

295     /**
296      * All pulses of the stand.
297      * @param pulses All pulses to show
298      * @param maximum Maximum to show
299      * @return Collection of JAXB stands
300      */
301     private JaxbBundle pulses(final Iterator<Pulse> pulses, final int maximum) {
302         JaxbBundle bundle = new JaxbBundle("pulses");
303         int pos;
304         for (pos = 0; pos < maximum; ++pos) {
305             if (!pulses.hasNext()) {
306                 break;
307             }
308             bundle = bundle.add(this.pulse(pulses.next()));
309         }
310         return bundle;
311     }

313     /**
314      * Convert pulse to JaxbBundle.
315      * @param pulse The pulse
316      * @return Bundle
317      */
318     private JaxbBundle pulse(final Pulse pulse) {
319         final Coordinates coords = pulse.coordinates();
320         JaxbBundle bundle = new JaxbBundle("pulse")
321             .add("coordinates")
322             .add("rule", coords.rule()).up()
323             .add("owner", coords.owner().toString()).up()
324             .add("scheduled", coords.scheduled().toString()).up()
325             .up();
326         final String label = new Coordinates.Simple(coords).toString();
327         final ArraySet<String> now = new ArraySet<String>(this.open);
328         if (this.open.contains(label)) {
329             bundle = this
330                 .render(bundle, pulse)
331                 .link(new Link("close", this.self(now.without(label))))
332                 .link(
333                     new Link(
334                         "fetch",
335                         this.uriInfo().getBaseUriBuilder()
336                             .clone()
337                             .path(StandRs.class)
338                             .path(StandRs.class, "fetch")
339                             .queryParam(StandRs.QUERY_ID, "{id}")
340                             .build(this.name, label)
341                     )
342                 );
343         } else {
344             bundle = bundle
345                 .link(new Link("open", this.self(now.with(label))))
346                 .add("tags")
347                 .add(
348                     new JaxbBundle.Group<Tag>(pulse.tags()) {
349                         @Override
350                         public JaxbBundle bundle(final Tag tag) {
351                             return StandRs.this.bundle(tag);
352                         }
353                     }
354                 )
355                 .up();
356         }
357         return bundle;
358     }

360     /**
361      * Build URI to itself with new list of open pulses.
362      * @param coords Names of pulses
363      * @return URI
364      */
365     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
366     private URI self(final Collection<String> coords) {
367         final Object[] args = new Object[coords.size() + 1];
368         final Object[] labels = new String[coords.size()];
369         args[0] = this.name;
370         int idx = 0;
371         for (String coord : coords) {
372             labels[idx] = String.format("{arg%d}", idx);
373             args[idx + 1] = coord;
374             ++idx;
375         }
376         return this.uriInfo().getBaseUriBuilder()
377             .clone()
378             .path(StandRs.class)
379             .queryParam(StandRs.QUERY_OPEN, labels)
380             .build(args);
381     }

383     /**
384      * Render snapshot into bundle.
385      * @param bundle Bundle to render into
386      * @param pulse The pulse
387      * @return Bundle
388      */
389     private JaxbBundle render(final JaxbBundle bundle, final Pulse pulse) {
390         JaxbBundle output = bundle;
391         try {
392             final Snapshot snapshot = this.snapshot(pulse.xembly());
393             try {
394                 output = output.add(
395                     new XSLT(
396                         snapshot,
397                         this.getClass().getResourceAsStream("post.xsl")
398                     ).dom().getDocumentElement()
399                 );
400             } catch (XemblyException ex) {
401                 output = this.bug(output, ex);
402             } catch (TransformerException ex) {
403                 output = this.bug(output, ex);
404             }
405         } catch (IOException ex) {
406             output = this.bug(output, ex);
407         } catch (SyntaxException ex) {
408             output = this.bug(output, ex);
409         }
410         return output;
411     }

463             };
464         }

504     /**
505      * Render one widget.
506      * @param widget Widget to render
507      * @return DOM element
508      */
509     private Element widget(final Widget widget) {
510         final Document dom;
511         try {
512             dom = DocumentBuilderFactory.newInstance()
513                 .newDocumentBuilder().newDocument();
514         } catch (ParserConfigurationException ex) {
515             throw new IllegalStateException(ex);
516         }
517         final Element root = dom.createElement("widget");
518         root.setAttribute("class", widget.getClass().getCanonicalName());
519         dom.appendChild(root);
520         try {
521             new Xembler(
522                 new Directives()
523                     .xpath("/widget")
524                     .append(widget.render(this.stand()))
525             ).apply(dom);
526         } catch (ImpossibleModificationException ex) {
527             final Element error = dom.createElement("error");
528             error.setTextContent(Exceptions.stacktrace(ex));
529             dom.getDocumentElement().appendChild(error);
530         }
531         return dom.getDocumentElement();
532     }

534     /**
535      * Bundle a tag.
536      * @param tag A tag
537      * @return Bundle
538      */
539     private JaxbBundle bundle(final Tag tag) {
540         final Set<String> labels = new TreeSet<String>();
541         labels.addAll(this.tags);
542         labels.add(tag.label());
543         final Object[] args = new String[labels.size()];
544         final Object[] values = new String[labels.size() + 1];
545         int idx = 0;
546         values[0] = this.name;
547         for (String label : labels) {
548             args[idx] = String.format("{arg%d}", idx);
549             values[idx + 1] = label;
550             ++idx;
551         }
552         return new JaxbBundle("tag")
553             .add("label", tag.label())
554             .up()
555             .add("level", tag.level().toString())
556             .up()
557             .add("markdown", tag.markdown())
558             .up()
559             .link(
560                 new Link(
561                     "filter",
562                     this.uriInfo().getBaseUriBuilder()
563                         .clone()
564                         .path(StandRs.class)
565                         .queryParam(StandRs.QUERY_TAGS, args)
566                         .build(values)
567                 )
568             );
569     }
