48  /**
49   * Index resource, front page of the website.
50   *
51   * @author Yegor Bugayenko (yegor@teamed.io)
52   * @version $Id: 7c20afb8c697686ec738b0be9d80bdc995af66b7 $
53   * @since 1.50
54   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
55   */
56  final class TkHome implements Take {
57  
58      /**
59       * Talks.
60       */
61      private final transient Talks talks;
62  
63      /**
64       * Toggles.
65       */
66      private final transient Toggles toggles;
67  
68      /**
69       * Ctor.
70       * @param tlks Talks
71       * @param tgls Toggles
72       */
73      TkHome(final Talks tlks, final Toggles tgls) {
74          this.talks = tlks;
75          this.toggles = tgls;
76      }
77  
78      @Override
79      public Response act(final Request req) throws IOException {
80          return new RsPage(
81              "/xsl/home.xsl",
82              req,
83              new XeSource() {
84                  @Override
85                  public Iterable<Directive> toXembly() throws IOException {
86                      return TkHome.this.recent();
87                  }
88              },
89              new XeLink("status", "/status"),
90              new XeLink("ticks", "/ticks", "image/png"),
91              new XeSource() {
92                  @Override
93                  public Iterable<Directive> toXembly() throws IOException {
94                      return TkHome.this.flags(req);
95                  }
96              }
97          );
98      }
99  
100     /**
101      * Turn recent talks into directives.
102      * @return Directives
103      * @throws IOException If fails
104      */
105     private Directives recent() throws IOException {
106         final Directives dirs = new Directives().add("recent");
107         final PrettyTime pretty = new PrettyTime();
108         for (final Talk talk : Iterables.limit(this.talks.recent(), Tv.FIVE)) {
109             dirs.add("talk").set(talk.name())
110                 .attr("timeago", pretty.format(talk.updated()));
111             final XML xml = talk.read();
112             if (!xml.nodes("/talk/wire/href").isEmpty()) {
113                 dirs.attr(
114                     "href",
115                     talk.read().xpath("/talk/wire/href/text()").get(0)
116                 );
117             }
118             dirs.up();
119         }
120         return dirs;
121     }
122 
123     /**
124      * Flags/toggles to show.
125      * @param req Request
126      * @return Directives
127      * @throws IOException If fails
128      */
129     private Iterable<Directive> flags(final Request req) throws IOException {
130         final Directives dirs = new Directives().add("toggles");
131         dirs.add("read-only")
132             .set(Boolean.toString(this.toggles.readOnly())).up();
133         if (!new RqUser(req).anonymous()) {
134             dirs.append(
135                 new XeLink("sw:read-only", "/toggles/read-only").toXembly()
136             );
137         }
138         return dirs;
139     }
