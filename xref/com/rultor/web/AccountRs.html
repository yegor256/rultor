195     /**
196      * Get entrance page JAX-RS response.
197      * @return The JAX-RS response
198      * @throws IOException If fails on tailing
199      */
200     @GET
201     @Path("/")
202     public Response index() throws IOException {
203         final Sheet sheet = this.sheet();
204         return new PageBuilder()
205             .stylesheet("/xsl/account.xsl")
206             .build(EmptyPage.class)
207             .init(this)
208             .append(new JaxbBundle("sql", sheet.toString()))
209             .append(new JaxbBundle("since", Integer.toString(this.since)))
210             .append(
211                 new JaxbBundle("columns").add(
212                     new JaxbBundle.Group<Column>(sheet.columns()) {
213                         @Override
214                         public JaxbBundle bundle(final Column column) {
215                             return AccountRs.this.column(column);
216                         }
217                     }
218                 )
219             )
220             .link(
221                 new Link(
222                     "latest",
223                     this.home().replaceQueryParam(AccountRs.QUERY_SINCE, 0)
224                 )
225             )
226             .append(this.receipts(sheet.tail(this.since).iterator(), Tv.TWENTY))
227             .render()
228             .build();
229     }

231     /**
232      * Show receipts.
233      * @param receipts Receipts
234      * @param maximum Maximum to show
235      * @return Collection of JAXB units
236      */
237     private JaxbBundle receipts(final Iterator<List<Object>> receipts,
238         final int maximum) {
239         JaxbBundle bundle = new JaxbBundle("receipts");
240         int pos;
241         for (pos = 0; pos < maximum; ++pos) {
242             if (!receipts.hasNext()) {
243                 break;
244             }
245             bundle = bundle.add(this.receipt(receipts.next()));
246         }
247         if (pos == maximum && receipts.hasNext()) {
248             bundle = bundle.link(
249                 new Link(
250                     "more",
251                     this.home().replaceQueryParam(
252                         AccountRs.QUERY_SINCE, this.since + pos
253                     )
254                 )
255             );
256         }
257         return bundle;
258     }

260     /**
261      * Receipt to bundle.
262      * @param receipt Receipt
263      * @return Bundle
264      */
265     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
266     private JaxbBundle receipt(final List<Object> receipt) {
267         JaxbBundle bundle = new JaxbBundle("receipt");
268         for (int idx = 0; idx < receipt.size(); ++idx) {
269             String cell = receipt.get(idx).toString();
270             if (idx == receipt.size() - 1) {
271                 cell = new Dollars(Long.parseLong(cell)).toString();
272             }
273             bundle = bundle.add("cell", cell).up();
274         }
275         return bundle;
276     }

278     /**
279      * Column to bundle.
280      * @param column Column
281      * @return Bundle
282      */
283     private JaxbBundle column(final Column column) {
284         JaxbBundle bundle = new JaxbBundle("column")
285             .add("title", column.title())
286             .up();
287         if (this.asc.contains(column.title())) {
288             bundle = bundle.attr("sorted", "asc");
289         }
290         if (this.desc.contains(column.title())) {
291             bundle = bundle.attr("sorted", "desc");
292         }
293         if (this.groups.contains(column.title())) {
294             bundle = bundle.attr("grouped", "yes");
295         }
296         if (column.isSum()) {
297             bundle = bundle.attr("sum", "yes");
298         }
299         if (column.isGroup()) {
300             bundle.link(
301                 new Link(
302                     "group",
303                     this.home().queryParam(
304                         AccountRs.QUERY_GROUP, column.title()
305                     )
306                 )
307             );
308         }
309         bundle.link(
310             new Link(
311                 "asc",
312                 this.home().queryParam(AccountRs.QUERY_ASC, column.title())
313             )
314         );
315         bundle.link(
316             new Link(
317                 "desc",
318                 this.home().queryParam(AccountRs.QUERY_DESC, column.title())
319             )
320         );
321         return bundle;
322     }

324     /**
325      * Create URI builder with currently set params.
326      * @return URI builder
327      */
328     private UriBuilder home() {
329         final UriBuilder builder = this.uriInfo()
330             .getBaseUriBuilder()
331             .clone()
332             .path(AccountRs.class)
333             .queryParam(AccountRs.QUERY_SINCE, this.since)
334             .queryParam(AccountRs.QUERY_START, this.start)
335             .queryParam(AccountRs.QUERY_FINISH, this.finish);
336         for (String grp : this.groups) {
337             builder.queryParam(AccountRs.QUERY_GROUP, grp);
338         }
339         for (String col : this.asc) {
340             builder.queryParam(AccountRs.QUERY_ASC, col);
341         }
342         for (String col : this.desc) {
343             builder.queryParam(AccountRs.QUERY_DESC, col);
344         }
345         return builder;
346     }

348     /**
349      * Fetch sheet.
350      * @return Sheet
351      */
352     private Sheet sheet() {
353         Sheet sheet = this.user().account().sheet()
354             .between(this.start, this.finish);
355         for (String grp : this.groups) {
356             sheet = sheet.groupBy(grp);
357         }
358         for (String col : this.asc) {
359             sheet = sheet.orderBy(col, true);
360         }
361         for (String col : this.desc) {
362             sheet = sheet.orderBy(col, false);
363         }
364         return sheet;
365     }
