57  /**
58   * Siblings.
59   *
60   * @author Yegor Bugayenko (yegor256@gmail.com)
61   * @version $Id: f4d3d3ded8c299d20ce3d20f1f21279b32255109 $
62   * @since 1.50
63   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
64   */
65  final class TkSiblings implements TkRegex {
66  
67      /**
68       * Talks.
69       */
70      private final transient Talks talks;
71  
72      /**
73       * Ctor.
74       * @param tks Talks
75       */
76      TkSiblings(final Talks tks) {
77          this.talks = tks;
78      }
79  
80      @Override
81      public Response act(final RqRegex req) throws IOException {
82          final Date since = new Date(
83              Long.parseLong(
84                  new RqHref.Smart(new RqHref.Base(req)).single(
85                      "s", Long.toString(Long.MAX_VALUE)
86                  )
87              )
88          );
89          final String repo = req.matcher().group(1);
90          final List<Talk> siblings = Lists.newArrayList(
91              Iterables.limit(
92                  this.talks.siblings(repo, since), Tv.TWENTY
93              )
94          );
95          if (!siblings.isEmpty()
96              && !new RqUser(req).canSee(siblings.get(0))) {
97              throw new RsForward(
98                  new RsFlash(
99                      "according to .rultor.yml, you're not allowed to see this",
100                     Level.WARNING
101                 )
102             );
103         }
104         return new RsPage(
105             "/xsl/siblings.xsl",
106             req,
107             new XeAppend("repo", repo),
108             new XeAppend("since", Long.toString(since.getTime())),
109             this.more(repo, siblings),
110             new XeDirectives(this.list(siblings))
111         );
112     }
113 
114     /**
115      * Link to more, if necessary.
116      * @param repo Repo name
117      * @param siblings Siblings
118      * @return Link or empty
119      * @throws IOException If fails
120      */
121     private XeSource more(final String repo, final List<Talk> siblings)
122         throws IOException {
123         final XeSource src;
124         if (siblings.size() == Tv.TWENTY) {
125             final Talk last = siblings.get(siblings.size() - 1);
126             src = new XeLink(
127                 "more",
128                 String.format(
129                     "/p/%s?s=%s",
130                     repo, last.updated().getTime()
131                 )
132             );
133         } else {
134             src = XeSource.EMPTY;
135         }
136         return src;
137     }
138 
139     /**
140      * List of directives.
141      * @param siblings List of them
142      * @return The directives
143      * @throws IOException If fails
144      */
145     private Iterable<Directive> list(final Iterable<Talk> siblings)
146         throws IOException {
147         final Directives dirs = new Directives().add("siblings");
148         for (final Talk talk : siblings) {
149             dirs.append(this.dirs(talk));
150         }
151         return dirs;
152     }
153 
154     /**
155      * Convert talk to directives.
156      * @param talk The talk to convert
157      * @return Directives
158      * @throws IOException If fails
159      */
160     private Iterable<Directive> dirs(final Talk talk) throws IOException {
161         final XML xml = talk.read();
162         final Directives dirs = new Directives().add("talk").add("archive");
163         for (final XML log : xml.nodes("/talk/archive/log")) {
164             dirs.append(TkSiblings.log(xml, log));
165         }
166         return dirs.up().add("name").set(talk.name()).up()
167             // @checkstyle MultipleStringLiteralsCheck (1 line)
168             .add("href").set(xml.xpath("/talk/wire/href/text()").get(0)).up()
169             .add("updated").set(Long.toString(talk.updated().getTime())).up()
170             .add("timeago").set(new PrettyTime().format(talk.updated())).up()
171             .up();
172     }
173 
174     /**
175      * Convert log to JAXB.
176      * @param talk Talk
177      * @param log The log to convert
178      * @return JAXB
179      */
180     private static Iterable<Directive> log(final XML talk, final XML log) {
181         final String hash = log.xpath("@id").get(0);
182         return new Directives().add("log")
183             .add("id").set(hash).up()
184             .add("href").set(new Home(talk, hash).uri().toString()).up()
185             .add("title").set(log.xpath("@title").get(0)).up()
186             .up();
187     }
