79  /**
80   * App.
81   *
82   * @author Yegor Bugayenko (yegor@tpc2.com)
83   * @version $Id$
84   * @since 1.50
85   * @checkstyle ClassDataAbstractionCouplingCheck (500 lines)
86   * @checkstyle ClassFanOutComplexityCheck (500 lines)

88  @SuppressWarnings({
89      "PMD.TooManyMethods", "PMD.ExcessiveMethodLength",
90      "PMD.ExcessiveImports"
91  })
92  public final class TsApp extends TsWrap {
93  
94      /**
95       * Revision of rultor.
96       */
97      private static final String REV = Manifests.read("Rultor-Revision");
98  
99      /**
100      * Ctor.
101      * @param talks Talks
102      * @param ticks Ticks
103      * @param toggles Toggles
104      */
105     public TsApp(final Talks talks, final Collection<Pulse.Tick> ticks,
106         final Toggles toggles) {
107         super(TsApp.make(talks, ticks, toggles));
108     }
109 
110     /**
111      * Ctor.
112      * @param talks Talks
113      * @param ticks Ticks
114      * @param toggles Toggles
115      * @return Takes
116      */
117     private static Takes make(final Talks talks,
118         final Collection<Pulse.Tick> ticks, final Toggles toggles) {
119         if (!"UTF-8".equals(Charset.defaultCharset().name())) {
120             throw new IllegalStateException(
121                 String.format(
122                     "default encoding is %s", Charset.defaultCharset()
123                 )
124             );
125         }
126         final Takes takes = TsApp.fallback(
127             new TsFlash(
128                 TsApp.auth(
129                     new TsForward(TsApp.regex(talks, ticks, toggles))
130                 )
131             )
132         );
133         return new TsWithHeaders(new TsVersioned(new TsMeasured(takes)))
134             .with("X-Rultor-Revision", TsApp.REV)
135             .with("Vary", "Cookie");
136     }
137 
138     /**
139      * Authenticated.
140      * @param takes Takes
141      * @return Authenticated takes
142      */
143     private static Takes fallback(final Takes takes) {
144         return new TsFallback(
145             takes,
146             new Fallback() {
147                 @Override
148                 public Take take(final RqFallback req) throws IOException {
149                     final String err = ExceptionUtils.getStackTrace(
150                         req.throwable()
151                     );
152                     return new TkFixed(
153                         new RsWithStatus(
154                             new RsWithType(
155                                 new RsVelocity(
156                                     this.getClass().getResource("error.html.vm")
157                                 ).with("err", err).with("rev", TsApp.REV),
158                                 "text/html"
159                             ),
160                             HttpURLConnection.HTTP_INTERNAL_ERROR
161                         )
162                     );
163                 }
164             }
165         );
166     }
167 
168     /**
169      * Authenticated.
170      * @param takes Takes
171      * @return Authenticated takes
172      */
173     private static Takes auth(final Takes takes) {
174         return new TsAuth(
175             takes,
176             new PsChain(
177                 new PsFake(
178                     Manifests.read("Rultor-DynamoKey").startsWith("AAAA")
179                 ),
180                 new PsByFlag(
181                     new PsByFlag.Pair(
182                         PsGithub.class.getSimpleName(),
183                         new PsGithub(
184                             Manifests.read("Rultor-GithubId"),
185                             Manifests.read("Rultor-GithubSecret")
186                         )
187                     ),
188                     new PsByFlag.Pair(
189                         PsLogout.class.getSimpleName(),
190                         new PsLogout()
191                     )
192                 ),
193                 new PsCookie(
194                     new CcSafe(
195                         new CcHex(
196                             new CcXOR(
197                                 new CcSalted(new CcCompact()),
198                                 Manifests.read("Rultor-SecurityKey")
199                             )
200                         )
201                     )
202                 )
203             )
204         );
205     }
206 
207     /**
208      * Regex takes.
209      * @param talks Talks
210      * @param ticks Ticks
211      * @param toggles Toggles
212      * @return Takes
213      */
214     private static Takes regex(final Talks talks,
215         final Collection<Pulse.Tick> ticks, final Toggles toggles) {
216         return new TsFork(
217             new FkParams(
218                 PsByFlag.class.getSimpleName(),
219                 Pattern.compile(".+"),
220                 new TkRedirect()
221             ),
222             new FkRegex("/robots.txt", ""),
223             new FkRegex("/svg", new TkSVG(ticks)),
224             new FkRegex("/s/.*", new TkRedirect()),
225             new FkRegex("/sitemap", new TkSitemap(talks)),
226             new FkRegex(
227                 "/xsl/.*",
228                 new TsWithType(new TsClasspath(), "text/xsl")
229             ),
230             new FkRegex(
231                 "/js/.*",
232                 new TsWithType(new TsClasspath(), "text/javascript")
233             ),
234             new FkRegex(
235                 "/css/.*",
236                 new TsWithType(new TsClasspath(), "text/css")
237             ),
238             new FkRegex(
239                 "/toggles/read-only",
240                 new Takes() {
241                     @Override
242                     public Take route(final Request req) throws IOException {
243                         return new TkAdminOnly(new TkToggles(toggles), req);
244                     }
245                 }
246             ),
247             new FkRegex(
248                 "/",
249                 new Target<RqRegex>() {
250                     @Override
251                     public Take route(final RqRegex req) {
252                         return new TkHome(talks, toggles, req);
253                     }
254                 }
255             ),
256             new FkRegex(
257                 "/b/([/a-zA-Z0-9_\\-\\.]+)",
258                 new Target<RqRegex>() {
259                     @Override
260                     public Take route(final RqRegex req) {
261                         return new TkButton(req.matcher().group(1));
262                     }
263                 }
264             ),
265             new FkRegex(
266                 "/t/([0-9]+)-([a-f0-9]+)",
267                 new Target<RqRegex>() {
268                     @Override
269                     public Take route(final RqRegex req) throws IOException {
270                         return new TkDaemon(
271                             req, talks,
272                             Long.parseLong(req.matcher().group(1)),
273                             req.matcher().group(2)
274                         );
275                     }
276                 }
277             ),
278             new FkRegex(
279                 "/p/([/a-zA-Z0-9_\\-\\.]+)",
280                 new Target<RqRegex>() {
281                     @Override
282                     public Take route(final RqRegex req) throws IOException {
283                         return new TkSiblings(
284                             talks, req.matcher().group(1), req
285                         );
286                     }
287                 }
288             ),
289             new FkRegex(
290                 "/t/([0-9]+)",
291                 new Target<RqRegex>() {
292                     @Override
293                     public Take route(final RqRegex req) throws IOException {
294                         return new TkAdminOnly(
295                             new TkTalk(
296                                 talks, req,
297                                 Long.parseLong(req.matcher().group(1))
298                             ),
299                             req
300                         );
301                     }
302                 }
303             ),
304             new FkRegex(
305                 "/t/([0-9]+)/kill",
306                 new Target<RqRegex>() {
307                     @Override
308                     public Take route(final RqRegex req) throws IOException {
309                         return new TkAdminOnly(
310                             new TkTalkKill(
311                                 talks, Long.parseLong(req.matcher().group(1))
312                             ),
313                             req
314                         );
315                     }
316                 }
317             ),
318             new FkRegex(
319                 "/t/([0-9]+)/delete",
320                 new Target<RqRegex>() {
321                     @Override
322                     public Take route(final RqRegex req) throws IOException {
323                         return new TkAdminOnly(
324                             new TkTalkDelete(
325                                 talks, Long.parseLong(req.matcher().group(1))
326                             ),
327                             req
328                         );
329                     }
330                 }
331             )
332         );
333     }
