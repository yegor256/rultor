60  /**
61   * Drain of a unit.
62   *
63   * @author Yegor Bugayenko (yegor@tpc2.com)
64   * @version $Id$

70  @Loggable(Loggable.DEBUG)

99       * Inject it from query.
100      * @param time Since what time

102     @QueryParam(DrainRs.QUERY_SINCE)

109     /**
110      * Get entrance page JAX-RS response.
111      * @return The JAX-RS response
112      */
113     @GET
114     @Path("/")
115     public Response index() {
116         EmptyPage page = new PageBuilder()
117             .stylesheet("/xsl/drain.xsl")
118             .build(EmptyPage.class)
119             .init(this)
120             .append(new JaxbBundle("unit", this.name));
121         final Drain drain = this.drain(new Time());
122         Pulses pulses = this.pulses(drain);
123         final int total;
124         if (this.since == null) {
125             total = Tv.FIVE;
126         } else {
127             try {
128                 pulses = pulses.tail(this.since);
129             } catch (IOException ex) {
130                 throw this.flash().redirect(
131                     this.uriInfo().getBaseUri(),
132                     String.format(
133                         "I/O problem with the tail of drain of \"%s\": %s",
134                         this.name,
135                         ExceptionUtils.getRootCauseMessage(ex)
136                     ),
137                     Level.SEVERE
138                 );
139             }
140             total = Tv.TWENTY;
141             page = page
142                 .append(new JaxbBundle("since", this.since.toString()))
143                 .link(
144                     new Link(
145                         "latest",
146                         this.uriInfo().getBaseUriBuilder()
147                             .clone()
148                             .path(DrainRs.class)
149                             .build(this.name)
150                     )
151                 );
152         }
153         return page
154             .append(this.pulses(pulses.iterator(), total))
155             .render()
156             .build();
157     }

159     /**
160      * Get drain.
161      * @param time Time where we should create a drain
162      * @return The drain
163      */
164     private Drain drain(final Time time) {
165         try {
166             return Drain.Source.class.cast(
167                 new Repo.Cached(
168                     this.repo(), this.user(), this.unit().spec()
169                 ).get().instantiate(
170                     this.users(),
171                     new Arguments(
172                         new Work.Simple(
173                             this.user().urn(), this.name,
174                             this.unit().spec(), time
175                         )
176                     )
177                 )
178             ).drain();
179         } catch (SpecException ex) {
180             throw this.flash().redirect(
181                 this.uriInfo().getBaseUri(),
182                 String.format(
183                     "Can't render drain of \"%s\": %s",
184                     this.name,
185                     ExceptionUtils.getRootCauseMessage(ex)
186                 ),
187                 Level.SEVERE
188             );
189         }
190     }

213     /**
214      * Get unit.
215      * @return The unit
216      */
217     private Unit unit() {
218         if (!this.user().units().contains(this.name)) {
219             throw this.flash().redirect(
220                 this.uriInfo().getBaseUri(),
221                 String.format("Unit `%s` doesn't exist", this.name),
222                 Level.SEVERE
223             );
224         }
225         return this.user().get(this.name);
226     }

228     /**
229      * All pulses of the unit.
230      * @param pulses All pulses to show
231      * @param maximum Maximum to show
232      * @return Collection of JAXB units
233      */
234     private JaxbBundle pulses(final Iterator<Time> pulses, final int maximum) {
235         JaxbBundle bundle = new JaxbBundle("pulses");
236         int pos;
237         for (pos = 0; pos < maximum; ++pos) {
238             if (!pulses.hasNext()) {
239                 break;
240             }
241             bundle = bundle.add(this.pulse(pulses.next()));
242         }
243         if (pos == maximum && pulses.hasNext()) {
244             bundle = bundle.link(
245                 new Link(
246                     "more",
247                     this.uriInfo()
248                         .getBaseUriBuilder()
249                         .clone()
250                         .path(DrainRs.class)
251                         .queryParam(
252                             DrainRs.QUERY_SINCE,
253                             pulses.next().millis()
254                         )
255                         .build(this.name)
256                 )
257             );
258         }
259         return bundle;
260     }

262     /**
263      * Convert pulse to JaxbBundle.
264      * @param date Date of it
265      * @return Bundle
266      */
267     private JaxbBundle pulse(final Time date) {
268         final Pulse pulse = new Pulse(this.drain(date));
269         final Collection<Stage> stages;
270         try {
271             stages = pulse.stages();
272         } catch (IOException ex) {
273             throw this.flash().redirect(
274                 this.uriInfo().getBaseUri(),
275                 String.format(
276                     "I/O problem with the stages of \"%s\": %s",
277                     date,
278                     ExceptionUtils.getRootCauseMessage(ex)
279                 ),
280                 Level.SEVERE
281             );
282         }
283         return new JaxbBundle("pulse")
284             .add("stages")
285             .add(
286                 new JaxbBundle.Group<Stage>(stages) {
287                     @Override
288                     public JaxbBundle bundle(final Stage stage) {
289                         return DrainRs.this.stage(date, stage);
290                     }
291                 }
292             )
293             .up()
294             .add("date", date.toString())
295             .up()
296             .add(
297                 "when",
298                 DurationFormatUtils.formatDurationWords(
299                     System.currentTimeMillis() - date.millis(), true, true
300                 )
301             )
302             .up()
303             .link(
304                 new Link(
305                     "see",
306                     this.uriInfo().getBaseUriBuilder()
307                         .clone()
308                         .path(PulseRs.class)
309                         .build(this.name, date.millis())
310                 )
311             )
312             .link(
313                 new Link(
314                     // @checkstyle MultipleStringLiterals (1 line)
315                     "stream",
316                     this.uriInfo().getBaseUriBuilder()
317                         .clone()
318                         .path(PulseRs.class)
319                         .path(PulseRs.class, "stream")
320                         .build(this.name, date.millis())
321                 )
322             );
323     }

325     /**
326      * Convert stage to JaxbBundle.
327      * @param date Date of it
328      * @param stage Stage to convert
329      * @return Bundle
330      */
331     private JaxbBundle stage(final Time date, final Stage stage) {
332         return new JaxbBundle("stage")
333             .add("result", stage.result().toString())
334             .up()
335             .add("start", Long.toString(stage.start()))
336             .up()
337             .add("stop", Long.toString(stage.stop()))
338             .up()
339             .add("msec", Long.toString(stage.stop() - stage.start()))
340             .up()
341             .add("output", new Markdown(stage.output()).html())
342             .up()
343             .link(
344                 new Link(
345                     "log",
346                     this.uriInfo().getBaseUriBuilder()
347                         .clone()
348                         .path(PulseRs.class)
349                         .queryParam(PulseRs.QUERY_START, stage.start())
350                         .queryParam(PulseRs.QUERY_STOP, stage.stop())
351                         .build(this.name, date.millis())
352                 )
353             );
354     }
