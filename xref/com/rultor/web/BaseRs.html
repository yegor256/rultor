99      private static final Provider TEST_PROVIDER = new Provider() {

101         public Identity identity() {

103             if ("12345".equals(Manifests.read("Rultor-Revision"))) {
104                 identity = new Identity.Simple(
105                     BaseRs.TEST_URN,
106                     "Local Host",
107                     URI.create("http://img.rultor.com/none.png")
108                 );
109             } else {
110                 identity = Identity.ANONYMOUS;

112             return identity;
113         }

163     /**

169     public final Inset insetFinances() {

173             public void render(final BasePage<?, ?> page,

177                         new Link(

183                         )
184                     );
185                     page.append(
186                         new JaxbBundle(
187                             "balance",
188                             BaseRs.this.balance().toString()
189                         )
190                     );
191                 }
192             }
193         };
194     }

262     /**
263      * Authentication inset.
264      * @return The inset
265      */
266     @Inset.Runtime
267     @NotNull(message = "auth inset can never be NULL")
268     public final AuthInset auth() {
269         // @checkstyle LineLength (4 lines)
270         return new AuthInset(this, Manifests.read("Rultor-SecurityKey"))
271             .with(new Facebook(this, Manifests.read("Rultor-FbId"), Manifests.read("Rultor-FbSecret")))
272             .with(new Github(this, Manifests.read("Rultor-GithubId"), Manifests.read("Rultor-GithubSecret")))
273             .with(new Google(this, Manifests.read("Rultor-GoogleId"), Manifests.read("Rultor-GoogleSecret")))
274             .with(new HttpBasic(this, BaseRs.KEYS))
275             .with(BaseRs.TEST_PROVIDER);
276     }
277 
278     /**
279      * Get currently logged in user.
280      * @return The user
281      */
282     @NotNull(message = "User can't be NULL")
283     protected final User user() {
284         final Identity self = this.auth().identity();
285         if (self.equals(Identity.ANONYMOUS)) {
286             throw this.flash().redirect(
287                 this.uriInfo().getBaseUri(),
288                 "please login first",
289                 Level.WARNING
290             );
291         }
292         return this.users().get(self.urn());
293     }

334     /**
335      * Get ACL of the stand.
336      * @param stand The stand
337      * @return ACL
338      */
339     protected final ACL acl(final Stand stand) {
340         ACL acl;
341         try {
342             acl = ACL.class.cast(
343                 this.repo().make(new User.Nobody(), stand.acl()).instantiate(
344                     this.users(),
345                     new Arguments(
346                         new Coordinates.None(), new Wallet.Empty()
347                     )
348                 )
349             );
350         } catch (final SpecException ex) {
351             Exceptions.warn(this, ex);
352             acl = new ACL() {
353                 @Override
354                 public boolean canView(final URN urn) {
355                     return false;
356                 }
357                 @Override
358                 public boolean canPost(final String key) {
359                     return false;
360                 }
361             };
362         }
363         return acl;
364     }
