34  import java.util.Collection;
35  import java.util.LinkedList;
36  import java.util.List;
37  import org.apache.commons.lang3.StringUtils;
38  import org.apache.commons.lang3.SystemUtils;
39  import org.apache.commons.lang3.exception.ExceptionUtils;

41  /**
42   * Exception logging utility class.
43   *
44   * @author Krzysztof Krason (Krzysztof.Krason@gmail.com)
45   * @version $Id$
46   * @since 1.0
47   */
48  public final class Exceptions {
49  
50      /**
51       * This is utility class.
52       */
53      private Exceptions() {
54          // intentionally empty
55      }
56  
57      /**
58       * Log exception in a compact syslog-friendly format with INFO log level.
59       * @param source Source of exception.
60       * @param exception Throwable to log.
61       */
62      public static void info(final Object source, final Throwable exception) {
63          Logger.info(source, ExceptionUtils.getRootCauseMessage(exception));
64      }
65  
66      /**
67       * Log exception in a compact syslog-friendly format with WARN log level.
68       * @param source Source of exception.
69       * @param exception Throwable to log.
70       */
71      public static void warn(final Object source, final Throwable exception) {
72          Logger.warn(source, ExceptionUtils.getRootCauseMessage(exception));
73      }
74  
75      /**
76       * Log exception in a compact syslog-friendly format with ERROR log level.
77       * @param source Source of exception.
78       * @param exception Throwable to log.
79       */
80      public static void error(final Object source, final Throwable exception) {
81          Logger.error(source, ExceptionUtils.getRootCauseMessage(exception));
82      }
83  
84      /**
85       * Get exception message by composing messages of chained exceptions.
86       * @param exception Throwable from which to retrieve messages.
87       * @return Composed message.
88       */
89      public static String message(final Throwable exception) {
90          Throwable cause = exception;
91          final List<String> messages = new ArrayList<String>();
92          while (cause != null) {
93              messages.add(ExceptionUtils.getMessage(cause));
94              cause = cause.getCause();
95          }
96          return StringUtils.join(messages, SystemUtils.LINE_SEPARATOR);
97      }
98  
99      /**
100      * Get clean version of exception stacktrace.
101      * @param exception Throwable for stacktrace cleanup.
102      * @return Clean one
103      */
104     public static String stacktrace(final Throwable exception) {
105         final Collection<String> lines = new LinkedList<String>();
106         final String[] markers = new String[] {
107             "org.aspectj.runtime",
108             "com.jcabi.aspects.aj",
109             "_aroundBody",
110             "AjcClosure",
111         };
112         for (String line : ExceptionUtils.getStackTrace(exception)
113             .split(SystemUtils.LINE_SEPARATOR)) {
114             boolean found = false;
115             for (String marker : markers) {
116                 if (line.contains(marker)) {
117                     found = true;
118                     break;
119                 }
120             }
121             if (!found) {
122                 lines.add(line);
123             }
124         }
125         return StringUtils.join(lines, SystemUtils.LINE_SEPARATOR);
126     }
