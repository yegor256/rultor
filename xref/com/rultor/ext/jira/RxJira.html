51  /**
52   * Jira with ReXSL.
53   *
54   * @author Yegor Bugayenko (yegor@tpc2.com)
55   * @version $Id$
56   * @since 1.0
57   * @see <a href="https://docs.atlassian.com/jira/REST/latest/">JIRA REST API</a>
58   */
59  @Immutable
60  @ToString
61  @EqualsAndHashCode(of = "url")
62  @Loggable(Loggable.DEBUG)
63  public final class RxJira implements Jira {
64  
65      /**
66       * URL of the server.
67       */
68      private final transient String url;
69  
70      /**
71       * Public ctor.
72       * @param srv Server URL
73       */
74      public RxJira(final String srv) {
75          this.url = srv;
76      }
77  
78      @Override
79      @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
80      public Iterable<JiraIssue> search(final String jql) {
81          final URI uri = UriBuilder.fromUri(this.url)
82              .path("/search")
83              .queryParam("jql", "{jql}")
84              .queryParam("fields", "")
85              .queryParam("expand", "")
86              .build(jql);
87          final JsonArray json;
88          try {
89              json = new JdkRequest(uri)
90                  .header(HttpHeaders.ACCEPT, MediaType.APPLICATION_JSON)
91                  .fetch()
92                  .as(RestResponse.class)
93                  .assertStatus(HttpURLConnection.HTTP_OK)
94                  .as(JsonResponse.class)
95                  .json()
96                  .readObject()
97                  .getJsonArray("issues");
98          } catch (IOException ex) {
99              throw new IllegalStateException(ex);
100         }
101         final Collection<JiraIssue> lst = new ArrayList<JiraIssue>(json.size());
102         for (JsonValue obj : json) {
103             lst.add(
104                 new RxJiraIssue(
105                     UriBuilder.fromUri(
106                         JsonObject.class.cast(obj).getString("self")
107                     ).userInfo(uri.getUserInfo()).build()
108                 )
109             );
110         }
111         return lst;
112     }
