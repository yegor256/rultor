113     /**
114      * Make XML out of it.
115      * @return The XML
116      * @throws XemblyException If can't apply
117      * @checkstyle RedundantThrows (3 lines)
118      */
119     public XML xml() throws XemblyException {
120         final Document dom;
121         try {
122             dom = new Xembler(this.directives).dom();
123         } catch (ImpossibleModificationException ex) {
124             throw new XemblyException(ex);
125         }
126         final InputStream xsl = this.getClass().getResourceAsStream(
127             "remove-duplicate-tags.xsl"
128         );
129         try {
130             return new XMLDocument(new DOMSource(new XSLT(dom, xsl).dom()));
131         } catch (TransformerException ex) {
132             throw new XemblyException(ex);
133         } finally {
134             IOUtils.closeQuietly(xsl);
135         }
136     }

138     /**
139      * Get all tags.
140      * @return Collection of tags found
141      * @throws XemblyException If fails
142      */
143     public Collection<Tag> tags() throws XemblyException {
144         final Collection<XML> nodes = this.xml()
145             .nodes("/snapshot/tags/tag");
146         final Collection<Tag> tags = new ArrayList<Tag>(nodes.size());
147         for (final XML node : nodes) {
148             tags.add(this.tag(node));
149         }
150         return Collections.unmodifiableCollection(tags);
151     }

153     /**
154      * Make tag from XML node.
155      * @param node The node
156      * @return Tag made
157      */
158     public Tag tag(final XML node) {
159         final Level level;
160         if (node.nodes("level").isEmpty()) {
161             level = Level.INFO;
162         } else {
163             level = Level.parse(node.xpath("level/text()").get(0));
164         }
165         final String markdown;
166         if (node.nodes("markdown[.!='']").isEmpty()) {
167             markdown = "";
168         } else {
169             markdown = node.xpath("markdown/text()").get(0);
170         }
171         final ImmutableMap.Builder<String, String> attrs =
172             new ImmutableMap.Builder<String, String>();
173         for (final XML attr : node.nodes("attributes/attribute")) {
174             attrs.put(
175                 attr.xpath("name/text()").get(0),
176                 attr.xpath("value/text()").get(0)
177             );
178         }
179         return new Tag.Simple(
180             node.xpath("label/text()").get(0),
181             level,
182             attrs.build(),
183             markdown
184         );
185     }

187     /**
188      * Fetch script from the stream.
189      * @param stream Input stream where to find details
190      * @return The script
191      * @throws IOException If IO problem inside
192      * @throws SyntaxException If broken syntax
193      * @checkstyle ThrowsCount (5 lines)
194      * @checkstyle RedundantThrows (4 lines)
195      */
196     private static String fetch(final InputStream stream)
197         throws IOException, SyntaxException {
198         final BufferedReader reader = new BufferedReader(
199             new InputStreamReader(stream, CharEncoding.UTF_8)
200         );
201         try {
202             final StringBuilder buf = new StringBuilder(0);
203             while (true) {
204                 final String line = reader.readLine();
205                 if (line == null) {
206                     break;
207                 }
208                 if (XemblyLine.existsIn(line)) {
209                     buf.append(XemblyLine.parse(line).xembly());
210                 }
211             }
212             if (buf.length() == 0) {
213                 buf.append("XPATH '/snapshot';");
214             }
215             return buf.toString();
216         } finally {
217             reader.close();
218         }
219     }
