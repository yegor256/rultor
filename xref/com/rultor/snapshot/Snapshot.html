115     /**
116      * Make XML out of it.

118      * @throws ImpossibleModificationException If can't apply

121     public XmlDocument xml() throws ImpossibleModificationException {
122         final Document dom = new Xembler(this.directives).dom("snapshot");
123         try {
124             return new SimpleXml(
125                 new DOMSource(
126                     new XSLT(
127                         dom,

131                     ).dom()
132                 )
133             );
134         } catch (TransformerException ex) {
135             throw new ImpossibleModificationException(
136                 "Failed to transform",
137                 ex
138             );
139         }
140     }

142     /**
143      * Get all tags.
144      * @return Collection of tags found
145      * @throws ImpossibleModificationException If can't parse it
146      * @checkstyle RedundantThrows (3 lines)
147      */
148     public Collection<Tag> tags() throws ImpossibleModificationException {
149         final Collection<XmlDocument> nodes = this.xml()
150             .nodes("/snapshot/tags/tag");
151         final Collection<Tag> tags = new ArrayList<Tag>(nodes.size());
152         for (XmlDocument node : nodes) {
153             tags.add(this.tag(node));
154         }
155         return Collections.unmodifiableCollection(tags);
156     }

158     /**
159      * Make tag from XML node.
160      * @param node The node
161      * @return Tag made
162      */
163     public Tag tag(final XmlDocument node) {
164         final Level level;
165         if (node.nodes("level").isEmpty()) {
166             level = Level.INFO;
167         } else {
168             level = Level.parse(node.xpath("level/text()").get(0));
169         }
170         final String markdown;
171         if (node.nodes("markdown[.!='']").isEmpty()) {
172             markdown = "";
173         } else {
174             markdown = node.xpath("markdown/text()").get(0);
175         }
176         final ImmutableMap.Builder<String, String> attrs =
177             new ImmutableMap.Builder<String, String>();
178         for (XmlDocument attr : node.nodes("attributes/attribute")) {
179             attrs.put(
180                 attr.xpath("name/text()").get(0),
181                 attr.xpath("value/text()").get(0)
182             );
183         }
184         return new Tag.Simple(
185             node.xpath("label/text()").get(0),
186             level,
187             attrs.build(),
188             markdown
189         );
190     }

192     /**
193      * Fetch script from the stream.
194      * @param stream Input stream where to find details
195      * @return The script
196      * @throws IOException If IO problem inside
197      * @throws XemblySyntaxException If broken syntax
198      * @checkstyle ThrowsCount (5 lines)
199      * @checkstyle RedundantThrows (4 lines)
200      */
201     private static String fetch(final InputStream stream)
202         throws IOException, XemblySyntaxException {
203         final BufferedReader reader = new BufferedReader(
204             new InputStreamReader(stream, CharEncoding.UTF_8)
205         );
206         final StringBuilder buf = new StringBuilder();
207         while (true) {
208             final String line = reader.readLine();
209             if (line == null) {
210                 break;
211             }
212             if (XemblyLine.existsIn(line)) {
213                 buf.append(XemblyLine.parse(line).xembly());
214             }
215         }
216         if (buf.length() == 0) {
217             buf.append("XPATH '/snapshot';");
218         }
219         return buf.toString();
220     }
